## 实践 1：实现简单的纹理上传管理器

### 完整代码

```cpp
// SimpleTextureManager.h
// 模拟 KrKr2 的纹理上传逻辑，可独立编译运行

#ifndef SIMPLE_TEXTURE_MANAGER_H
#define SIMPLE_TEXTURE_MANAGER_H

#include <GLES2/gl2.h>
#include <GLES2/gl2ext.h>
#include <cstring>
#include <cstdint>
#include <string>
#include <set>
#include <sstream>

// 检查 GL 错误的辅助宏
#define CHECK_GL_ERROR() do { \
    GLenum err = glGetError(); \
    if(err != GL_NO_ERROR) { \
        printf("GL Error 0x%04X at %s:%d\n", err, __FILE__, __LINE__); \
    } \
} while(0)

// 纹理格式枚举（模拟 KrKr2 的 TVPTextureFormat）
enum class TextureFormat : uint32_t {
    Gray = 0x11,    // 1 字节/像素
    RGB  = 0x23,    // 3 字节/像素
    RGBA = 0x44     // 4 字节/像素
};

// 获取像素大小
inline int getPixelSize(TextureFormat fmt) {
    return static_cast<int>(fmt) & 0xF;
}

// 扩展检测类
class GLExtensionChecker {
    static std::set<std::string> extensionSet;
    static bool initialized;
    
public:
    static bool hasExtension(const char* name) {
        if(!initialized) {
            const char* ext = (const char*)glGetString(GL_EXTENSIONS);
            if(ext) {
                std::istringstream stream(ext);
                std::string token;
                while(stream >> token) {
                    extensionSet.insert(token);
                }
            }
            initialized = true;
        }
        return extensionSet.count(name) > 0;
    }
    
    static bool hasUnpackSubimage() {
        return hasExtension("GL_EXT_unpack_subimage");
    }
};

// 静态成员初始化
std::set<std::string> GLExtensionChecker::extensionSet;
bool GLExtensionChecker::initialized = false;

// 纹理管理类
class SimpleTexture {
    GLuint textureId = 0;
    int width = 0, height = 0;
    int internalW = 0, internalH = 0;
    TextureFormat format = TextureFormat::RGBA;
    
public:
    SimpleTexture() = default;
    
    ~SimpleTexture() {
        if(textureId) {
            glDeleteTextures(1, &textureId);
            textureId = 0;
        }
    }
    
    // 禁止拷贝
    SimpleTexture(const SimpleTexture&) = delete;
    SimpleTexture& operator=(const SimpleTexture&) = delete;
    
    // 允许移动
    SimpleTexture(SimpleTexture&& other) noexcept 
        : textureId(other.textureId), width(other.width), height(other.height),
          internalW(other.internalW), internalH(other.internalH), format(other.format) {
        other.textureId = 0;
    }
    
    // 初始化纹理（对应 InternalInit）
    bool init(int w, int h, TextureFormat fmt, const void* pixels, int pitch) {
        width = w;
        height = h;
        format = fmt;
        internalW = w;
        internalH = h;
        
        // 生成纹理对象
        glGenTextures(1, &textureId);
        glBindTexture(GL_TEXTURE_2D, textureId);
        
        // 设置纹理参数
        glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
        glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
        glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
        glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
        
        // 确定像素格式
        GLenum pixfmt = GL_RGBA;
        GLenum internalfmt = GL_RGBA;
        switch(format) {
            case TextureFormat::Gray:
                pixfmt = GL_LUMINANCE;
                internalfmt = GL_LUMINANCE;
                break;
            case TextureFormat::RGB:
                pixfmt = GL_RGB;
                internalfmt = GL_RGB;
                break;
            case TextureFormat::RGBA:
                pixfmt = GL_RGBA;
                internalfmt = GL_RGBA;
                break;
        }
        
        // 设置对齐（参考 KrKr2 实现）
        switch(pitch & 7) {
            case 0: glPixelStorei(GL_UNPACK_ALIGNMENT, 8); break;
            case 4: glPixelStorei(GL_UNPACK_ALIGNMENT, 4); break;
            case 2: case 6: glPixelStorei(GL_UNPACK_ALIGNMENT, 2); break;
            default: glPixelStorei(GL_UNPACK_ALIGNMENT, 1); break;
        }
        
        // 上传纹理数据
        glTexImage2D(GL_TEXTURE_2D, 0, internalfmt, w, h, 0, 
                     pixfmt, GL_UNSIGNED_BYTE, pixels);
        
        CHECK_GL_ERROR();
        return true;
    }
    
    // 更新纹理（对应 InternalUpdate）
    void update(const void* pixels, int pitch, int x, int y, int w, int h) {
        if(!textureId) return;
        
        glBindTexture(GL_TEXTURE_2D, textureId);
        
        int pixelSize = getPixelSize(format);
        GLenum pixfmt = GL_RGBA;
        
        switch(format) {
            case TextureFormat::Gray:
                pixfmt = GL_LUMINANCE;
                glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
                break;
            case TextureFormat::RGB:
                pixfmt = GL_RGB;
                glPixelStorei(GL_UNPACK_ALIGNMENT, 4);
                break;
            case TextureFormat::RGBA:
                pixfmt = GL_RGBA;
                glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
                break;
        }
        
        const uint8_t* src = static_cast<const uint8_t*>(pixels);
        bool needsRearrange = false;
        uint8_t* tempBuffer = nullptr;
        
        // GL_UNPACK_ROW_LENGTH 扩展处理
        if(GLExtensionChecker::hasUnpackSubimage()) {
            // 高效路径：使用扩展
            glPixelStorei(GL_UNPACK_ROW_LENGTH, pitch / pixelSize);
        } else if(pitch != w * pixelSize) {
            // 兼容路径：手动重排数据
            needsRearrange = true;
            int rowBytes = w * pixelSize;
            tempBuffer = new uint8_t[rowBytes * h];
            uint8_t* dst = tempBuffer;
            
            for(int row = 0; row < h; ++row) {
                memcpy(dst, src, rowBytes);
                src += pitch;
                dst += rowBytes;
            }
            src = tempBuffer;
        }
        
        // 执行局部更新
        glTexSubImage2D(GL_TEXTURE_2D, 0, x, y, w, h, 
                        pixfmt, GL_UNSIGNED_BYTE, src);
        
        // 清理
        if(GLExtensionChecker::hasUnpackSubimage()) {
            glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
        }
        if(tempBuffer) {
            delete[] tempBuffer;
        }
        
        CHECK_GL_ERROR();
    }
    
    GLuint getId() const { return textureId; }
    int getWidth() const { return width; }
    int getHeight() const { return height; }
};

#endif // SIMPLE_TEXTURE_MANAGER_H
```

### 使用示例

```cpp
// main.cpp - 使用纹理管理器
#include "SimpleTextureManager.h"
#include <vector>

int main() {
    // 假设 OpenGL 上下文已初始化
    
    // 创建 256x256 RGBA 纹理
    std::vector<uint8_t> pixels(256 * 256 * 4, 128);  // 灰色填充
    
    SimpleTexture tex;
    tex.init(256, 256, TextureFormat::RGBA, pixels.data(), 256 * 4);
    
    printf("Created texture ID: %u\n", tex.getId());
    
    // 更新 64x64 区域（左上角）
    std::vector<uint8_t> updateData(64 * 64 * 4, 255);  // 白色
    tex.update(updateData.data(), 64 * 4, 0, 0, 64, 64);
    
    printf("Texture updated successfully\n");
    
    return 0;
}
```

## 实践 2：FBO 渲染目标管理器

```cpp
// FBOManager.h
// 模拟 KrKr2 的 FBO 管理逻辑

#ifndef FBO_MANAGER_H
#define FBO_MANAGER_H

#include <GLES2/gl2.h>

class FBOManager {
    GLuint fbo = 0;
    GLint savedFramebuffer = 0;
    GLuint currentTarget = 0;
    bool isActive = false;
    
public:
    FBOManager() = default;
    
    ~FBOManager() {
        if(fbo) {
            glDeleteFramebuffers(1, &fbo);
            fbo = 0;
        }
    }
    
    // 初始化 FBO
    bool init() {
        glGenFramebuffers(1, &fbo);
        if(fbo == 0) {
            return false;
        }
        return true;
    }
    
    // 设置渲染目标（对应 TVPSetRenderTarget）
    void setRenderTarget(GLuint textureId) {
        if(textureId != 0) {
            // 切换到离屏渲染
            if(!isActive) {
                // 保存当前帧缓冲
                glGetIntegerv(GL_FRAMEBUFFER_BINDING, &savedFramebuffer);
                glBindFramebuffer(GL_FRAMEBUFFER, fbo);
                isActive = true;
            }
            
            // 避免重复绑定相同目标
            if(currentTarget == textureId) {
                return;
            }
            
            // 附加纹理作为颜色附件
            glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0,
                                   GL_TEXTURE_2D, textureId, 0);
            
            // 验证帧缓冲完整性
            GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
            if(status != GL_FRAMEBUFFER_COMPLETE) {
                printf("FBO incomplete: 0x%04X\n", status);
            }
            
            currentTarget = textureId;
        } else {
            // 恢复屏幕渲染
            glBindFramebuffer(GL_FRAMEBUFFER, savedFramebuffer);
            isActive = false;
            currentTarget = 0;
        }
    }
    
    // 获取当前渲染目标
    GLuint getCurrentTarget() const {
        return currentTarget;
    }
    
    // 是否正在离屏渲染
    bool isOffscreen() const {
        return isActive;
    }
};

#endif // FBO_MANAGER_H
```

## 常见错误与解决方案

### 错误 1：纹理出现斜纹或错位

**症状**：上传的纹理显示时出现明显的倾斜或像素错位。

**原因**：`GL_UNPACK_ALIGNMENT` 设置与源数据不匹配。

**解决方案**：

```cpp
// 根据 pitch 动态设置对齐
int pitch = imageWidth * bytesPerPixel;
if(pitch % 8 == 0) {
    glPixelStorei(GL_UNPACK_ALIGNMENT, 8);
} else if(pitch % 4 == 0) {
    glPixelStorei(GL_UNPACK_ALIGNMENT, 4);
} else if(pitch % 2 == 0) {
    glPixelStorei(GL_UNPACK_ALIGNMENT, 2);
} else {
    glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
}
```

### 错误 2：FBO 状态不完整

**症状**：`glCheckFramebufferStatus` 返回非 `GL_FRAMEBUFFER_COMPLETE`。

**常见错误码**：

| 错误码 | 含义 | 解决方案 |
|--------|------|----------|
| `GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT` | 附件无效 | 检查纹理是否正确创建 |
| `GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT` | 缺少附件 | 确保调用了 `glFramebufferTexture2D` |
| `GL_FRAMEBUFFER_UNSUPPORTED` | 格式不支持 | 更换纹理格式 |

### 错误 3：渲染到纹理后内容为空

**症状**：使用 FBO 渲染后，纹理采样结果为黑色或未定义。

**原因**：
1. 忘记恢复渲染目标
2. 视口设置不正确

**解决方案**：

```cpp
// 正确的 FBO 使用流程
fboManager.setRenderTarget(targetTexture);

// 设置视口匹配纹理尺寸
glViewport(0, 0, textureWidth, textureHeight);

// 执行渲染...
glDrawArrays(GL_TRIANGLES, 0, 6);

// 恢复到屏幕
fboManager.setRenderTarget(0);

// 恢复原视口
glViewport(0, 0, screenWidth, screenHeight);
```

## 性能优化技巧

### 1. 减少纹理绑定切换

```cpp
// 低效：每次绘制都绑定
for(auto& sprite : sprites) {
    glBindTexture(GL_TEXTURE_2D, sprite.texture);  // 频繁切换
    drawSprite(sprite);
}

// 高效：按纹理分组
std::sort(sprites.begin(), sprites.end(), 
          [](auto& a, auto& b) { return a.texture < b.texture; });
GLuint lastTex = 0;
for(auto& sprite : sprites) {
    if(sprite.texture != lastTex) {
        glBindTexture(GL_TEXTURE_2D, sprite.texture);
        lastTex = sprite.texture;
    }
    drawSprite(sprite);
}
```

### 2. 使用纹理池

```cpp
// 纹理池：避免频繁创建/销毁
class TexturePool {
    std::vector<GLuint> available;
    std::unordered_map<GLuint, std::pair<int,int>> textureSize;
    
public:
    GLuint acquire(int w, int h) {
        // 尝试复用相同大小的纹理
        for(auto it = available.begin(); it != available.end(); ++it) {
            auto& size = textureSize[*it];
            if(size.first == w && size.second == h) {
                GLuint tex = *it;
                available.erase(it);
                return tex;
            }
        }
        // 创建新纹理
        GLuint tex;
        glGenTextures(1, &tex);
        textureSize[tex] = {w, h};
        return tex;
    }
    
    void release(GLuint tex) {
        available.push_back(tex);
    }
};
```

### 3. 批量更新优化

```cpp
// 低效：多次小更新
for(int i = 0; i < 10; ++i) {
    texture.update(data[i], pitch, x[i], y[i], 32, 32);
}

// 高效：合并为一次大更新（如果区域连续）
// 或使用 PBO（Pixel Buffer Object）异步上传
```

### 题目 2：实现 FBO 深度附件

**问题**：扩展 `FBOManager`，支持创建带深度缓冲的 FBO。

<details>
<summary>查看答案</summary>

```cpp
class FBOManagerWithDepth {
    GLuint fbo = 0;
    GLuint depthRbo = 0;    // 深度渲染缓冲
    GLint savedFramebuffer = 0;
    GLuint currentTarget = 0;
    bool isActive = false;
    int depthWidth = 0, depthHeight = 0;
    
public:
    bool init() {
        glGenFramebuffers(1, &fbo);
        return fbo != 0;
    }
    
    void setRenderTargetWithDepth(GLuint textureId, int width, int height) {
        if(textureId != 0) {
            if(!isActive) {
                glGetIntegerv(GL_FRAMEBUFFER_BINDING, &savedFramebuffer);
                glBindFramebuffer(GL_FRAMEBUFFER, fbo);
                isActive = true;
            }
            
            if(currentTarget == textureId) return;
            
            // 附加颜色纹理
            glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0,
                                   GL_TEXTURE_2D, textureId, 0);
            
            // 创建或更新深度缓冲
            if(depthWidth != width || depthHeight != height) {
                if(depthRbo) {
                    glDeleteRenderbuffers(1, &depthRbo);
                }
                glGenRenderbuffers(1, &depthRbo);
                glBindRenderbuffer(GL_RENDERBUFFER, depthRbo);
                glRenderbufferStorage(GL_RENDERBUFFER, GL_DEPTH_COMPONENT16,
                                     width, height);
                glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT,
                                         GL_RENDERBUFFER, depthRbo);
                depthWidth = width;
                depthHeight = height;
            }
            
            // 验证
            GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
            if(status != GL_FRAMEBUFFER_COMPLETE) {
                printf("FBO with depth incomplete: 0x%04X\n", status);
            }
            
            currentTarget = textureId;
        } else {
            glBindFramebuffer(GL_FRAMEBUFFER, savedFramebuffer);
            isActive = false;
            currentTarget = 0;
        }
    }
    
    ~FBOManagerWithDepth() {
        if(depthRbo) glDeleteRenderbuffers(1, &depthRbo);
        if(fbo) glDeleteFramebuffers(1, &fbo);
    }
};
```
