## 1. IDA 类型库机制概述

IDA 的反编译质量很大程度上取决于它能否识别函数参数和局部变量的类型。类型信息的来源有三种：

### 1.1 三种类型信息来源

```
来源优先级（高 → 低）：
  1. 数据库内置定义（.i64 中直接记录的 tinfo_t）   ← 用 SetType/apply_tinfo 写入
  2. 本地类型（Local Types，数据库内的用户自定义）   ← 用 parse_decl 或 idc_parse_types 写入
  3. 系统类型库（.til 文件，如 mssdk.til、vc.til）   ← IDA 安装时自带，Win32 API 类型从这里来
```

当 IDA 反编译 `SendMessageW(*(HWND *)(a1 + 4), ...)` 时，它能正确推断 `SendMessageW` 的参数类型，是因为 `mssdk.til` 类型库里有 `SendMessageW` 的原型。

我们的目标是把 `WindowEx`、`MenuItemEx` 等自定义类型也注册进去，让 IDA 在反编译时能推断出 `WindowEx*` 参数。

### 1.2 Local Types 视图

在 IDA 中，`View → Open Subviews → Local Types`（快捷键 `Shift+F1`）可以查看当前数据库中已注册的所有本地类型。我们用 `idc_parse_types` 写入的结构体定义就会出现在这里。

## 2. 用 idc_parse_types 从 C 代码注册类型

`ida_typeinf.idc_parse_types(input_str, flags)` 接受一段 C 声明代码字符串，解析其中的结构体、枚举、typedef 等定义，并注册到当前数据库的本地类型中。

### 代码示例 1：将 WindowEx 头文件注册到 IDA

```python
# stuffs/ida_scripts/register_types.py
# 用途：将 windowEx.dll 相关类型定义注册到 IDA 本地类型库
# 运行：idat64.exe -A -S"register_types.py" windowEx.dll.i64

import ida_typeinf
import idc
import ida_loader

# 从 cpp/plugins/windowEx.cpp 和逆向分析中还原的 windowEx 类型定义
# 这段代码会被 IDA 解析为 C 类型声明
WINDOWEX_HEADER = r"""
/* ============================================================
   windowEx.dll 还原头文件
   基于：
   - MSVC RTTI 字符串（类名）
   - 反汇编代码偏移访问（成员布局）
   - windowEx.cpp 跨平台适配版源码（成员语义）
   - ncbind.hpp 模板框架（注册接口）
   ============================================================ */

/* Windows 基本类型（若 IDA 未从 mssdk.til 导入则手动定义） */
typedef unsigned int  UINT;
typedef int           INT;
typedef long          LONG;
typedef unsigned long DWORD;
typedef void*         HWND;
typedef void*         HMENU;
typedef void*         HINSTANCE;
typedef void*         WNDPROC;
typedef int           BOOL;
typedef unsigned int  WPARAM;
typedef long          LPARAM;
typedef long          LRESULT;

/* TJS2 相关类型（来自 tp_stub.h） */
typedef int           tjs_error;   /* TJS2 函数返回码，0=成功 */
typedef int           tjs_int;     /* TJS2 整数类型 */
typedef void*         tTJSVariant; /* TJS2 变体类型（简化为 void*，实际是 union） */
typedef void*         iTJSDispatch2; /* TJS2 脚本对象接口 */

/* OverlayBitmap：窗口上的透明位图叠加子窗口
   注：此类用于在游戏窗口上绘制自定义图形（如标题栏图像）
   偏移来自对 OverlayBitmap 相关函数的反汇编分析 */
struct OverlayBitmap {
    void*     _vftable;   /* +0:  虚函数表指针 */
    HWND      hwnd;       /* +4:  叠加层窗口句柄 */
    HWND      parentHwnd; /* +8:  父窗口句柄 */
    int       width;      /* +12: 位图宽度（像素） */
    int       height;     /* +16: 位图高度（像素） */
    void*     hBitmap;    /* +20: GDI 位图句柄（HBITMAP） */
    iTJSDispatch2* tjsObj;/* +24: 对应的 TJS2 脚本对象 */
};

/* MenuItemEx：扩展菜单项，支持弹出菜单和菜单消息事件
   偏移来自 sub_10004690（析构函数，33684B 处）的反汇编分析 */
struct MenuItemEx {
    void*     _vftable;   /* +0:  虚函数表指针 */
    HMENU     hMenu;      /* +4:  Win32 菜单句柄 */
    HWND      parentHwnd; /* +8:  父窗口句柄（用于 TrackPopupMenu） */
    UINT      itemId;     /* +12: 菜单项标识符（0 = 根菜单） */
    void*     caption;    /* +16: 宽字符标题字符串指针（wchar_t*） */
    BOOL      isPopup;    /* +20: 是否为弹出菜单 */
    iTJSDispatch2* tjsObj;/* +24: 对应的 TJS2 脚本对象 */
};

/* WindowEx：主类，代表一个可接收 TJS2 脚本事件的扩展 Win32 窗口
   偏移来自 sub_10005970（onMessage，最大函数 33684B）的反汇编分析
   这是 windowEx 插件的核心数据结构 */
struct WindowEx {
    void*          _vftable;     /* +0:  虚函数表指针 */
    HWND           hwnd;         /* +4:  被管理的 Win32 窗口句柄 */
    HMENU          hMenu;        /* +8:  窗口菜单句柄（可空） */
    HINSTANCE      hInst;        /* +12: 模块实例句柄 */
    WNDPROC        originalProc; /* +16: 子类化前的原始窗口过程 */
    int            posX;         /* +20: 窗口当前 X 位置 */
    int            posY;         /* +24: 窗口当前 Y 位置 */
    DWORD          flags;        /* +28: 状态标志位（bit0=处于消息中, bit1=已子类化...）*/
    iTJSDispatch2* tjsObj;       /* +32: 关联的 TJS2 窗口脚本对象 */
};

/* V2Link / V2Unlink：KiriKiri 插件标准入口函数签名
   每个 ncbind 插件 DLL 都必须导出这两个函数 */
typedef void (__cdecl *V2LinkFunc)(void *exporter);
typedef void (__cdecl *V2UnlinkFunc)(void);
"""

print("=== 向 IDA 本地类型库注册 windowEx 类型定义 ===\n")

# idc_parse_types(input_str, flags)
# flags = PT_FILE（0x0001）：从文件读取；PT_REPLACE（0x0004）：替换已有类型
# 使用 PT_REPLACE 确保重跑脚本时覆盖旧版本
flags = ida_typeinf.PT_REPLACE
result = ida_typeinf.idc_parse_types(WINDOWEX_HEADER, flags)

if result == 0:
    print("✓ 类型定义成功注册到 IDA 本地类型库")
    print("  可在 View → Open Subviews → Local Types 中查看")
else:
    print(f"✗ 注册失败（错误代码 {result}）")
    print("  常见原因：类型语法错误、循环引用、已有同名类型（用 PT_REPLACE 可覆盖）")

# 验证：检查 WindowEx 类型是否已注册
tif = ida_typeinf.tinfo_t()
if tif.get_named_type(None, "WindowEx"):
    print(f"\n✓ 验证通过：WindowEx 类型已在类型库中（大小 = {tif.get_size()} 字节）")
else:
    print("\n✗ 验证失败：WindowEx 类型未能在类型库中找到")

ida_loader.save_database("", 0)
print("数据库已保存。")
idc.qexit(0)
```

## 3. 生成可导入 IDA 的头文件

除了用 Python 脚本直接注册类型，另一种方式是生成标准 C 头文件（`.h`），然后用 IDA 的 `tilib` 工具将其编译为 `.til` 类型库文件。

### 3.1 tilib 工具使用方法

```bash
# IDA 自带的 tilib 工具（在 IDA 安装目录下）
# Windows 路径示例（IDA Pro 7.6）:
# C:\Program Files\IDA Pro 7.6\tilib64.exe

# 将 .h 文件编译为 .til 类型库
tilib64.exe -c -h windowex_types.h -t windowex windowex.til

# 参数说明：
# -c        创建新的 .til 文件
# -h        指定输入头文件
# -t NAME   设置类型库标题（显示在 IDA 的 "Type Libraries" 列表中）
# windowex.til  输出文件名

# 在 IDA 中导入（运行后在 View → Type Libraries 中可见）：
# idc.add_til("windowex", 2)  # 2 = 在已有库后面追加
```

### 代码示例 2：生成 windowEx 头文件（用于导入 IDA 或作为接口文档）

```python
# stuffs/ida_scripts/generate_header.py
# 用途：从 IDA 数据库中提取已还原的信息，生成 windowEx_restored.h 头文件
# 这个头文件既可以导入 IDA，也可以作为跨平台重新实现的接口文档
# 运行：idat64.exe -A -S"generate_header.py" windowEx.dll.i64

import idc
import idautils
import ida_struct
import ida_typeinf
import ida_funcs
import os

OUTPUT_FILE = os.path.join(os.path.dirname(__file__), "windowEx_restored.h")

def get_struct_members(struct_name):
    """提取结构体成员列表，返回 [(偏移, 名称, 大小)] 列表"""
    sid = idc.get_struc_id(struct_name)
    if sid == idc.BADADDR:
        return []

    members = []
    struc = ida_struct.get_struc(sid)
    member = ida_struct.get_first_member(struc)
    while member:
        offset = member.soff
        name = member.name
        size = idc.get_member_size(sid, offset)
        cmt = idc.get_member_cmt(sid, offset, 1) or ""
        members.append((offset, name, size, cmt))
        member = ida_struct.get_next_member(struc, offset)
    return members

def size_to_ctype(size):
    """将成员大小映射到 C 类型"""
    return {1: "unsigned char", 2: "unsigned short",
            4: "unsigned int", 8: "unsigned long long"}.get(size, f"char[{size}]")

def generate_struct_decl(struct_name):
    """生成结构体的 C 声明字符串"""
    members = get_struct_members(struct_name)
    if not members:
        return f"/* {struct_name}: 结构体未定义或为空 */\n"

    total_size = idc.get_struc_size(idc.get_struc_id(struct_name))
    lines = [f"/* 总大小: {total_size} 字节 */",
             f"struct {struct_name} {{"]
    for offset, name, size, cmt in members:
        ctype = size_to_ctype(size)
        cmt_str = f"  /* +{offset}: {cmt} */" if cmt else f"  /* +{offset} */"
        lines.append(f"    {ctype:<20} {name};{cmt_str}")
    lines.append("};")
    return "\n".join(lines) + "\n"

def get_known_funcs():
    """获取数据库中所有已命名（非 sub_*）的函数"""
    funcs = []
    for func_ea in idautils.Functions():
        name = idc.get_func_name(func_ea)
        if not name.startswith("sub_") and not name.startswith("nullsub_"):
            size = idc.get_func_attr(func_ea, idc.FUNCATTR_SIZE)
            type_str = idc.get_type(func_ea) or ""
            funcs.append((func_ea, name, size, type_str))
    return sorted(funcs, key=lambda x: x[0])

print(f"=== 生成 windowEx 还原头文件 → {OUTPUT_FILE} ===\n")

lines = []
lines.append("/* ============================================================")
lines.append("   windowEx_restored.h")
lines.append("   自动生成：通过 IDAPython 从 windowEx.dll.i64 提取")
lines.append("   生成时间：由 generate_header.py 生成")
lines.append("")
lines.append("   用途：")
lines.append("     1. 导入 IDA 改善反编译质量（tilib 编译为 .til 后导入）")
lines.append("     2. 作为跨平台重新实现的接口参考文档")
lines.append("     3. 对照 cpp/plugins/windowEx.cpp 验证还原准确性")
lines.append("   ============================================================ */")
lines.append("")
lines.append("#pragma once")
lines.append("#ifndef WINDOWEX_RESTORED_H")
lines.append("#define WINDOWEX_RESTORED_H")
lines.append("")
lines.append("/* ── 基础类型 ─────────────────────────────────────────────── */")
lines.append("typedef unsigned int    UINT;")
lines.append("typedef unsigned long   DWORD;")
lines.append("typedef void*           HWND;")
lines.append("typedef void*           HMENU;")
lines.append("typedef void*           HINSTANCE;")
lines.append("typedef int             tjs_error;  /* TJS2 返回码 */")
lines.append("typedef void*           iTJSDispatch2; /* TJS2 对象接口 */")
lines.append("")
lines.append("/* ── 还原的结构体定义 ──────────────────────────────────────── */")

for struct_name in ("OverlayBitmap", "MenuItemEx", "WindowEx"):
    lines.append("")
    lines.append(generate_struct_decl(struct_name))

lines.append("")
lines.append("/* ── 还原的函数声明 ────────────────────────────────────────── */")
lines.append("/* 格式：地址  函数名  [已知签名] */")
lines.append("")
for ea, name, size, type_str in get_known_funcs():
    if type_str:
        lines.append(f"/* 0x{ea:08X} ({size}B) */")
        lines.append(f"{type_str};")
        lines.append("")
    else:
        lines.append(f"/* 0x{ea:08X} ({size}B) void {name}(); */")

lines.append("")
lines.append("/* ── 导出函数（DLL 入口点）──────────────────────────────────── */")
lines.append("/* V2Link:   0x1001A580 — 插件初始化，引擎加载时调用 */")
lines.append("/* V2Unlink: 0x1001A5A0 — 插件卸载，引擎退出时调用 */")
lines.append("void V2Link(void *exporter);")
lines.append("void V2Unlink(void);")
lines.append("")
lines.append("#endif /* WINDOWEX_RESTORED_H */")

header_content = "\n".join(lines)

with open(OUTPUT_FILE, 'w', encoding='utf-8') as f:
    f.write(header_content)

print(f"✓ 已写入 {len(lines)} 行到 {OUTPUT_FILE}")
print("\n使用方式:")
print("  1. 导入 IDA：tilib64.exe -c -h windowEx_restored.h -t windowex windowex.til")
print("     然后在 IDA 中：View → Type Libraries → Insert → 选择 windowex.til")
print("  2. 作为文档：对照 cpp/plugins/windowEx.cpp 验证结构体布局正确性")

idc.qexit(0)
```

## 4. 从 ncbind 框架推断更多类型信息

ncbind 框架的模板实例化留下了丰富的类型信息线索。理解这些模式可以自动化推断大量类型。

### 4.1 ncbAttachTJS2ClassAutoRegister 的类型参数

RTTI 字符串 `.?AU?$ncbAttachTJS2ClassAutoRegister@UWindowEx@@@@` 告诉我们：

- `ncbAttachTJS2ClassAutoRegister<WindowEx>` 被实例化
- 模板参数 `T = WindowEx`（结构体）
- 这意味着 `WindowEx` 类实现了 ncbind 所需的接口

从 `cpp/core/plugin/ncbind.hpp` 中，`ncbAttachTJS2ClassAutoRegister<T>` 的声明包含：

```cpp
template<typename T>
struct ncbAttachTJS2ClassAutoRegister {
    typedef ncbClassInfo<T> ClassInfoType;
    // 构造函数中调用 ncbClassInfo<T>::Instance().Regist(...)
    ncbAttachTJS2ClassAutoRegister();
};
```

因此，如果我们在 IDA 中看到 `ncbAttachTJS2ClassAutoRegister<WindowEx>` 的构造函数（由 RTTI 定位），就可以知道它调用了 `ncbClassInfo<WindowEx>::Instance().Regist`——这个调用就是整个类的 TJS2 注册入口！

### 4.2 ncbInstanceAdaptor 的结构

RTTI 字符串 `.?AU?$ncbInstanceAdaptor@UWindowEx@@@@` 告诉我们存在 `ncbInstanceAdaptor<WindowEx>` 的实例。这个适配器类负责在 TJS2 脚本调用成员方法时，将脚本参数转换为 C++ 参数并调用 `WindowEx` 的方法。

从它的构造函数中可以找到所有"每个 TJS2 方法注册了哪些 C++ 函数指针"的完整列表。

### 代码示例 3：分析 ncbind 注册链，提取所有 TJS2 方法

```python
# stuffs/ida_scripts/analyze_ncb_registration.py
# 用途：通过分析 ncbAttachTJS2ClassAutoRegister 的构造函数
#       提取 windowEx.dll 中所有向 TJS2 注册的方法和属性
# 运行：idat64.exe -A -S"analyze_ncb_registration.py" windowEx.dll.i64

import idc
import idautils
import ida_bytes

def find_string_ea(s, encoding='ascii'):
    """在数据段中查找字符串，返回地址（找不到返回 None）"""
    target = s.encode(encoding) + b'\x00'
    for seg in idautils.Segments():
        seg_name = idc.get_segm_name(seg)
        if seg_name not in ('.rdata', '.data'):
            continue
        seg_end = idc.get_segm_end(seg)
        ea = seg
        while ea < seg_end - len(target):
            # 逐字节比对
            candidate = ida_bytes.get_bytes(ea, len(target))
            if candidate == target:
                return ea
            ea += 1
    return None

def get_refs_in_func(string_ea):
    """返回引用某地址的所有函数（去重）"""
    funcs = set()
    for ref in idautils.XrefsTo(string_ea, 0):
        func_ea = idc.get_func_attr(ref.frm, idc.FUNCATTR_START)
        if func_ea != idc.BADADDR:
            funcs.add(func_ea)
    return funcs

# windowEx.dll 中已知的 TJS2 方法名字符串（来自 windowex_analysis.txt）
# ncbind 在注册时，这些字符串被传给 TJS2 引擎作为方法名
KNOWN_TJS_METHODS = [
    # Window 类的方法
    ("Window", [
        "getCPUType",
        "getSystemMetrics",
        "setCursorPos",
        "getCursorPos",
        "getKeyState",
        "postMessage",
        "sendMessage",
    ]),
    # Window.debug 命名空间
    ("Window.debug", [
        "console.bringAfter",
        "console.getVisible",
        "console.setVisible",
        "console.getOpacity",
        "console.setOpacity",
        "console.getAlwaysOnTop",
        "console.setAlwaysOnTop",
        "console.bringBefore",
    ]),
    # MenuItem 类的方法
    ("MenuItem", [
        "popupEx",
    ]),
]

print("=== 分析 ncbind TJS2 方法注册链 ===\n")
print(f"{'TJS2 类':15} {'方法名':30} {'注册函数地址':12} {'函数大小':8}")
print("-" * 70)

for tjs_class, methods in KNOWN_TJS_METHODS:
    for method in methods:
        # 尝试找到方法名字符串（ncbFunctionTag 格式：ClassName_MethodName）
        # 或 "MethodName"（纯方法名）
        tag_candidates = [
            method,                          # 直接方法名
            method.replace(".", "_"),        # console.bringAfter → console_bringAfter
            f"{tjs_class}_{method}",         # Window_getCPUType
            f"{tjs_class.replace('.', '_')}_{method}",
        ]

        found_ea = None
        for candidate in tag_candidates:
            ea = find_string_ea(candidate)
            if ea:
                found_ea = ea
                break

        if found_ea:
            # 找引用这个字符串的函数
            ref_funcs = get_refs_in_func(found_ea)
            for func_ea in ref_funcs:
                func_size = idc.get_func_attr(func_ea, idc.FUNCATTR_SIZE)
                func_name = idc.get_func_name(func_ea)
                print(f"  {tjs_class:13} {method:28} 0x{func_ea:08X}    {func_size:6}B  ({func_name})")
                # 重命名注册函数
                safe_method = method.replace(".", "_")
                new_name = f"ncb_{tjs_class.replace('.', '_')}_{safe_method}_register"
                idc.set_name(func_ea, new_name, idc.SN_NOCHECK | idc.SN_FORCE)
        else:
            print(f"  {tjs_class:13} {method:28} [字符串未找到]")

idc.qexit(0)
```

```python
#       不依赖已知地址，纯粹通过 RTTI + 字符串扫描推断
# 运行：idat64.exe -A -S"generic_ncbind_report.py" 任意插件.i64

import idc
import idautils
import ida_bytes
import ida_entry
import os

OUTPUT = os.path.join(os.path.dirname(__file__), "ncbind_analysis_report.txt")

def scan_rtti():
    """扫描所有 RTTI TypeDescriptor"""
    results = []
    for seg in idautils.Segments():
        if idc.get_segm_name(seg) not in ('.rdata', '.data'):
            continue
        ea = seg
        end = idc.get_segm_end(seg)
        while ea < end:
            s = idc.get_strlit_contents(ea, -1, idc.STRTYPE_C)
            if s:
                try:
                    s_str = s.decode('ascii')
                    if s_str.startswith('.?A'):
                        results.append((ea, s_str))
                    ea += len(s) + 1
                except Exception:
                    ea += 1
            else:
                ea += 1
    return results

def scan_ncb_tags():
    """扫描 ncbFunctionTag 样式的字符串（含下划线、纯 ASCII 可打印）"""
    results = []
    for seg in idautils.Segments():
        if idc.get_segm_name(seg) not in ('.rdata', '.data'):
            continue
        ea = seg
        end = idc.get_segm_end(seg)
        while ea < end:
            s = idc.get_strlit_contents(ea, -1, idc.STRTYPE_C)
            if s:
                try:
                    s_str = s.decode('ascii')
                    if ('_' in s_str and 4 < len(s_str) < 64
                            and s_str.replace('_', '').isalnum()
                            and not s_str.startswith('.')):
                        results.append((ea, s_str))
                    ea += len(s) + 1
                except Exception:
                    ea += 1
            else:
                ea += 1
    return results

def get_exports():
    """获取 DLL 导出函数列表"""
    exports = []

        ea = ida_entry.get_entry(ordinal)

        exports.append((ea, ordinal, name))
    return exports

    """统计函数信息"""
    total = named = anon = 0

        name = idc.get_func_name(func_ea)
        if name.startswith('sub_') or name.startswith('nullsub_'):

        else:

    return total, named, anon

lines.append(f"命名覆盖率：{named/total*100:.1f}%")
lines.append("")

# 导出表
exports = get_exports()
lines.append(f"导出函数（{len(exports)} 个）：")
for ea, ord_, name in exports:
    lines.append(f"  序号 {ord_:3d}  0x{ea:08X}  {name}")
lines.append("")

lines.append(f"RTTI TypeDescriptor（{len(rtti_list)} 个，推断的 C++ 类）：")
```

## 练习题与答案

### 题目 1：`idc_parse_types` 和 `idc.add_struc` / `add_struc_member` 两种方式各有什么适用场景？

<details>
<summary>查看答案</summary>

**`idc.add_struc` / `add_struc_member`（程序化 API）适用于：**
- 结构体成员数量多、需要根据运行时数据动态生成的情况
- 需要精确控制每个成员类型（如通过 `set_member_tinfo` 设置复杂指针类型）
- 需要为成员添加注释（`ida_struct.set_member_cmt`）
- 在自动化脚本中根据反汇编分析结果动态创建结构体

**`idc_parse_types`（头文件解析）适用于：**
- 已经有 C/C++ 源码或头文件的情况（直接粘贴或生成）
- 需要一次性注册大量相关类型（枚举、typedef、多个结构体）
- 想要保持类型定义以人类可读的 C 代码形式保存
- 生成通用类型库（`.til`），供多个分析项目复用

**实践建议**：先用 `idc_parse_types` 快速注册类型原型，再用 `add_struc_member` 精确补充每个成员的类型信息和注释。两种方式互补，不互斥。

</details>

### 题目 2：如果 windowEx.dll 升级了一个新版本（同样没有源码），新版本的结构体多了一个成员导致所有偏移都偏移了 4 字节，如何快速用脚本更新 IDA 数据库？

<details>
<summary>查看答案</summary>

```python
# 方案：删除旧结构体，重建新结构体，然后重新应用偏移标注

import idc
import ida_struct
import ida_loader

def rebuild_struct_with_new_member(struct_name, new_member_offset,
                                    new_member_name, shift_after_offset):
    """
    在指定偏移处插入新成员，并将之后所有成员偏移增加 shift_after_offset 字节

    参数：
        struct_name: 结构体名称
        new_member_offset: 新成员插入的偏移
        new_member_name: 新成员名称
        shift_after_offset: 要移位的起始偏移（新成员之后的所有成员）
    """
    sid = idc.get_struc_id(struct_name)
    if sid == idc.BADADDR:
        print(f"错误：结构体 {struct_name} 不存在")
        return

    # 备份现有成员（偏移 >= shift_after_offset 的）
    struc = ida_struct.get_struc(sid)
    old_members = []
    member = ida_struct.get_first_member(struc)
    while member:
        offset = member.soff
        name = member.name
        size = idc.get_member_size(sid, offset)
        cmt = idc.get_member_cmt(sid, offset, 1) or ""
        old_members.append((offset, name, size, cmt))
        member = ida_struct.get_next_member(struc, offset)

    # 删除并重建结构体
    idc.del_struc(sid)
    new_sid = idc.add_struc(-1, struct_name, 0)

    # 重新添加成员，在目标偏移处插入新成员
    inserted = False
    for offset, name, size, cmt in old_members:
        if offset >= shift_after_offset and not inserted:
            # 先插入新成员
            idc.add_struc_member(new_sid, new_member_name, new_member_offset,
                                  idc.FF_DWORD | idc.FF_DATA, -1, 4)
            inserted = True

        # 添加原成员（偏移后移 4 字节）
        new_offset = offset + 4 if offset >= shift_after_offset else offset
        idc.add_struc_member(new_sid, name, new_offset,
                              idc.FF_DWORD | idc.FF_DATA, -1, size)

    print(f"✓ {struct_name} 已更新（在 +{new_member_offset} 插入 {new_member_name}）")
    print(f"  +{shift_after_offset} 之后的所有成员偏移已增加 4 字节")

# 示例：在 WindowEx.hwnd（+4）前插入一个新的 version 字段（+4），
#        原 hwnd 及之后的所有成员偏移 +4
rebuild_struct_with_new_member(
    struct_name="WindowEx",
    new_member_offset=4,
    new_member_name="version",
    shift_after_offset=4  # hwnd 和之后的成员都后移 4 字节
)

ida_loader.save_database("", 0)
print("数据库已保存。")
idc.qexit(0)
```

**注意**：重建结构体后，之前用 `op_stroff` 标注的偏移引用也会因为偏移值变了而失效，需要重新运行 `apply_struct_offsets.py`。

</details>

### 题目 3：解释为什么 `generate_header.py` 生成的头文件在 tilib 编译时必须是纯 C 代码，不能有 C++ 语法，以及如何处理虚函数表指针成员？

<details>
<summary>查看答案</summary>

**为什么必须是纯 C：**

IDA 的 `tilib` 工具（以及 `idc_parse_types`）内部使用的是一个 C 语言语法解析器，**不支持** C++ 特性，包括：
- `class`（支持 `struct`，但语义相同）
- `virtual` 关键字
- 模板（`template<>`）
- 引用类型（`int&`）
- 继承（`struct A : B {}`）
- 方法声明（结构体内的函数）

这是因为 IDA 的类型系统设计是语言无关的（可以分析 C/C++/Pascal/Delphi 等），因此选择了最小公分母——C 类型语言。

**如何处理虚函数表指针：**

```c
/* 方案 1：用匿名 void* 成员表示 vtable 指针 */
struct WindowEx_noVtbl {
    void*   _vftable;   /* 虚函数表指针（4字节，指向函数指针数组）*/
    HWND    hwnd;
    /* ... */
};

/* 方案 2：创建 vtable 结构体，用具名指针成员 */
struct WindowEx_vtbl {
    int (*destructor)(struct WindowEx* this);
    int (*onMessage)(struct WindowEx* this, HWND hWnd, UINT uMsg, int wParam, int lParam);
    /* ... 其他虚函数 ... */
};

struct WindowEx_withVtbl {
    struct WindowEx_vtbl* _vftable; /* 明确指向 vtable 结构体 */
    HWND    hwnd;
    /* ... */
};
```

方案 2 更精确，但要求先确定 vtable 的完整虚函数列表和每个函数的签名。对于快速分析，方案 1（`void* _vftable`）已经足够实用——反编译视图至少能正确显示 `this->_vftable` 而不是 `*(_DWORD *)ecx`。

</details>
