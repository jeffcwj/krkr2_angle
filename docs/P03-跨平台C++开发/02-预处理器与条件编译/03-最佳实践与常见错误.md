## 条件编译最佳实践

### 原则 1：最小化条件编译范围

```cpp
// ❌ 差：整个函数都在条件编译中，代码重复严重
#ifdef _WIN32
void processFile(const std::string& path) {
    HANDLE hFile = CreateFileA(path.c_str(), GENERIC_READ, 0, nullptr,
                               OPEN_EXISTING, 0, nullptr);
    // ... 100行 Windows 特有代码 ...
    char buffer[4096];
    // ... 处理逻辑（与平台无关） ...
    CloseHandle(hFile);
}
#else
void processFile(const std::string& path) {
    int fd = open(path.c_str(), O_RDONLY);
    // ... 100行 POSIX 特有代码 ...
    char buffer[4096];
    // ... 处理逻辑（与平台无关，与上面重复） ...
    close(fd);
}
#endif

// ✅ 好：仅隔离平台差异部分，共享公共逻辑
class FileHandle {
public:
    explicit FileHandle(const std::string& path) {
#ifdef _WIN32
        handle_ = CreateFileA(path.c_str(), GENERIC_READ, 0, nullptr,
                              OPEN_EXISTING, 0, nullptr);
#else
        fd_ = open(path.c_str(), O_RDONLY);
#endif
    }

    size_t read(char* buffer, size_t size) {
#ifdef _WIN32
        DWORD bytesRead = 0;
        ReadFile(handle_, buffer, static_cast<DWORD>(size), &bytesRead, nullptr);
        return bytesRead;
#else
        return ::read(fd_, buffer, size);
#endif
    }

    ~FileHandle() {
#ifdef _WIN32
        if (handle_ != INVALID_HANDLE_VALUE) CloseHandle(handle_);
#else
        if (fd_ >= 0) close(fd_);
#endif
    }

private:
#ifdef _WIN32
    HANDLE handle_ = INVALID_HANDLE_VALUE;
#else
    int fd_ = -1;
#endif
};

// 平台无关的处理逻辑——不需要任何 #ifdef
void processFile(const std::string& path) {
    FileHandle file(path);
    char buffer[4096];
    size_t bytesRead = file.read(buffer, sizeof(buffer));
    // ... 处理逻辑 ...
}
```

### 原则 2：优先用文件分离替代 `#ifdef`

当平台差异较大时（如整个函数的实现完全不同），把不同平台的代码放在不同文件中：

```
src/
├── file_handle.h            ← 公共接口（纯声明）
├── file_handle_win32.cpp    ← Windows 实现
├── file_handle_posix.cpp    ← Linux/macOS 实现
└── file_handle_android.cpp  ← Android 实现
```

```cmake
# CMakeLists.txt
target_sources(mylib PRIVATE
    file_handle.h
    $<$<PLATFORM_ID:Windows>:file_handle_win32.cpp>
    $<$<PLATFORM_ID:Linux>:file_handle_posix.cpp>
    $<$<PLATFORM_ID:Darwin>:file_handle_posix.cpp>
    $<$<PLATFORM_ID:Android>:file_handle_android.cpp>
)
```

### 原则 3：使用 `#error` 捕获未处理的平台

```cpp
#if defined(_WIN32)
    // Windows
#elif defined(__ANDROID__)
    // Android
#elif defined(__linux__)
    // Linux
#elif defined(__APPLE__)
    // macOS
#else
    #error "Unsupported platform: please add support for this OS"
#endif
```

### 原则 4：避免嵌套条件编译

```cpp
// ❌ 差：三层嵌套，难以阅读和维护
#ifdef _WIN32
    #ifdef _DEBUG
        #if _MSC_VER >= 1920
            // ...
        #endif
    #endif
#endif

// ✅ 好：用 #if defined + 逻辑运算符压平嵌套
#if defined(_WIN32) && defined(_DEBUG) && (_MSC_VER >= 1920)
    // ...
#endif
```

## 常见错误及解决方案

### 错误 1：Android 被 `__linux__` 匹配

```cpp
// 问题：Android 上 __linux__ 也为 true
#ifdef __linux__
    #include <gtk/gtk.h>  // Android 上没有 GTK！编译失败
#endif

// 解决：Android 判断放在 Linux 前面
#if defined(__ANDROID__)
    #include <android/log.h>
#elif defined(__linux__)
    #include <gtk/gtk.h>
#endif
```

### 错误 2：MSVC 的 `__cplusplus` 不准确

```cpp
// 问题：MSVC 默认 __cplusplus == 199711L，即使用了 /std:c++17
#if __cplusplus >= 201703L
    // 这段代码在 MSVC 上不会编译！
    #include <optional>
#endif

// 解决方案 1：在 CMake 中添加 /Zc:__cplusplus
// target_compile_options(mytarget PRIVATE /Zc:__cplusplus)

// 解决方案 2：同时检查 _MSC_VER
#if __cplusplus >= 201703L || (defined(_MSC_VER) && _MSC_VER >= 1914)
    #include <optional>
#endif
```

### 错误 3：忘记 `#endif`

```cpp
#ifdef _WIN32
    void winFunc() { /* ... */ }
// 忘记 #endif → 后面所有代码都只在 Windows 编译

void commonFunc() { /* 这个函数在非 Windows 上消失了！ */ }
```

**预防措施：** 在 `#endif` 后面加注释标明对应的条件：

```cpp
#ifdef _WIN32
    void winFunc() { /* ... */ }
#endif // _WIN32
```
