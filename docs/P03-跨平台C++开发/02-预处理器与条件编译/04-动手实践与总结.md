## 实验 1：跨平台日志宏

### 目标

创建一个 `cross_log.h` 头文件，实现跨平台日志输出。不同平台使用不同的底层 API：

| 平台 | 日志 API | 特点 |
|------|---------|------|
| Android | `__android_log_print` | 输出到 Logcat，需要链接 `log` 库 |
| Windows | `OutputDebugStringA` + `printf` | 同时输出到调试器和控制台 |
| Linux/macOS | `printf` / `fprintf` | 标准 C 输出 |

### 完整实现

```cpp
// cross_log.h — 跨平台日志宏
#pragma once

#include <cstdio>  // printf, fprintf, snprintf

// ========== 平台特定的日志实现 ==========

#if defined(__ANDROID__)
    // Android: 使用 NDK 的日志 API
    // 输出到 Logcat，可以用 adb logcat 查看
    #include <android/log.h>
    #define LOG_TAG "MyApp"

    // ##__VA_ARGS__ 是 GCC/Clang 扩展：
    // 当没有可变参数时，## 会自动去掉前面的逗号
    #define LOG_INFO(fmt, ...) \
        __android_log_print(ANDROID_LOG_INFO, LOG_TAG, \
                            fmt, ##__VA_ARGS__)
    #define LOG_ERROR(fmt, ...) \
        __android_log_print(ANDROID_LOG_ERROR, LOG_TAG, \
                            fmt, ##__VA_ARGS__)
    #define LOG_DEBUG(fmt, ...) \
        __android_log_print(ANDROID_LOG_DEBUG, LOG_TAG, \
                            fmt, ##__VA_ARGS__)

#elif defined(_WIN32)
    // Windows: 同时输出到调试器的 Output 窗口和控制台
    #include <windows.h>

    // do { ... } while(0) 是宏的标准写法：
    // 让宏在 if/else 中当作一条语句使用，避免语法问题
    #define LOG_INFO(fmt, ...) \
        do { \
            char _log_buf[1024]; \
            snprintf(_log_buf, sizeof(_log_buf), \
                     "[INFO] " fmt "\n", ##__VA_ARGS__); \
            OutputDebugStringA(_log_buf); \
            printf("[INFO] " fmt "\n", ##__VA_ARGS__); \
        } while(0)

    #define LOG_ERROR(fmt, ...) \
        do { \
            char _log_buf[1024]; \
            snprintf(_log_buf, sizeof(_log_buf), \
                     "[ERROR] " fmt "\n", ##__VA_ARGS__); \
            OutputDebugStringA(_log_buf); \
            fprintf(stderr, "[ERROR] " fmt "\n", ##__VA_ARGS__); \
        } while(0)

    #define LOG_DEBUG(fmt, ...) \
        do { \
            char _log_buf[1024]; \
            snprintf(_log_buf, sizeof(_log_buf), \
                     "[DEBUG] " fmt "\n", ##__VA_ARGS__); \
            OutputDebugStringA(_log_buf); \
            printf("[DEBUG] " fmt "\n", ##__VA_ARGS__); \
        } while(0)

#else
    // Linux / macOS: 标准输出
    #define LOG_INFO(fmt, ...) \
        printf("[INFO] " fmt "\n", ##__VA_ARGS__)
    #define LOG_ERROR(fmt, ...) \
        fprintf(stderr, "[ERROR] " fmt "\n", ##__VA_ARGS__)
    #define LOG_DEBUG(fmt, ...) \
        printf("[DEBUG] " fmt "\n", ##__VA_ARGS__)
#endif

// ========== Release 模式禁用 DEBUG 日志 ==========

// NDEBUG 是 CMake 在 Release/RelWithDebInfo 配置时自动定义的宏
// Debug 配置不定义 NDEBUG → LOG_DEBUG 正常输出
// Release 配置定义 NDEBUG → LOG_DEBUG 变成空操作
#ifdef NDEBUG
    #undef LOG_DEBUG
    #define LOG_DEBUG(fmt, ...) ((void)0)  // 空操作，编译器会完全优化掉
#endif
```

### 使用示例

```cpp
// main.cpp
#include "cross_log.h"

int main() {
    LOG_INFO("Application started, version %d.%d", 2, 0);
    LOG_DEBUG("Debug pointer value: %p", (void*)main);
    LOG_ERROR("Failed to open file: %s", "data.bin");

    // 不带参数也可以（得益于 ##__VA_ARGS__）
    LOG_INFO("Simple message without args");

    return 0;
}
```

### 为什么用 `do { ... } while(0)` 包裹宏？

这是 C/C++ 中宏的经典技巧。看以下场景：

```cpp
// ❌ 如果宏不用 do-while 包裹
#define LOG(msg) printf("[LOG] "); printf(msg); printf("\n")

if (hasError)
    LOG("error occurred");  // 展开后只有第一个 printf 在 if 里
else                        // 编译错误：else 没有匹配的 if
    doSomething();
```

展开后：

```cpp
if (hasError)
    printf("[LOG] ");       // 只有这行在 if 中
printf("error occurred");   // 这行永远执行
printf("\n");               // 这行永远执行
else                        // 语法错误！
    doSomething();
```

用 `do { ... } while(0)` 包裹就不会有这个问题，因为它让宏展开后始终是一条完整的语句。

## 实验 2：跨平台配置头文件

### 目标

创建一个 `platform_config.h`，集中所有平台检测逻辑，让项目中其他文件只需 `#include "platform_config.h"` 即可使用统一的平台宏。

### 完整实现

```cpp
// platform_config.h — 统一的平台配置
#pragma once

// ========== 1. 操作系统检测 ==========

#if defined(_WIN32)
    #define PLATFORM_WINDOWS 1
    #define PLATFORM_NAME "Windows"
#elif defined(__ANDROID__)
    // Android 必须在 __linux__ 前面！
    #define PLATFORM_ANDROID 1
    #define PLATFORM_NAME "Android"
#elif defined(__linux__)
    #define PLATFORM_LINUX 1
    #define PLATFORM_NAME "Linux"
#elif defined(__APPLE__)
    #include <TargetConditionals.h>
    #if TARGET_OS_IPHONE
        #define PLATFORM_IOS 1
        #define PLATFORM_NAME "iOS"
    #else
        #define PLATFORM_MACOS 1
        #define PLATFORM_NAME "macOS"
    #endif
#else
    #error "Unsupported platform: please add detection for this OS"
#endif

// 桌面平台 vs 移动平台
#if defined(PLATFORM_WINDOWS) || defined(PLATFORM_LINUX) || \
    defined(PLATFORM_MACOS)
    #define PLATFORM_DESKTOP 1
#elif defined(PLATFORM_ANDROID) || defined(PLATFORM_IOS)
    #define PLATFORM_MOBILE 1
#endif

// POSIX 兼容平台
#if defined(PLATFORM_LINUX) || defined(PLATFORM_MACOS) || \
    defined(PLATFORM_ANDROID)
    #define PLATFORM_POSIX 1
#endif

// ========== 2. 编译器检测 ==========

#if defined(__clang__)
    // Clang 必须在 __GNUC__ 前面（Clang 也定义 __GNUC__）
    #define COMPILER_CLANG 1
    #define COMPILER_NAME "Clang"
    #define COMPILER_VERSION \
        (__clang_major__ * 10000 + __clang_minor__ * 100 + __clang_patchlevel__)
#elif defined(__GNUC__)
    #define COMPILER_GCC 1
    #define COMPILER_NAME "GCC"
    #define COMPILER_VERSION \
        (__GNUC__ * 10000 + __GNUC_MINOR__ * 100 + __GNUC_PATCHLEVEL__)
#elif defined(_MSC_VER)
    #define COMPILER_MSVC 1
    #define COMPILER_NAME "MSVC"
    #define COMPILER_VERSION _MSC_VER
#else
    #error "Unsupported compiler"
#endif

// ========== 3. 架构检测 ==========

#if defined(__x86_64__) || defined(_M_X64)
    #define ARCH_X64 1
    #define ARCH_NAME "x86_64"
#elif defined(__i386__) || defined(_M_IX86)
    #define ARCH_X86 1
    #define ARCH_NAME "x86"
#elif defined(__aarch64__) || defined(_M_ARM64)
    #define ARCH_ARM64 1
    #define ARCH_NAME "ARM64"
#elif defined(__arm__) || defined(_M_ARM)
    #define ARCH_ARM32 1
    #define ARCH_NAME "ARM32"
#else
    #define ARCH_NAME "Unknown"
#endif

// ========== 4. C++ 标准版本（兼容 MSVC）==========

#if defined(_MSVC_LANG)
    #define CPP_VERSION _MSVC_LANG
#else
    #define CPP_VERSION __cplusplus
#endif

#if CPP_VERSION >= 202002L
    #define HAS_CPP20 1
#endif
#if CPP_VERSION >= 201703L
    #define HAS_CPP17 1
#endif
#if CPP_VERSION >= 201402L
    #define HAS_CPP14 1
#endif

// ========== 5. 实用的编译器特性宏 ==========

// 标记可能不使用的变量，避免编译警告
#define UNUSED(x) ((void)(x))

// 标记函数为 deprecated（已弃用）
#if defined(COMPILER_MSVC)
    #define DEPRECATED(msg) __declspec(deprecated(msg))
#elif defined(COMPILER_GCC) || defined(COMPILER_CLANG)
    #define DEPRECATED(msg) __attribute__((deprecated(msg)))
#else
    #define DEPRECATED(msg)
#endif

// 分支预测提示（C++20 有 [[likely]]/[[unlikely]]）
#if defined(HAS_CPP20)
    #define LIKELY(x)   (x) [[likely]]
    #define UNLIKELY(x) (x) [[unlikely]]
#elif defined(COMPILER_GCC) || defined(COMPILER_CLANG)
    #define LIKELY(x)   __builtin_expect(!!(x), 1)
    #define UNLIKELY(x) __builtin_expect(!!(x), 0)
#else
    #define LIKELY(x)   (x)
    #define UNLIKELY(x) (x)
#endif
```

## 对照项目源码

KrKr2 使用 spdlog 作为日志框架而非手写日志宏，但条件编译的运用贯穿整个项目：

### 关键文件

| 文件路径 | 条件编译用途 |
|----------|-------------|
| `CMakeLists.txt` 第 12-25 行 | 平台检测：设置自定义 `WINDOWS`/`LINUX`/`MACOS`/`ANDROID` 变量 |
| `cpp/core/CMakeLists.txt` 第 35-50 行 | `INTERFACE` 编译定义传递（`USE_UNICODE_FSTRING` 等） |
| `cpp/core/environ/` 各子目录 | 文件级平台分离——win32/linux/android/apple/sdl 各自独立实现 |
| `platforms/android/cpp/krkr2_android.cpp` | `__ANDROID__` 特有的 JNI 桥接代码 |
| `platforms/windows/main.cpp` | `_WIN32` 特有的 WinMain 入口 |
| `cpp/core/visual/` 图像加载器 | OpenMP 条件编译（`#ifdef _OPENMP`） |

**KrKr2 的做法与我们实验的异同：**

- **相同点**：KrKr2 也使用自定义平台宏（`WINDOWS`、`LINUX` 等），类似我们实验 2 的 `PLATFORM_WINDOWS`、`PLATFORM_LINUX`
- **不同点**：KrKr2 通过 CMake 的 `target_compile_definitions` 传入这些宏，而不是在头文件中用预处理器检测。这是因为 KrKr2 使用 CMake 管理整个构建流程，CMake 层面的平台变量比预处理器宏更可靠

### 核心概念速查表

| 概念 | 说明 | 典型用法 |
|------|------|---------|
| `#ifdef X` | 检查宏 X 是否已定义 | `#ifdef _WIN32` |
| `#if defined(X)` | 同上，但支持逻辑运算 | `#if defined(X) && defined(Y)` |
| `#elif` | else-if 分支 | 多平台判断链 |
| `#error "msg"` | 触发编译错误 | 未支持平台的安全网 |
| `#pragma once` | 防止头文件重复包含 | 所有 `.h` 文件顶部 |
| `##__VA_ARGS__` | 可变参数宏的 GCC 扩展 | 日志宏（无参数时去逗号） |
| `do { } while(0)` | 多语句宏的安全包裹 | 避免 if-else 中的语法问题 |
| `target_compile_definitions` | CMake 传递编译宏 | 项目级配置（比硬编码更灵活） |
| Generator Expressions | CMake 条件宏 | `$<$<CONFIG:Debug>:X>` |
| `configure_file` | CMake 生成配置头文件 | 版本号、特性开关 |

### 五条黄金规则

1. **Android 先于 Linux** — `__ANDROID__` 检查必须在 `__linux__` 前面
2. **Clang 先于 GCC** — `__clang__` 检查必须在 `__GNUC__` 前面
3. **MSVC 用 `_MSVC_LANG`** — 不要只靠 `__cplusplus` 判断 C++ 版本
4. **`#else #error`** — 每个平台判断链都以 `#error` 结尾
5. **能用 CMake 就不用 `#ifdef`** — CMake 的 `target_compile_definitions` 比源码内硬编码更灵活可控

## 练习题与答案

### 题目 1：宏判断顺序

以下代码在 Android 设备上编译运行，会输出什么？为什么？

```cpp
#include <cstdio>
int main() {
#ifdef __linux__
    printf("Linux\n");
#elif defined(__ANDROID__)
    printf("Android\n");
#endif
    return 0;
}
```

<details>
<summary>查看答案</summary>

**输出：** `Linux`

**原因：** Android NDK 编译时同时定义了 `__linux__` 和 `__ANDROID__`。`#ifdef __linux__` 先被匹配到（因为 `#elif` 只在前面的条件不成立时才检查），`#elif defined(__ANDROID__)` 分支被跳过。

**修复方法：** 把 `__ANDROID__` 的判断放在 `__linux__` 前面：

```cpp
#include <cstdio>
int main() {
#if defined(__ANDROID__)
    printf("Android\n");
#elif defined(__linux__)
    printf("Linux\n");
#endif
    return 0;
}
```

</details>

### 题目 2：CMake 编译定义

写出 CMake 代码，实现以下需求：
- 为目标 `myapp` 添加编译定义 `APP_VERSION=3`
- 仅在 Debug 配置时定义 `ENABLE_LOGGING`
- 仅在 Windows 平台时定义 `USE_WINSOCK`

<details>
<summary>查看答案</summary>

```cmake
target_compile_definitions(myapp PRIVATE
    APP_VERSION=3
    $<$<CONFIG:Debug>:ENABLE_LOGGING>
    $<$<PLATFORM_ID:Windows>:USE_WINSOCK>
)
```

**解释：**
- `APP_VERSION=3` — 无条件定义，所有平台所有配置都有
- `$<$<CONFIG:Debug>:ENABLE_LOGGING>` — 生成器表达式，仅 Debug 配置时展开为 `ENABLE_LOGGING`，Release 时为空字符串
- `$<$<PLATFORM_ID:Windows>:USE_WINSOCK>` — 仅 Windows 平台时展开为 `USE_WINSOCK`

</details>

### 题目 3：条件编译重构

以下代码使用了过多的 `#ifdef`，请将其重构为文件分离的方式，给出文件结构和 CMakeLists.txt：

```cpp
// network.cpp
#include "network.h"

void initNetwork() {
#ifdef _WIN32
    WSADATA wsaData;
    WSAStartup(MAKEWORD(2, 2), &wsaData);
#elif defined(__linux__) || defined(__APPLE__)
    signal(SIGPIPE, SIG_IGN);
#elif defined(__ANDROID__)
    signal(SIGPIPE, SIG_IGN);
#endif
}

void cleanupNetwork() {
#ifdef _WIN32
    WSACleanup();
#endif
}
```

<details>
<summary>查看答案</summary>

**文件结构：**

```
src/
├── network.h                  ← 公共接口声明
├── network_win32.cpp          ← Windows 实现
├── network_posix.cpp          ← Linux/macOS 实现
├── network_android.cpp        ← Android 实现
└── CMakeLists.txt
```

**network.h：**

```cpp
#pragma once

void initNetwork();
void cleanupNetwork();
```

**network_win32.cpp：**

```cpp
#include "network.h"
#include <winsock2.h>

void initNetwork() {
    WSADATA wsaData;
    WSAStartup(MAKEWORD(2, 2), &wsaData);
}

void cleanupNetwork() {
    WSACleanup();
}
```

**network_posix.cpp：**

```cpp
#include "network.h"
#include <csignal>

void initNetwork() {
    // 忽略 SIGPIPE：当写入已关闭的 socket 时不会导致进程终止
    signal(SIGPIPE, SIG_IGN);
}

void cleanupNetwork() {
    // POSIX 套接字不需要全局清理
}
```

**network_android.cpp：**

```cpp
#include "network.h"
#include <csignal>

void initNetwork() {
    signal(SIGPIPE, SIG_IGN);
    // Android 特有：可以在此检查网络权限或初始化 JNI 网络回调
}

void cleanupNetwork() {
    // Android 不需要全局清理
}
```

**CMakeLists.txt：**

```cmake
add_library(network STATIC
    network.h
    $<$<PLATFORM_ID:Windows>:network_win32.cpp>
    $<$<PLATFORM_ID:Linux>:network_posix.cpp>
    $<$<PLATFORM_ID:Darwin>:network_posix.cpp>
    $<$<PLATFORM_ID:Android>:network_android.cpp>
)

target_include_directories(network PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})

if(WIN32)
    target_link_libraries(network PRIVATE ws2_32)
endif()
```

**重构优势：**
- 每个文件专注一个平台，无需 `#ifdef`
- 添加新平台只需新增一个 `.cpp` 文件和一行 CMake
- 编译时只编译当前平台的文件
- 代码审查时立即知道影响哪个平台
