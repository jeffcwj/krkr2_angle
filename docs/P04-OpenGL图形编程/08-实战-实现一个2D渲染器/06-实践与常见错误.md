### CMakeLists.txt

```cmake
cmake_minimum_required(VERSION 3.20)
project(My2DRenderer LANGUAGES CXX)

# C++17 标准（KrKr2 项目同样使用 C++17）
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# ----- 查找依赖库 -----
# GLFW：跨平台窗口和输入管理库
# 在 Windows/Linux/macOS 上提供统一的窗口创建、键盘鼠标输入、OpenGL 上下文管理
find_package(glfw3 REQUIRED)

# OpenGL：图形渲染 API（系统自带）
find_package(OpenGL REQUIRED)

# glad：OpenGL 函数加载器（运行时动态获取 GL 函数地址）
# 因为 OpenGL 是动态链接的，函数地址需要在运行时从驱动获取
find_package(glad CONFIG REQUIRED)

# ----- 定义可执行目标 -----
add_executable(renderer
    src/main.cpp
    src/sprite_batcher.cpp
    src/texture.cpp
)

# 头文件搜索路径
target_include_directories(renderer PRIVATE
    ${CMAKE_SOURCE_DIR}/include        # 项目头文件
    ${CMAKE_SOURCE_DIR}/third_party    # stb_image.h
)

# 链接库
target_link_libraries(renderer PRIVATE
    glfw            # 窗口管理
    OpenGL::GL      # OpenGL 核心库
    glad::glad      # GL 函数加载器
)

# ----- 复制资源文件到构建目录 -----
# 这样运行程序时能找到 assets/ 下的图片
file(COPY ${CMAKE_SOURCE_DIR}/assets
     DESTINATION ${CMAKE_BINARY_DIR})

# ----- 平台特定设置 -----
if(APPLE)
    # macOS 需要链接 Cocoa、IOKit、CoreVideo 框架
    # GLFW 的 find_package 通常会自动处理，但显式添加更安全
    target_link_libraries(renderer PRIVATE
        "-framework Cocoa"
        "-framework IOKit"
        "-framework CoreVideo"
    )
endif()

if(UNIX AND NOT APPLE)
    # Linux 需要链接 X11、pthread、dl
    find_package(Threads REQUIRED)
    target_link_libraries(renderer PRIVATE
        Threads::Threads
        ${CMAKE_DL_LIBS}   # dlopen/dlsym 用于动态加载
    )
endif()
```

### shader.h — 着色器编译辅助

```cpp
#pragma once
#include <glad/glad.h>
#include <iostream>
#include <string>

// compileShader — 编译单个着色器
// 参数：
//   type   — GL_VERTEX_SHADER（顶点着色器）或 GL_FRAGMENT_SHADER（片段着色器）
//   source — GLSL 源码字符串
// 返回值：成功返回着色器 ID，失败返回 0
inline GLuint compileShader(GLenum type, const char* source) {
    // glCreateShader(type) — 创建空的着色器对象
    GLuint shader = glCreateShader(type);

    // glShaderSource(shader, count, strings, lengths)
    //   设置着色器的源码
    //   count=1 表示只有一个源码字符串；lengths=nullptr 表示字符串以 '\0' 结尾
    glShaderSource(shader, 1, &source, nullptr);

    // glCompileShader(shader) — 编译着色器源码为 GPU 可执行的中间码
    glCompileShader(shader);

    // 检查编译结果
    GLint success;
    // glGetShaderiv(shader, pname, params)
    //   查询着色器属性
    //   GL_COMPILE_STATUS: 编译是否成功（GL_TRUE/GL_FALSE）
    glGetShaderiv(shader, GL_COMPILE_STATUS, &success);
    if (!success) {
        char log[512];
        // glGetShaderInfoLog — 获取编译错误信息
        glGetShaderInfoLog(shader, sizeof(log), nullptr, log);
        std::cerr << "着色器编译失败:\n" << log << std::endl;
        glDeleteShader(shader);
        return 0;
    }
    return shader;
}

// createProgram — 编译并链接着色器程序
// 着色器程序 = 顶点着色器 + 片段着色器链接在一起
inline GLuint createProgram(const char* vertSrc, const char* fragSrc) {
    GLuint vert = compileShader(GL_VERTEX_SHADER, vertSrc);
    GLuint frag = compileShader(GL_FRAGMENT_SHADER, fragSrc);
    if (!vert || !frag) return 0;

    // glCreateProgram() — 创建空的着色器程序对象
    GLuint program = glCreateProgram();

    // glAttachShader(program, shader) — 将编译好的着色器附加到程序
    glAttachShader(program, vert);
    glAttachShader(program, frag);

    // glLinkProgram(program) — 链接所有附加的着色器
    //   将顶点着色器的输出（out 变量）与片段着色器的输入（in 变量）匹配
    //   如果变量名或类型不匹配，链接会失败
    glLinkProgram(program);

    GLint success;
    glGetProgramiv(program, GL_LINK_STATUS, &success);
    if (!success) {
        char log[512];
        glGetProgramInfoLog(program, sizeof(log), nullptr, log);
        std::cerr << "着色器链接失败:\n" << log << std::endl;
        glDeleteProgram(program);
        program = 0;
    }

    // 链接完成后，着色器对象可以删除（程序已包含编译结果）
    glDeleteShader(vert);
    glDeleteShader(frag);
    return program;
}

// ortho — 生成正交投影矩阵（列主序，直接写入 16 个 float）
// 将像素坐标 [left, right] × [bottom, top] 映射到 NDC [-1, 1]
inline void ortho(float* mat,
                  float left, float right,
                  float bottom, float top) {
    // 16 个元素全部清零
    for (int i = 0; i < 16; i++) mat[i] = 0.0f;
    // 列主序：mat[col*4 + row]
    mat[0]  =  2.0f / (right - left);           // X 缩放
    mat[5]  =  2.0f / (top - bottom);            // Y 缩放
    mat[10] = -1.0f;                              // Z（2D 不需要深度）
    mat[12] = -(right + left) / (right - left);   // X 平移
    mat[13] = -(top + bottom) / (top - bottom);   // Y 平移
    mat[15] =  1.0f;                              // 齐次坐标 w
}
```

### texture.h / texture.cpp

```cpp
// include/texture.h
#pragma once
#include <glad/glad.h>

// 从文件加载图片并创建 OpenGL 纹理
// 返回纹理 ID，失败返回 0
GLuint loadTexture(const char* path);
```

```cpp
// src/texture.cpp
#define STB_IMAGE_IMPLEMENTATION  // 整个项目只在这里定义一次！
#include "stb_image.h"
#include "texture.h"
#include <iostream>

GLuint loadTexture(const char* path) {
    int width, height, channels;
    stbi_set_flip_vertically_on_load(true);
    unsigned char* data = stbi_load(path, &width, &height, &channels, 0);
    if (!data) {
        std::cerr << "纹理加载失败: " << path << std::endl;
        return 0;
    }

    GLenum fmt;
    if (channels == 4) fmt = GL_RGBA;
    else if (channels == 3) fmt = GL_RGB;
    else fmt = GL_RED;

    if (channels != 4) glPixelStorei(GL_UNPACK_ALIGNMENT, 1);

    GLuint tex;
    glGenTextures(1, &tex);
    glBindTexture(GL_TEXTURE_2D, tex);
    glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
    glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
    glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
    glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
    glTexImage2D(GL_TEXTURE_2D, 0, fmt, width, height, 0, fmt,
                 GL_UNSIGNED_BYTE, data);

    if (channels != 4) glPixelStorei(GL_UNPACK_ALIGNMENT, 4);
    stbi_image_free(data);
    return tex;
}
```

## 章节总结：从零到完整 2D 渲染器

回顾整个第 08 章的学习路径：

```
第 1 节：着色器          → 编写 GLSL，编译链接着色器程序
    ↓
第 2 节：正交投影        → 像素坐标 → NDC 的数学变换
    ↓
第 3 节：坐标变换        → 平移、缩放、旋转精灵
    ↓
第 4 节：批处理          → 合并 Draw Call，提升渲染性能
    ↓
第 5 节：纹理+渲染循环   → 加载图片，组合所有组件
    ↓
第 6 节：本节            → 完整项目搭建，知识体系总结
```

**核心思想**：
1. **GPU 是流水线**：顶点数据 → 顶点着色器 → 光栅化 → 片段着色器 → 屏幕
2. **状态机模型**：OpenGL 是全局状态机，`glBind*` 设置当前操作对象，后续调用作用于它
3. **CPU 与 GPU 分工**：CPU 准备数据（顶点、纹理、矩阵），GPU 并行处理像素
4. **性能关键在减少通信**：批处理减少 Draw Call，纹理图集减少绑定切换

### 题目 2：添加键盘控制

修改渲染循环，使角色可以通过 WASD 键移动。要求：
- W/S 控制上下移动
- A/D 控制左右移动
- 移动速度为 200 像素/秒
- 角色不能移出窗口边界

<details>
<summary>查看答案</summary>

```cpp
// 在 while 循环前定义角色位置和上一帧时间
float charX = 550.0f, charY = 150.0f;
float lastFrame = 0.0f;
const float SPEED = 200.0f;   // 像素/秒
const float CHAR_W = 200.0f, CHAR_H = 300.0f;

// 在 while 循环内，glfwPollEvents() 后添加：
float currentFrame = (float)glfwGetTime();
// deltaTime = 两帧之间的时间差（秒）
// 用时间差乘以速度，保证不同帧率下移动速度一致
float deltaTime = currentFrame - lastFrame;
lastFrame = currentFrame;

// 检测 WASD 按键状态
if (glfwGetKey(window, GLFW_KEY_W) == GLFW_PRESS)
    charY -= SPEED * deltaTime;  // 屏幕坐标 Y 轴向下，W=上移=减小 Y
if (glfwGetKey(window, GLFW_KEY_S) == GLFW_PRESS)
    charY += SPEED * deltaTime;
if (glfwGetKey(window, GLFW_KEY_A) == GLFW_PRESS)
    charX -= SPEED * deltaTime;
if (glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS)
    charX += SPEED * deltaTime;

// 边界限制（clamp 到窗口范围内）
if (charX < 0.0f) charX = 0.0f;
if (charY < 0.0f) charY = 0.0f;
if (charX + CHAR_W > SCREEN_W) charX = SCREEN_W - CHAR_W;
if (charY + CHAR_H > SCREEN_H) charY = SCREEN_H - CHAR_H;

// 绘制角色时使用 charX, charY
batcher.drawSprite(charTex, charX, charY, CHAR_W, CHAR_H,
    0.0f, 0.0f, 1.0f, 1.0f,
    1.0f, 1.0f, 1.0f, 1.0f);
```

**关键概念**：`deltaTime`（帧间隔时间）确保移动速度与帧率无关。如果直接写 `charX += 5`，在 60FPS 上移动 300 像素/秒，在 30FPS 上只移动 150 像素/秒——不一致。

</details>

### 题目 3：分析 KrKr2 与本章实现的差异

列出至少 5 个 KrKr2 渲染器与我们实现的差异，并解释 KrKr2 为什么这样做。

<details>
<summary>查看答案</summary>

| 差异点 | 我们的实现 | KrKr2 | 原因 |
|--------|-----------|-------|------|
| **纹理绑定** | 直接 `glBindTexture` | `cocos2d::GL::bindTexture2D`（带缓存） | Cocos2d-x 维护绑定缓存，跳过重复绑定，减少状态切换 |
| **像素对齐** | 简单规则（RGB→1, RGBA→4） | 按 `pitch & 7` 动态选择最优对齐 | 处理非标准 pitch 的图片（TLG 格式、内存生成等） |
| **绘制方式** | `glDrawElements`（索引缓冲） | `glDrawArrays`（6 顶点/矩形） | KrKr2 不做引擎级批处理，每次画一个矩形；批处理由 Cocos2d-x 上层处理 |
| **投影方式** | 投影矩阵在 GPU 端（uniform） | NDC 坐标在 CPU 端预计算 | 顶点着色器极简化，减少 GPU 计算；坐标变换量少（每次只画一两个矩形） |
| **显存管理** | 无 | `_totalVMemSize` 全局跟踪 | 移动设备显存有限，需要在超出预算时主动释放低优先级纹理 |
| **灰度格式** | `GL_RED`（Core Profile） | `GL_LUMINANCE`（ES 2.0） | KrKr2 需要兼容 OpenGL ES 2.0（Android），`GL_RED` 在 ES 2.0 中不可用 |
| **着色器生成** | 硬编码 GLSL 字符串 | `tTVPOGLRenderMethod_Script` 运行时动态生成 | 支持多种混合模式和渲染效果，每种效果需要不同的着色器 |

</details>
