## 1. 实战：迷你混合器

### 1.1 项目目标

我们要实现一个命令行程序，支持以下功能：
- 6 种混合模式（bmAlpha、bmAdd、bmSub、bmMul、bmScreen、bmDifference）
- HDA 变体
- Opacity 控制
- 批量扫描线处理（模拟引擎实际工作方式）

### 1.2 完整源码

```cpp
#include <cstdint>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <chrono>

using tjs_uint32 = uint32_t;
using tjs_int = int;

// ========== 基础工具 ==========

// 饱和加法（无分支版本）
// 原理：检测每个字节是否溢出，溢出则饱和到 0xFF
tjs_uint32 saturated_add(tjs_uint32 a, tjs_uint32 b) {
    // 步骤1: 计算每个字节的进位（是否溢出到第9位）
    // (a & b) 是必定进位的部分
    // (a ^ b) >> 1 是可能进位的部分（只看最高位）
    // 两者相加后，bit 7（每个字节的最高位）如果为1，表示该字节溢出
    tjs_uint32 tmp = ((a & b) + (((a ^ b) >> 1) & 0x7f7f7f7f)) & 0x80808080;
    // 步骤2: 将溢出标志扩展为 0xFF 掩码
    // 0x80 → (0x80<<1) - (0x80>>7) = 0x100 - 0x01 = 0xFF
    // 0x00 → (0x00<<1) - (0x00>>7) = 0x000 - 0x00 = 0x00
    tmp = (tmp << 1) - (tmp >> 7);
    // 步骤3: 溢出的字节用 0xFF 替代，未溢出的保留正常结果
    return (a + b - tmp) | tmp;
}

// 提取通道
inline uint8_t ch_r(tjs_uint32 p) { return (p >> 16) & 0xff; }
inline uint8_t ch_g(tjs_uint32 p) { return (p >> 8) & 0xff; }
inline uint8_t ch_b(tjs_uint32 p) { return p & 0xff; }
inline uint8_t ch_a(tjs_uint32 p) { return (p >> 24) & 0xff; }

// 合成像素
inline tjs_uint32 make_pixel(uint8_t a, uint8_t r, uint8_t g, uint8_t b) {
    return ((tjs_uint32)a << 24) | ((tjs_uint32)r << 16)
         | ((tjs_uint32)g << 8) | b;
}

// ========== 核心混合函数 ==========

// Alpha 混合（双通道并行）
tjs_uint32 alpha_blend(tjs_uint32 d, tjs_uint32 s, tjs_uint32 a) {
    tjs_uint32 d1 = d & 0xff00ff;
    d1 = (d1 + (((s & 0xff00ff) - d1) * a >> 8)) & 0xff00ff;
    d &= 0xff00;
    s &= 0xff00;
    return d1 + ((d + ((s - d) * a >> 8)) & 0xff00);
}

// ========== 混合模式 Functor ==========

// bmAlpha: 标准 SrcOver
struct blend_alpha_func {
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        tjs_uint32 a = s >> 24;
        if (a == 0) return d;       // 完全透明，跳过
        if (a == 255) return s;     // 完全不透明，覆盖
        return alpha_blend(d, s, a);
    }
};

// bmAdd: 饱和加法
struct blend_add_func {
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        tjs_uint32 a = s >> 24;
        // 源颜色先乘以 alpha
        tjs_uint32 s_rb = ((s & 0xff00ff) * a >> 8) & 0xff00ff;
        tjs_uint32 s_g  = ((s & 0x00ff00) * a >> 8) & 0x00ff00;
        return saturated_add(d, s_rb | s_g);
    }
};

// bmSub: 饱和减法 — max(D + S - 255, 0)
struct blend_sub_func {
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        // 使用位运算实现无分支饱和减法
        // 原理: D + S - 255 可能小于 0，需要 clamp 到 0
        tjs_uint32 tmp = ((s & d) + (((s ^ d) >> 1) & 0x7f7f7f7f)) & 0x80808080;
        tmp = (tmp << 1) - (tmp >> 7);
        // tmp 中溢出字节为 0xFF，未溢出字节为 0x00
        // 溢出时保留 (s+d) 的低8位；未溢出时为 0
        return (s + d - tmp) & tmp;
    }
};

// bmMul: 逐通道乘法 — D × S / 255
struct blend_mul_func {
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        uint8_t r = (uint8_t)((uint32_t)ch_r(d) * ch_r(s) / 255);
        uint8_t g = (uint8_t)((uint32_t)ch_g(d) * ch_g(s) / 255);
        uint8_t b = (uint8_t)((uint32_t)ch_b(d) * ch_b(s) / 255);
        return make_pixel(0, r, g, b);
    }
};

// bmScreen: 1 - (1-D)(1-S) / 255
struct blend_screen_func {
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        uint8_t r = 255 - (uint8_t)((255 - ch_r(d)) * (255 - ch_r(s)) / 255);
        uint8_t g = 255 - (uint8_t)((255 - ch_g(d)) * (255 - ch_g(s)) / 255);
        uint8_t b = 255 - (uint8_t)((255 - ch_b(d)) * (255 - ch_b(s)) / 255);
        return make_pixel(0, r, g, b);
    }
};

// bmDifference: |D - S|
struct blend_diff_func {
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        auto abs_d = [](uint8_t a, uint8_t b) -> uint8_t {
            return a > b ? a - b : b - a;
        };
        return make_pixel(0, abs_d(ch_r(d), ch_r(s)),
                             abs_d(ch_g(d), ch_g(s)),
                             abs_d(ch_b(d), ch_b(s)));
    }
};

// ========== 变体模板（模拟 KrKr2 blend_variation.h） ==========

// 基本版: functor + alpha
template<typename FUNC>
struct normal_op {
    FUNC func_;
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        tjs_uint32 a = s >> 24;
        if (a == 0) return d;
        tjs_uint32 blended = func_(d, s);
        if (a == 255) return blended;
        return alpha_blend(d, blended, a);
    }
};

// HDA 版: 保持 dst alpha
template<typename FUNC>
struct hda_op {
    FUNC func_;
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        tjs_uint32 result = normal_op<FUNC>{func_}(d, s);
        return (result & 0x00ffffff) | (d & 0xff000000);
    }
};

// Opacity 版: alpha × opacity
template<typename FUNC>
struct opacity_op {
    FUNC func_;
    tjs_int opa_;
    opacity_op(tjs_int opa) : opa_(opa) {}
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        tjs_uint32 a = (s >> 24) * opa_ >> 8;
        if (a == 0) return d;
        tjs_uint32 blended = func_(d, s);
        return alpha_blend(d, blended, a);
    }
};

// ========== 批量扫描线处理 ==========

template<typename OP>
void blend_scanline(tjs_uint32* dest, const tjs_uint32* src,
                    tjs_int len, OP op) {
    for (tjs_int i = 0; i < len; i++) {
        dest[i] = op(dest[i], src[i]);
    }
}

// ========== 输出辅助 ==========

void print_pixel(const char* label, tjs_uint32 p) {
    printf("  %-16s: A=%3d R=%3d G=%3d B=%3d  (0x%08X)\n",
           label, ch_a(p), ch_r(p), ch_g(p), ch_b(p), p);
}

// ========== 主程序 ==========

int main() {
    // 测试像素
    tjs_uint32 dst = 0xFF804020;  // 不透明暗橙色
    tjs_uint32 src = 0x80FF0000;  // 半透明红色 (alpha=128)

    printf("===== 迷你混合器 =====\n\n");
    printf("输入像素:\n");
    print_pixel("DST (暗橙)", dst);
    print_pixel("SRC (半透明红)", src);

    // 基础模式
    printf("\n--- 基础混合模式 ---\n");
    print_pixel("bmAlpha",  blend_alpha_func{}(dst, src));
    print_pixel("bmAdd",    blend_add_func{}(dst, src));
    print_pixel("bmSub",    blend_sub_func{}(dst, src));

    // 使用 normal_op 模板的模式
    printf("\n--- 模板变体模式 ---\n");
    print_pixel("Mul+alpha", normal_op<blend_mul_func>{}(dst, src));
    print_pixel("Screen+alpha", normal_op<blend_screen_func>{}(dst, src));
    print_pixel("Diff+alpha", normal_op<blend_diff_func>{}(dst, src));

    // HDA 变体
    printf("\n--- HDA 变体 (注意 Alpha 通道) ---\n");
    print_pixel("Alpha+HDA", hda_op<blend_alpha_func>{{}}(dst, src));
    print_pixel("Mul+HDA",   hda_op<blend_mul_func>{{}}(dst, src));

    // Opacity 变体
    printf("\n--- Opacity 变体 (opacity=64, 即25%%) ---\n");
    print_pixel("Alpha+opa64", opacity_op<blend_alpha_func>(64)(dst, src));
    print_pixel("Mul+opa64",
                opacity_op<blend_mul_func>(64)(dst, src));

    // 批量扫描线处理
    printf("\n--- 扫描线批量处理 ---\n");
    const int WIDTH = 8;
    tjs_uint32 scanline_dst[WIDTH];
    tjs_uint32 scanline_src[WIDTH];

    // 创建渐变扫描线
    for (int i = 0; i < WIDTH; i++) {
        int v = i * 255 / (WIDTH - 1);            // 0→255 渐变
        scanline_dst[i] = make_pixel(255, v, v, v);  // 灰度渐变
        scanline_src[i] = 0x80FF0000;                 // 统一半透明红色
    }

    // 混合前
    printf("混合前 DST (灰度渐变):\n  ");
    for (int i = 0; i < WIDTH; i++)
        printf("%3d ", ch_r(scanline_dst[i]));
    printf("\n");

    // 执行混合
    blend_scanline(scanline_dst, scanline_src, WIDTH,
                   blend_alpha_func{});

    // 混合后
    printf("混合后 (bmAlpha):\n  R: ");
    for (int i = 0; i < WIDTH; i++)
        printf("%3d ", ch_r(scanline_dst[i]));
    printf("\n  G: ");
    for (int i = 0; i < WIDTH; i++)
        printf("%3d ", ch_g(scanline_dst[i]));
    printf("\n");

    // 性能测试
    printf("\n--- 性能测试 (100万像素 × 100次) ---\n");
    const int PIXELS = 1000000;
    const int ITERS = 100;
    auto* perf_dst = new tjs_uint32[PIXELS];
    auto* perf_src = new tjs_uint32[PIXELS];

    // 初始化测试数据
    for (int i = 0; i < PIXELS; i++) {
        perf_dst[i] = 0xFF804020;
        perf_src[i] = 0x80FF0000;
    }

    auto start = std::chrono::high_resolution_clock::now();
    for (int iter = 0; iter < ITERS; iter++) {
        blend_scanline(perf_dst, perf_src, PIXELS, blend_alpha_func{});
    }
    auto end = std::chrono::high_resolution_clock::now();

    double ms = std::chrono::duration<double, std::milli>(end - start).count();
    double mpix_per_sec = (double)PIXELS * ITERS / ms / 1000.0;
    printf("bmAlpha: %.1f ms, %.1f M像素/秒\n", ms, mpix_per_sec);

    // 同样测试 bmMul
    for (int i = 0; i < PIXELS; i++) perf_dst[i] = 0xFF804020;
    start = std::chrono::high_resolution_clock::now();
    for (int iter = 0; iter < ITERS; iter++) {
        blend_scanline(perf_dst, perf_src, PIXELS,
                       normal_op<blend_mul_func>{});
    }
    end = std::chrono::high_resolution_clock::now();
    ms = std::chrono::duration<double, std::milli>(end - start).count();
    mpix_per_sec = (double)PIXELS * ITERS / ms / 1000.0;
    printf("bmMul:   %.1f ms, %.1f M像素/秒\n", ms, mpix_per_sec);

    delete[] perf_dst;
    delete[] perf_src;

    printf("\n===== 完成 =====\n");
    return 0;
}
```

## 2. 饱和加法的位运算详解

饱和加法（Saturated Addition）是 Alpha 混合子系统中最巧妙的位运算之一。它需要对 4 个字节同时做 `min(a + b, 255)`，且不使用任何分支。

### 2.1 朴素实现（有分支）

```cpp
// 朴素版：4 个 if 判断
tjs_uint32 sat_add_naive(tjs_uint32 a, tjs_uint32 b) {
    auto clamp = [](int v) -> uint8_t { return v > 255 ? 255 : (uint8_t)v; };
    return make_pixel(
        clamp(ch_a(a) + ch_a(b)),
        clamp(ch_r(a) + ch_r(b)),
        clamp(ch_g(a) + ch_g(b)),
        clamp(ch_b(a) + ch_b(b))
    );
}
```

这需要 4 次比较分支。在紧密的像素处理循环中，分支预测失败的代价（Branch Misprediction Penalty，CPU 猜错执行路径时需要丢弃已预取的指令并重新开始，通常浪费 10~20 个时钟周期）会严重影响性能。

### 2.2 KrKr2 的无分支版本

```cpp
tjs_uint32 saturated_add(tjs_uint32 a, tjs_uint32 b) {
    // 步骤 1: 检测溢出
    tjs_uint32 tmp = ((a & b) + (((a ^ b) >> 1) & 0x7f7f7f7f)) & 0x80808080;
    // 步骤 2: 溢出标志→全 FF 掩码
    tmp = (tmp << 1) - (tmp >> 7);
    // 步骤 3: 饱和
    return (a + b - tmp) | tmp;
}
```

逐步分析步骤 1：

```
目标: 判断每个字节 a[i] + b[i] 是否 > 255

观察: 两个 8-bit 数相加溢出的条件是进位到第 9 位
      即 bit 7 是否产生进位

用半加器原理:
  carry  = a & b          // 必定产生进位的位
  maybe  = a ^ b          // 可能产生进位的位
  
  完整进位链太复杂，KrKr2 用近似:
  tmp = carry + (maybe >> 1)  // 近似估算高位进位
  
  只看每个字节的 bit 7 (0x80808080):
  如果 bit 7 为 1 → 该字节会溢出
```

步骤 2 将溢出标志扩展为掩码：

```
tmp = 0x80  →  (0x80 << 1) - (0x80 >> 7)  =  0x100 - 0x01  =  0xFF
tmp = 0x00  →  (0x00 << 1) - (0x00 >> 7)  =  0x000 - 0x00  =  0x00
```

步骤 3 应用掩码：

```
溢出字节:   (a+b-0xFF) | 0xFF = 0xFF       ← 饱和到 255
未溢出字节: (a+b-0x00) | 0x00 = a+b        ← 保持正常结果
```

### 2.3 验证示例

```cpp
#include <cstdint>
#include <cstdio>

using tjs_uint32 = uint32_t;

tjs_uint32 saturated_add(tjs_uint32 a, tjs_uint32 b) {
    tjs_uint32 tmp = ((a & b) + (((a ^ b) >> 1) & 0x7f7f7f7f)) & 0x80808080;
    tmp = (tmp << 1) - (tmp >> 7);
    return (a + b - tmp) | tmp;
}

tjs_uint32 sat_add_naive(tjs_uint32 a, tjs_uint32 b) {
    auto cl = [](int v) -> uint8_t { return v > 255 ? 255 : (uint8_t)v; };
    int a0 = (a>>24)&0xff, a1 = (a>>16)&0xff, a2 = (a>>8)&0xff, a3 = a&0xff;
    int b0 = (b>>24)&0xff, b1 = (b>>16)&0xff, b2 = (b>>8)&0xff, b3 = b&0xff;
    return ((uint32_t)cl(a0+b0)<<24) | ((uint32_t)cl(a1+b1)<<16)
         | ((uint32_t)cl(a2+b2)<<8)  |  (uint32_t)cl(a3+b3);
}

int main() {
    // 测试用例：包含溢出和不溢出的混合情况
    tjs_uint32 test_cases[][2] = {
        {0x80804020, 0x80804020},  // A,R 溢出; G,B 不溢出
        {0xFFFFFFFF, 0x01010101},  // 全溢出
        {0x00000000, 0x00000000},  // 全零
        {0x7F7F7F7F, 0x7F7F7F7F},  // 边界：254，不溢出
        {0x80808080, 0x80808080},  // 边界：256，溢出
    };

    printf("=== 饱和加法验证 ===\n");
    for (auto& tc : test_cases) {
        tjs_uint32 fast  = saturated_add(tc[0], tc[1]);
        tjs_uint32 naive = sat_add_naive(tc[0], tc[1]);
        printf("0x%08X + 0x%08X = 0x%08X  %s\n",
               tc[0], tc[1], fast,
               fast == naive ? "OK" : "MISMATCH!");
    }
    return 0;
}
```

### 3.4 KrKr2 源码中的优化差异

我们的实现是教学版本，KrKr2 源码中有以下额外优化：

1. **mul_blend_functor** 不使用 `/255` 而用位移 `>> 8`，且把 3 个通道用位操作并行处理为 2 步
2. **screen_blend_functor** 使用预计算的 `TVPNegativeMulTable[256*256]` 查找表
3. **所有函数** 通过函数指针实现运行时分发，启动时可替换为 SSE2/NEON 加速版本
4. **batch 函数** 展开了循环（Loop Unrolling，减少循环控制开销），每次迭代处理 4 个像素

## 4. 扩展练习：自定义混合模式

### 4.1 实现 Overlay 混合

以下代码展示如何为 Overlay 模式编写 functor，并使用我们的模板系统生成所有变体：

```cpp
#include <cstdint>
#include <cstdio>

using tjs_uint32 = uint32_t;

// Overlay functor
struct overlay_blend_func {
    static inline uint8_t blend_ch(uint8_t d, uint8_t s) {
        if (d < 128) {
            return (uint8_t)(2 * (uint32_t)d * s / 255);
        } else {
            return (uint8_t)(255 - 2 * (uint32_t)(255 - d) * (255 - s) / 255);
        }
    }
    inline tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        uint8_t r = blend_ch((d >> 16) & 0xff, (s >> 16) & 0xff);
        uint8_t g = blend_ch((d >> 8) & 0xff, (s >> 8) & 0xff);
        uint8_t b = blend_ch(d & 0xff, s & 0xff);
        return ((uint32_t)r << 16) | ((uint32_t)g << 8) | b;
    }
};

// alpha_blend 工具
tjs_uint32 alpha_blend(tjs_uint32 d, tjs_uint32 s, tjs_uint32 a) {
    tjs_uint32 d1 = d & 0xff00ff;
    d1 = (d1 + (((s & 0xff00ff) - d1) * a >> 8)) & 0xff00ff;
    d &= 0xff00;
    s &= 0xff00;
    return d1 + ((d + ((s - d) * a >> 8)) & 0xff00);
}

// 直接使用 normal_op 模板
template<typename FUNC>
struct normal_op {
    FUNC func_;
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        tjs_uint32 a = s >> 24;
        if (a == 0) return d;
        tjs_uint32 blended = func_(d, s);
        if (a == 255) return blended;
        return alpha_blend(d, blended, a);
    }
};

int main() {
    tjs_uint32 dark_dst  = 0xFF402010;  // 暗色目标
    tjs_uint32 light_dst = 0xFFC0A080;  // 亮色目标
    tjs_uint32 src       = 0x80808080;  // 半透明中灰色

    normal_op<overlay_blend_func> overlay;

    printf("=== Overlay 混合效果 ===\n");
    printf("暗色背景 + 中灰源:\n");
    tjs_uint32 r1 = overlay(dark_dst, src);
    printf("  暗背景: R=%3d → Overlay后: R=%3d (变更暗)\n",
           (dark_dst >> 16) & 0xff, (r1 >> 16) & 0xff);

    printf("亮色背景 + 中灰源:\n");
    tjs_uint32 r2 = overlay(light_dst, src);
    printf("  亮背景: R=%3d → Overlay后: R=%3d (变更亮)\n",
           (light_dst >> 16) & 0xff, (r2 >> 16) & 0xff);

    printf("\n这就是 Overlay 的特点：暗部更暗，亮部更亮，增强对比度\n");
    return 0;
}
```

## 常见错误及解决方案

### 错误 1：饱和加法使用了有分支的版本导致性能差

**症状**：混合 100 万像素的 bmAdd 比 bmAlpha 慢 3-5 倍。

**原因**：使用 `std::min(a+b, 255)` 的朴素饱和加法，每个字节一次分支判断，4 字节 4 次分支。

**解决**：使用无分支版本 `saturated_add()`。在现代 CPU 上，无分支版本通常快 2-4 倍。

### 错误 2：模板变体的 alpha 处理重复

**症状**：`bmMul` 模式下，半透明源的混合结果过于接近目标色（混合效果太弱）。

**原因**：在 functor 内部已经做了 alpha 混合，`normal_op` 模板又做了一次 alpha 混合 → alpha 被应用了两次。

```cpp
// ❌ 错误：functor 内已混合 alpha
struct bad_mul_func {
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        tjs_uint32 a = s >> 24;
        tjs_uint32 mul = mul_result(d, s);
        return alpha_blend(d, mul, a);  // alpha 在这里用了一次
    }
};
// normal_op<bad_mul_func> 又会用 alpha 混合一次 → 重复！

// ✅ 正确：functor 只计算纯颜色运算
struct good_mul_func {
    tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
        return mul_result(d, s);  // 只返回颜色，不处理 alpha
    }
};
// normal_op<good_mul_func> 负责 alpha → 只应用一次
```

### 错误 3：性能测试在 Debug 模式下运行

**症状**：性能数据显示仅 5-10 M像素/秒，远低于预期的 200+ M像素/秒。

**原因**：Debug 模式下编译器不做优化（`-O0`），模板不会被内联，每次 `operator()` 调用都有函数调用开销。

**解决**：确保用 Release 模式编译（`-DCMAKE_BUILD_TYPE=Release`）。Release 模式下编译器会内联所有模板函数，性能可提升 10-50 倍。

### 题目 1：实现饱和减法的无分支版本

KrKr2 的 `sub_blend_functor`（`blend_functor_c.h` 第 202-207 行）使用纯位运算实现了饱和减法 `max(D + S - 255, 0)`。请：

(a) 用带分支的方式实现，作为参考
(b) 分析 KrKr2 无分支版本的每一步
(c) 解释为什么公式是 `(s + d - tmp) & tmp` 而不是 `(s + d - tmp) | tmp`（与饱和加法不同）

<details>
<summary>查看答案</summary>

```cpp
// (a) 带分支版本
tjs_uint32 sub_blend_naive(tjs_uint32 d, tjs_uint32 s) {
    auto sub = [](uint8_t a, uint8_t b) -> uint8_t {
        int r = (int)a + (int)b - 255;
        return r < 0 ? 0 : (uint8_t)r;
    };
    return ((uint32_t)sub(ch_r(d), ch_r(s)) << 16)
         | ((uint32_t)sub(ch_g(d), ch_g(s)) << 8)
         |  (uint32_t)sub(ch_b(d), ch_b(s));
}

// (b) KrKr2 无分支版本分析:
// struct sub_blend_functor {
//     tjs_uint32 operator()(tjs_uint32 d, tjs_uint32 s) const {
//         tjs_uint32 tmp = ((s & d) + (((s ^ d) >> 1) & 0x7f7f7f7f))
//                          & 0x80808080;
//         tmp = (tmp << 1) - (tmp >> 7);
//         return (s + d - tmp) & tmp;
//     }
// };
//
// 步骤 1: 与饱和加法相同——检测 s+d 是否 >= 256（溢出）
//   tmp = 溢出标志 (bit 7 of each byte)
//
// 步骤 2: 扩展为掩码
//   溢出字节 → 0xFF, 未溢出字节 → 0x00
//
// 步骤 3: 与饱和加法不同的关键！
//   return (s + d - tmp) & tmp;
//
//   溢出字节 (tmp=0xFF):
//     (s+d-0xFF) & 0xFF = (s+d-255) & 0xFF
//     → 就是 max(s+d-255, 0) 的结果（因为 s+d >= 256）
//
//   未溢出字节 (tmp=0x00):
//     (s+d-0x00) & 0x00 = 0
//     → 未溢出意味着 s+d < 256，即 s+d-255 < 0
//     → 饱和到 0，正确！
```

**(c)** 饱和加法用 `| tmp`（溢出时结果=0xFF），饱和减法用 `& tmp`（未溢出时结果=0）。这是因为：
- 饱和加法的上界是 255 → 溢出时用 `|0xFF` 设为 255
- 饱和减法的下界是 0 → 未溢出时用 `&0x00` 设为 0

</details>

### 题目 2：Alpha 混合的结合律

三个图层 A（底层）、B（中层）、C（顶层）按顺序叠加。以下两种计算方式结果是否相同？

方式 1：先 A+B，再 (A+B)+C
方式 2：先 B+C，再 A+(B+C)

<details>
<summary>查看答案</summary>

**不同！Alpha 混合不满足结合律。**

数学证明：
```
SrcOver(A, SrcOver(B, C)) ≠ SrcOver(SrcOver(A, B), C)

设 A=(Ra, αa), B=(Rb, αb), C=(Rc, αc)

方式 1: result = C + (B + A(1-αb))(1-αc)
方式 2: result = (C + B(1-αc)) + A(1-(αb + αc(1-αb)))

展开后两个表达式不等。
```

直觉理解：先叠加 B+C 再叠加到 A 上，C 的颜色会被 B 的 alpha 影响一次；但先叠加 A+B 再叠加 C，C 的颜色完全不受 B 影响。顺序不同导致结果不同。

**实际影响**：在 KrKr2 中，图层必须严格从底到顶逐层合成，不能随意调换合成顺序。引擎内部的 `LayerTreeOwner` 管理图层树，保证合成顺序正确。

但在**预乘格式**下，SrcOver 满足结合律——这是预乘格式的另一个优势（可以任意分组并行合成）。

</details>

### 题目 3：性能分析

根据 KrKr2 的 `sBmFactor` 权重表，bmDodge 的因子是 10，bmAlpha 的因子是 52。

(a) 如果处理 100 万像素的 bmAlpha 需要 5ms，估算处理同样像素数的 bmDodge 需要多少毫秒
(b) 假设线程创建开销为 0.5ms，4 核 CPU，处理 100 万像素的 bmDodge 是否值得用多线程？
(c) 处理 100 万像素的 bmPsExclusion（因子=66）呢？

<details>
<summary>查看答案</summary>

**(a)** 性能权重因子与速度成正比：
```
time_dodge = time_alpha × (factor_alpha / factor_dodge)
           = 5ms × (52 / 10)
           = 26ms
```
bmDodge 大约需要 **26ms**。

**(b)** 多线程分析：
```
单线程: 26ms
4线程理想加速: 26ms / 4 = 6.5ms
加上线程开销: 6.5ms + 0.5ms = 7.0ms

加速比: 26ms / 7.0ms = 3.7x
```
**值得用多线程**。26ms 降到 7ms，加速 3.7 倍。

**(c)** bmPsExclusion 分析：
```
time_exclusion = 5ms × (52 / 66) = 3.9ms
单线程已经只需 3.9ms
4线程: 3.9/4 + 0.5 = 1.5ms
加速比: 3.9/1.5 = 2.6x
```
可以用多线程但收益较小。实际中 KrKr2 可能根据像素数量设置阈值——比如少于 10 万像素时 bmPsExclusion 就不启用多线程。

</details>
