### 3.1.2 media_probe.c 完整源码

```c
// media_probe.c —— 简化版 ffprobe，展示文件信息、流信息和包统计
#include <libavformat/avformat.h>
#include <libavcodec/avcodec.h>
#include <libavutil/avutil.h>
#include <libavutil/dict.h>
#include <stdio.h>
#include <inttypes.h>

// ── 辅助函数：格式化时长 ──
// 将微秒（AV_TIME_BASE 单位）转为 HH:MM:SS.mmm 格式的可读字符串
static void format_duration(int64_t duration_us, char *buf, size_t buf_size) {
    if (duration_us == AV_NOPTS_VALUE) {
        // AV_NOPTS_VALUE 是 FFmpeg 的"无效时间戳"常量，
        // 当容器不包含时长信息时 duration 字段会设为此值
        snprintf(buf, buf_size, "N/A");
        return;
    }
    int64_t total_sec = duration_us / AV_TIME_BASE;
    int ms = (int)((duration_us % AV_TIME_BASE) * 1000 / AV_TIME_BASE);
    int h = (int)(total_sec / 3600);
    int m = (int)((total_sec % 3600) / 60);
    int s = (int)(total_sec % 60);
    snprintf(buf, buf_size, "%02d:%02d:%02d.%03d", h, m, s, ms);
}

// ── 辅助函数：打印元数据 ──
// 遍历 AVDictionary（FFmpeg 的键值对容器）中的所有条目
static void print_metadata(AVDictionary *metadata, const char *indent) {
    if (!metadata) return;
    AVDictionaryEntry *tag = NULL;
    // av_dict_get 用空字符串和 AV_DICT_IGNORE_SUFFIX 标志来遍历所有条目
    while ((tag = av_dict_get(metadata, "", tag, AV_DICT_IGNORE_SUFFIX))) {
        printf("%s%s: %s\n", indent, tag->key, tag->value);
    }
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <媒体文件路径或URL>\n", argv[0]);
        fprintf(stderr, "示例: %s test.mp4\n", argv[0]);
        fprintf(stderr, "      %s http://example.com/stream.m3u8\n", argv[0]);
        return 1;
    }

    const char *filename = argv[1];

    // 旧版 FFmpeg（< 4.0）需要手动注册解码器和格式
    // 新版自动注册，但为了兼容性可以加上版本检查
    #if LIBAVFORMAT_VERSION_INT < AV_VERSION_INT(58, 9, 100)
    av_register_all();       // 注册所有容器格式和编码器
    avformat_network_init(); // 初始化网络协议（HTTP/RTSP 等）
    #endif

    // ═══════════════════════════════════════
    // 步骤 1：打开文件
    // ═══════════════════════════════════════
    AVFormatContext *fmt_ctx = NULL;
    int ret = avformat_open_input(&fmt_ctx, filename, NULL, NULL);
    if (ret < 0) {
        char errbuf[AV_ERROR_MAX_STRING_SIZE];
        av_strerror(ret, errbuf, sizeof(errbuf));
        fprintf(stderr, "错误: 无法打开 '%s': %s\n", filename, errbuf);
        return 1;
    }

    // ═══════════════════════════════════════
    // 步骤 2：探测流信息
    // ═══════════════════════════════════════
    if (avformat_find_stream_info(fmt_ctx, NULL) < 0) {
        fprintf(stderr, "错误: 无法获取流信息\n");
        avformat_close_input(&fmt_ctx);
        return 1;
    }

    // ═══════════════════════════════════════
    // 文件级信息
    // ═══════════════════════════════════════
    char duration_str[64];
    format_duration(fmt_ctx->duration, duration_str, sizeof(duration_str));

    printf("═══════════════════════════════════════\n");
    printf("文件  : %s\n", filename);
    printf("格式  : %s (%s)\n", fmt_ctx->iformat->name,
           fmt_ctx->iformat->long_name);
    printf("时长  : %s\n", duration_str);
    printf("码率  : %"PRId64" kbps\n",
           fmt_ctx->bit_rate > 0 ? fmt_ctx->bit_rate / 1000 : 0);
    printf("流数  : %u\n", fmt_ctx->nb_streams);

    // 文件级元数据（标题、作者、编码工具等）
    if (fmt_ctx->metadata) {
        printf("\n── 元数据 ──\n");
        print_metadata(fmt_ctx->metadata, "  ");
    }

    // ═══════════════════════════════════════
    // 步骤 3：逐流信息
    // ═══════════════════════════════════════
    printf("\n── 流信息 ──\n");
    for (unsigned int i = 0; i < fmt_ctx->nb_streams; i++) {
        AVStream *stream = fmt_ctx->streams[i];
        AVCodecParameters *par = stream->codecpar;

        printf("\n流 #%u:\n", i);

        switch (par->codec_type) {
        case AVMEDIA_TYPE_VIDEO:
            printf("  类型    : 视频\n");
            printf("  编码    : %s (profile: %d)\n",
                   avcodec_get_name(par->codec_id), par->profile);
            printf("  分辨率  : %dx%d\n", par->width, par->height);
            printf("  像素格式: %s\n",
                   av_get_pix_fmt_name((enum AVPixelFormat)par->format));
            printf("  帧率    : %.2f fps\n", av_q2d(stream->avg_frame_rate));
            printf("  码率    : %"PRId64" kbps\n", par->bit_rate / 1000);
            break;

        case AVMEDIA_TYPE_AUDIO:
            printf("  类型    : 音频\n");
            printf("  编码    : %s\n", avcodec_get_name(par->codec_id));
            printf("  采样率  : %d Hz\n", par->sample_rate);
            printf("  声道数  : %d\n", par->channels);
            printf("  采样格式: %s\n",
                   av_get_sample_fmt_name((enum AVSampleFormat)par->format));
            printf("  码率    : %"PRId64" kbps\n", par->bit_rate / 1000);
            break;

        case AVMEDIA_TYPE_SUBTITLE:
            printf("  类型    : 字幕\n");
            printf("  编码    : %s\n", avcodec_get_name(par->codec_id));
            break;

        default:
            printf("  类型    : 其他 (%d)\n", par->codec_type);
        }

        // 时间基准和时长
        printf("  time_base: %d/%d\n",
               stream->time_base.num, stream->time_base.den);

        char stream_dur[64];
        if (stream->duration != AV_NOPTS_VALUE) {
            // 将流的 duration 转为 AV_TIME_BASE 单位再格式化
            int64_t dur_us = av_rescale_q(stream->duration,
                                           stream->time_base,
                                           AV_TIME_BASE_Q);
            format_duration(dur_us, stream_dur, sizeof(stream_dur));
        } else {
            snprintf(stream_dur, sizeof(stream_dur), "N/A");
        }
        printf("  时长    : %s\n", stream_dur);

        // 流级元数据（语言、标题等）
        if (stream->metadata) {
            printf("  元数据:\n");
            print_metadata(stream->metadata, "    ");
        }
    }

    // ═══════════════════════════════════════
    // 步骤 4 & 5：包统计
    // ═══════════════════════════════════════
    printf("\n── 包统计 ──\n");
    AVPacket pkt;
    int total = 0;
    int per_stream[64] = {0};     // 每条流的包数量（上限 64 条流）
    int key_frames[64] = {0};     // 每条流的关键帧数量
    int64_t bytes[64] = {0};      // 每条流的数据量

    while (av_read_frame(fmt_ctx, &pkt) >= 0) {
        total++;
        if (pkt.stream_index < 64) {
            per_stream[pkt.stream_index]++;
            bytes[pkt.stream_index] += pkt.size;
            if (pkt.flags & AV_PKT_FLAG_KEY) {
                key_frames[pkt.stream_index]++;
            }
        }
        av_packet_unref(&pkt);    // 必须释放引用
    }

    printf("总包数: %d\n", total);
    for (unsigned int i = 0; i < fmt_ctx->nb_streams && i < 64; i++) {
        const char *type = "其他";
        switch (fmt_ctx->streams[i]->codecpar->codec_type) {
        case AVMEDIA_TYPE_VIDEO:    type = "视频"; break;
        case AVMEDIA_TYPE_AUDIO:    type = "音频"; break;
        case AVMEDIA_TYPE_SUBTITLE: type = "字幕"; break;
        default: break;
        }
        printf("  流 #%u [%s]: %d 包, 关键帧 %d, %.2f MB\n",
               i, type, per_stream[i], key_frames[i],
               bytes[i] / 1048576.0);
    }

    printf("═══════════════════════════════════════\n");

    // ═══════════════════════════════════════
    // 步骤 6：关闭文件
    // ═══════════════════════════════════════
    avformat_close_input(&fmt_ctx);
    return 0;
}
```

## 3.2 随机访问（Seek）

### 3.2.1 什么是 Seek？

Seek（随机访问，又叫跳转）是指让播放器跳转到视频的任意时间位置。例如用户拖动进度条从第 10 秒跳到第 60 秒，播放器就需要执行 Seek 操作。

在 FFmpeg 中，Seek 使用 `av_seek_frame()` 函数：

```c
int av_seek_frame(
    AVFormatContext *s,        // 格式上下文
    int stream_index,          // 目标流索引（-1 = 使用默认时间基准 AV_TIME_BASE）
    int64_t timestamp,         // 目标时间戳（单位取决于 stream_index）
    int flags                  // Seek 标志
);
// 返回值：≥0 = 成功，负数 = 错误
```

### 3.2.2 基本 Seek 示例

```c
// 跳转到第 30 秒位置
// stream_index=-1 时，timestamp 使用 AV_TIME_BASE（微秒）单位
int64_t target_ts = 30 * AV_TIME_BASE;  // 30 秒 = 30,000,000 微秒
int ret = av_seek_frame(fmt_ctx,
                         -1,             // -1 = 使用 AV_TIME_BASE 作为时间单位
                         target_ts,
                         AVSEEK_FLAG_BACKWARD);
if (ret < 0) {
    char errbuf[AV_ERROR_MAX_STRING_SIZE];
    av_strerror(ret, errbuf, sizeof(errbuf));
    fprintf(stderr, "Seek 失败: %s\n", errbuf);
}
```

也可以针对特定流进行 Seek（使用该流的 time_base 作为时间单位）：

```c
// 针对视频流 Seek 到第 30 秒
AVStream *video_stream = fmt_ctx->streams[video_stream_idx];
// 将 30 秒转换为该流的 time_base 单位
int64_t target_pts = (int64_t)(30.0 / av_q2d(video_stream->time_base));
int ret = av_seek_frame(fmt_ctx, video_stream_idx, target_pts,
                         AVSEEK_FLAG_BACKWARD);
```

### 3.2.3 Seek 标志详解

`av_seek_frame()` 的第四个参数是标志位（Flags），控制 Seek 的行为。可以用按位或 `|` 组合多个标志：

| 标志 | 值 | 含义 | 使用场景 |
|------|---|------|---------|
| `AVSEEK_FLAG_BACKWARD` | 1 | 跳到目标时间**之前**最近的关键帧 | 最常用，播放器默认行为 |
| `AVSEEK_FLAG_BYTE` | 2 | 按**字节偏移**（而非时间戳）Seek | 损坏的文件、无索引的流 |
| `AVSEEK_FLAG_ANY` | 4 | 允许跳到**非关键帧** | 极少使用，会导致花屏 |
| `AVSEEK_FLAG_FRAME` | 8 | 按**帧号**（而非时间戳）Seek | 视频编辑工具 |

```c
// 最常用的 Seek 方式：跳到目标之前的关键帧
av_seek_frame(fmt_ctx, -1, target_ts, AVSEEK_FLAG_BACKWARD);

// 跳到目标之后的关键帧
av_seek_frame(fmt_ctx, -1, target_ts, 0);  // 不带 BACKWARD 标志

// 按字节偏移 Seek（用于无时间戳的流或损坏文件）
av_seek_frame(fmt_ctx, -1, byte_offset, AVSEEK_FLAG_BYTE);
```

### 3.2.4 为什么 Seek 只能跳到关键帧？

视频编码中有三种帧类型：

| 帧类型 | 全称 | 特点 |
|--------|------|------|
| **I 帧** | Intra-coded Frame（帧内编码帧） | 独立完整的图像，不依赖其他帧即可解码。又叫关键帧 |
| **P 帧** | Predicted Frame（前向预测帧） | 只存储与前一帧的差异，需要参考前面的 I/P 帧 |
| **B 帧** | Bidirectional Frame（双向预测帧） | 存储与前后帧的差异，需要参考前后的 I/P 帧 |

如果直接从 P 帧或 B 帧开始解码，解码器没有它所依赖的参考帧，输出的画面就是花屏（马赛克、绿屏、色块混乱）。因此，Seek 必须跳到关键帧（I 帧），让解码器从一个完整图像开始工作。

```
                GOP（画面组）
├───────────────────────────────────┤
I    P    B    B    P    B    B    I    P ...
↑                                  ↑
可以 Seek 到这里                    也可以 Seek 到这里
    中间的 P/B 帧不能直接开始解码
```

> **KrKr2 中的处理**：`CDVDDemuxFFmpeg::SeekTime()` 使用 `AVSEEK_FLAG_BACKWARD` 跳到目标之前的关键帧，然后 `BasePlayer::SeekTime()`（在 `VideoPlayer.cpp` 中）调用 `FlushBuffers()` 清空解码器缓冲区中的旧数据，避免旧帧的残影影响新画面。

### 3.2.5 精确 Seek

`av_seek_frame()` 只能跳到关键帧，但用户可能想要精确到某一帧。精确 Seek 的策略是：先跳到目标之前的关键帧，然后逐帧解码并丢弃，直到到达目标位置。

```c
// 精确 Seek 到指定秒数
// 策略：Seek 到关键帧 → 逐包读取并解码 → 跳过不需要的帧 → 到达目标
void seek_precise(AVFormatContext *fmt_ctx, int stream_idx,
                  double target_sec) {
    AVStream *stream = fmt_ctx->streams[stream_idx];

    // 1. 将目标秒数转换为流的 time_base 单位
    int64_t target_ts = (int64_t)(target_sec / av_q2d(stream->time_base));

    // 2. Seek 到目标之前的关键帧
    int ret = av_seek_frame(fmt_ctx, stream_idx, target_ts,
                             AVSEEK_FLAG_BACKWARD);
    if (ret < 0) {
        fprintf(stderr, "Seek 失败\n");
        return;
    }

    // 3. 逐包读取，跳过关键帧到目标之间的包
    AVPacket pkt;
    int skipped = 0;
    while (av_read_frame(fmt_ctx, &pkt) >= 0) {
        if (pkt.stream_index == stream_idx) {
            if (pkt.pts >= target_ts) {
                // 到达或超过目标位置——这就是我们要的帧
                double actual_sec = pkt.pts * av_q2d(stream->time_base);
                printf("精确 Seek 成功: 目标=%.3f秒, 实际=%.3f秒, "
                       "跳过了 %d 个包\n",
                       target_sec, actual_sec, skipped);
                av_packet_unref(&pkt);
                break;
            }
            // 还没到目标，跳过这个包
            // 实际播放器中需要将此包送入解码器（但不渲染），
            // 以推进解码器的参考帧状态
            skipped++;
        }
        av_packet_unref(&pkt);
    }
}
```

使用示例：

```c
// 精确 Seek 到 45.5 秒处
seek_precise(fmt_ctx, video_stream_idx, 45.5);
// 输出类似：精确 Seek 成功: 目标=45.500秒, 实际=45.533秒, 跳过了 43 个包
```

### 3.2.6 `avformat_seek_file` —— 更精确的 Seek API

FFmpeg 还提供了一个增强版的 Seek 函数，可以指定时间范围：

```c
int avformat_seek_file(
    AVFormatContext *s,
    int stream_index,
    int64_t min_ts,        // 最小可接受时间戳
    int64_t ts,            // 目标时间戳
    int64_t max_ts,        // 最大可接受时间戳
    int flags
);
```

这个函数确保 Seek 结果落在 `[min_ts, max_ts]` 范围内：

```c
// Seek 到第 30 秒附近，允许 ±1 秒的误差
int64_t target = 30 * AV_TIME_BASE;
avformat_seek_file(fmt_ctx, -1,
                   target - 1 * AV_TIME_BASE,  // 最早 29 秒
                   target,                       // 目标 30 秒
                   target + 1 * AV_TIME_BASE,   // 最晚 31 秒
                   0);
```

## 3.3 自定义 I/O——从内存或归档读取

### 3.3.1 为什么需要自定义 I/O？

FFmpeg 默认从文件系统读取数据——调用操作系统的 `fopen()`/`fread()` 等函数。但有些场景中，视频数据不在普通文件里：

| 场景 | 说明 |
|------|------|
| **归档文件** | KrKr2 的游戏资源存在 XP3 归档中，FFmpeg 不认识 XP3 格式 |
| **内存缓冲** | 从网络下载的视频数据暂存在内存中，还没写入文件 |
| **加密文件** | 视频文件被加密过，需要先解密再喂给 FFmpeg |
| **Android assets** | Android 的 assets 目录不是标准文件系统路径，不能用 `fopen` 打开 |
| **嵌入式资源** | 视频数据编译到程序可执行文件中（通过 `xxd` 或资源文件） |

FFmpeg 通过 `AVIOContext`（Audio-Video I/O Context，音视频输入输出上下文）抽象了 I/O 层。你可以创建自定义的 `AVIOContext`，提供读取回调函数和 Seek 回调函数，让 FFmpeg 从任意数据源读取。

### 3.3.2 AVIOContext 的创建

```c
AVIOContext *avio_alloc_context(
    unsigned char *buffer,     // I/O 缓冲区（由 av_malloc 分配）
    int buffer_size,           // 缓冲区大小（推荐 4096-32768 字节）
    int write_flag,            // 0 = 只读, 1 = 可写
    void *opaque,              // 用户数据指针（会传给回调函数）
    int (*read_packet)(void *opaque, uint8_t *buf, int buf_size),   // 读回调
    int (*write_packet)(void *opaque, uint8_t *buf, int buf_size),  // 写回调（只读时传 NULL）
    int64_t (*seek)(void *opaque, int64_t offset, int whence)       // Seek 回调
);
```

### 3.3.3 完整示例：从 FILE* 自定义读取

以下示例演示如何用自定义 I/O 替代 FFmpeg 的默认文件读取。虽然这个例子仍然读文件（因为我们用 `FILE*`），但它展示了完整的自定义 I/O 模式——把 `FILE*` 换成 XP3 的流读取接口就是 KrKr2 的做法。

```c
// custom_io_demo.c —— 自定义 I/O 读取示例
#include <libavformat/avformat.h>
#include <libavformat/avio.h>
#include <libavutil/avutil.h>
#include <stdio.h>
#include <inttypes.h>

// ── 自定义读取回调 ──
// FFmpeg 需要数据时会调用这个函数
// opaque: 我们传入的 FILE* 指针
// buf:    FFmpeg 提供的缓冲区，我们需要往里面填数据
// buf_size: 缓冲区大小，表示 FFmpeg 最多想要多少字节
// 返回值: 实际读取的字节数，文件结束返回 AVERROR_EOF，错误返回负数
static int custom_read(void *opaque, uint8_t *buf, int buf_size) {
    FILE *fp = (FILE *)opaque;
    int bytes_read = (int)fread(buf, 1, buf_size, fp);
    if (bytes_read == 0) {
        // 没读到数据 → 文件结束
        return AVERROR_EOF;
    }
    return bytes_read;
}

// ── 自定义 Seek 回调 ──
// FFmpeg 需要在数据源中跳转位置时调用
// offset: 目标偏移量
// whence: SEEK_SET / SEEK_CUR / SEEK_END（与 fseek 相同）
//         特殊值 AVSEEK_SIZE：FFmpeg 想知道数据源的总大小
// 返回值: 新的文件位置（字节），或者 -1 表示错误
static int64_t custom_seek(void *opaque, int64_t offset, int whence) {
    FILE *fp = (FILE *)opaque;

    if (whence == AVSEEK_SIZE) {
        // 返回文件总大小（FFmpeg 用这个信息来计算进度等）
        long current_pos = ftell(fp);    // 保存当前位置
        fseek(fp, 0, SEEK_END);          // 跳到文件末尾
        long file_size = ftell(fp);      // 获取位置 = 文件大小
        fseek(fp, current_pos, SEEK_SET);// 恢复原来的位置
        return (int64_t)file_size;
    }

    // 标准 Seek 操作
    if (fseek(fp, (long)offset, whence) < 0) {
        return -1;  // Seek 失败
    }
    return (int64_t)ftell(fp);  // 返回新位置
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <媒体文件>\n", argv[0]);
        return 1;
    }

    // 用标准 C 库打开文件（模拟自定义数据源）
    FILE *fp = fopen(argv[1], "rb");
    if (!fp) {
        fprintf(stderr, "无法打开文件: %s\n", argv[1]);
        return 1;
    }

    // ── 分配 I/O 缓冲区 ──
    // 这块内存会交给 AVIOContext 管理，我们不要手动释放它
    const int io_buf_size = 4096;  // 4KB 缓冲区
    uint8_t *io_buf = (uint8_t *)av_malloc(io_buf_size);
    if (!io_buf) {
        fprintf(stderr, "内存分配失败\n");
        fclose(fp);
        return 1;
    }

    // ── 创建自定义 I/O 上下文 ──
    AVIOContext *avio_ctx = avio_alloc_context(
        io_buf,         // I/O 缓冲区
        io_buf_size,    // 缓冲区大小
        0,              // write_flag = 0（只读）
        fp,             // opaque = 我们的 FILE* 指针
        custom_read,    // 读回调函数
        NULL,           // 写回调 = NULL（不需要写）
        custom_seek     // Seek 回调函数
    );
    if (!avio_ctx) {
        fprintf(stderr, "创建 AVIOContext 失败\n");
        av_free(io_buf);
        fclose(fp);
        return 1;
    }

    // ── 创建 AVFormatContext 并关联自定义 I/O ──
    AVFormatContext *fmt_ctx = avformat_alloc_context();
    if (!fmt_ctx) {
        fprintf(stderr, "创建 AVFormatContext 失败\n");
        av_free(avio_ctx->buffer);
        avio_context_free(&avio_ctx);
        fclose(fp);
        return 1;
    }
    fmt_ctx->pb = avio_ctx;  // 关键：将自定义 I/O 上下文赋给 pb 字段

    // ── 打开（注意：不传文件名，因为 I/O 已经自定义） ──
    int ret = avformat_open_input(&fmt_ctx, NULL, NULL, NULL);
    if (ret < 0) {
        char errbuf[AV_ERROR_MAX_STRING_SIZE];
        av_strerror(ret, errbuf, sizeof(errbuf));
        fprintf(stderr, "打开失败: %s\n", errbuf);
        // 注意：avformat_open_input 失败时会释放 fmt_ctx
        // 但不会释放我们手动设置的 avio_ctx
        av_free(avio_ctx->buffer);
        avio_context_free(&avio_ctx);
        fclose(fp);
        return 1;
    }

    // 探测流信息
    avformat_find_stream_info(fmt_ctx, NULL);

    // 打印信息
    printf("通过自定义 I/O 成功打开!\n");
    printf("格式: %s\n", fmt_ctx->iformat->name);
    printf("流数: %u\n", fmt_ctx->nb_streams);
    printf("时长: %.2f 秒\n", fmt_ctx->duration / (double)AV_TIME_BASE);

    // 读取前 5 个包作为验证
    AVPacket pkt;
    int count = 0;
    while (av_read_frame(fmt_ctx, &pkt) >= 0 && count < 5) {
        printf("  包 #%d: stream=%d, size=%d, pts=%"PRId64"\n",
               count, pkt.stream_index, pkt.size, pkt.pts);
        av_packet_unref(&pkt);
        count++;
    }

    // ── 清理（顺序很重要！） ──
    avformat_close_input(&fmt_ctx);    // 1. 关闭格式上下文（会释放 streams 等）
    // 注意：avformat_close_input 不会释放我们手动创建的 avio_ctx
    av_free(avio_ctx->buffer);         // 2. 释放 I/O 缓冲区
    avio_context_free(&avio_ctx);      // 3. 释放 AVIOContext
    fclose(fp);                        // 4. 关闭底层文件

    return 0;
}
```

### 3.3.4 KrKr2 中的自定义 I/O

KrKr2 的游戏视频存储在 XP3 归档（KiriKiri 引擎的原生归档格式，参见 `cpp/core/base/XP3Archive.cpp`）中。FFmpeg 不认识 XP3 格式，因此 KrKr2 使用自定义 I/O 让 FFmpeg 能够从归档中读取视频数据：

```
KrKr2 视频文件的读取路径：

TVPMoviePlayer::BuildGraph()
  → BasePlayer::OpenInputStream()
    → CDVDDemuxFFmpeg::Open(CDVDInputStream *pInput)
      │
      │  // 1. 创建自定义 I/O 上下文
      ├─→ avio_alloc_context(
      │       buffer, buffer_size, 0,
      │       pInput,              // opaque = KrKr2 的输入流接口
      │       read_callback,       // 内部调用 pInput->Read()
      │       NULL,
      │       seek_callback        // 内部调用 pInput->Seek()
      │   )
      │
      │  // 2. 关联到格式上下文
      ├─→ m_pFormatContext->pb = m_ioContext;
      │
      │  // 3. 正常打开（FFmpeg 透过自定义 I/O 读取 XP3 归档内的视频数据）
      └─→ avformat_open_input(&m_pFormatContext, ...)
```

> **源码位置**：`cpp/core/movie/ffmpeg/DemuxFFmpeg.cpp` 的 `Open()` 方法（约第 100-200 行）。
>
> `read_callback` 内部通过 KrKr2 的 `CDVDInputStream` 接口（继承自 `IStream`）读取 XP3 归档中的视频数据。对 FFmpeg 来说，它并不知道数据来自 XP3 归档——它只是通过回调函数获取字节流，然后正常解析容器格式。

## 3.4 动手实践：给 media_probe 添加 Seek 功能

在 `media_probe.c` 的基础上，添加一个 `-seek` 参数，让用户可以 Seek 到指定时间后读取几个包：

```c
// seek_probe.c —— media_probe + Seek 功能
#include <libavformat/avformat.h>
#include <libavcodec/avcodec.h>
#include <libavutil/avutil.h>
#include <stdio.h>
#include <stdlib.h>
#include <inttypes.h>

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <媒体文件> [-seek <秒数>]\n", argv[0]);
        return 1;
    }

    const char *filename = argv[1];
    double seek_target = -1.0;

    // 解析 -seek 参数
    for (int i = 2; i < argc - 1; i++) {
        if (strcmp(argv[i], "-seek") == 0) {
            seek_target = atof(argv[i + 1]);
        }
    }

    // 打开文件
    AVFormatContext *fmt_ctx = NULL;
    if (avformat_open_input(&fmt_ctx, filename, NULL, NULL) < 0) {
        fprintf(stderr, "无法打开文件\n");
        return 1;
    }
    avformat_find_stream_info(fmt_ctx, NULL);

    // 找到视频流
    int video_idx = av_find_best_stream(fmt_ctx, AVMEDIA_TYPE_VIDEO,
                                         -1, -1, NULL, 0);
    if (video_idx < 0) {
        fprintf(stderr, "没有视频流\n");
        avformat_close_input(&fmt_ctx);
        return 1;
    }

    AVStream *video_stream = fmt_ctx->streams[video_idx];
    printf("视频流 #%d: %s, %dx%d, time_base=%d/%d\n",
           video_idx, avcodec_get_name(video_stream->codecpar->codec_id),
           video_stream->codecpar->width, video_stream->codecpar->height,
           video_stream->time_base.num, video_stream->time_base.den);

    // 如果指定了 Seek 目标
    if (seek_target >= 0.0) {
        printf("\n── Seek 到 %.2f 秒 ──\n", seek_target);

        // 转换目标时间为视频流的 time_base 单位
        int64_t target_pts = (int64_t)(seek_target /
                                        av_q2d(video_stream->time_base));

        // Seek 到目标之前的关键帧
        int ret = av_seek_frame(fmt_ctx, video_idx, target_pts,
                                 AVSEEK_FLAG_BACKWARD);
        if (ret < 0) {
            char errbuf[AV_ERROR_MAX_STRING_SIZE];
            av_strerror(ret, errbuf, sizeof(errbuf));
            fprintf(stderr, "Seek 失败: %s\n", errbuf);
        } else {
            printf("Seek 成功，读取后续包：\n");
        }
    }

    // 读取接下来的 10 个视频包
    AVPacket pkt;
    int count = 0;
    while (av_read_frame(fmt_ctx, &pkt) >= 0 && count < 10) {
        if (pkt.stream_index == video_idx) {
            double pts_sec = pkt.pts * av_q2d(video_stream->time_base);
            printf("  视频包 #%d: pts=%.3f秒, size=%d, key=%s\n",
                   count, pts_sec, pkt.size,
                   (pkt.flags & AV_PKT_FLAG_KEY) ? "是" : "否");
            count++;
        }
        av_packet_unref(&pkt);
    }

    avformat_close_input(&fmt_ctx);
    return 0;
}
```

编译和运行：

```bash

# 不带 Seek：从头读取前 10 个视频包

# Seek 到第 60 秒：观察 Seek 后第一个包是否为关键帧

```

预期输出（Seek 到 60 秒）：


## 常见错误及解决方案

### 错误 1：Seek 后画面花屏（马赛克/绿屏）

**现象**：`av_seek_frame()` 后解码出的前几帧有严重的画面破损

**原因**：Seek 改变了读取位置，但解码器内部的缓冲区中还残留着旧位置的参考帧数据。解码器试图用旧的参考帧来解码新位置的 P/B 帧，导致画面混乱

**解决**：Seek 后**必须**调用 `avcodec_flush_buffers()` 清空解码器的内部缓冲区：

```c
av_seek_frame(fmt_ctx, -1, target_ts, AVSEEK_FLAG_BACKWARD);

// 清空解码器缓冲区（第 4 章详讲解码器的使用）
avcodec_flush_buffers(video_codec_ctx);  // 清空视频解码器
avcodec_flush_buffers(audio_codec_ctx);  // 清空音频解码器
```

### 错误 2：自定义 I/O 的清理顺序错误导致崩溃

**现象**：程序退出时 `avformat_close_input()` 之后访问已释放的内存，导致 Segfault

**原因**：自定义 I/O 的清理顺序必须是：close_input → 释放 buffer → 释放 avio_ctx → 关闭数据源。顺序错误会导致双重释放或悬空指针

**解决**：严格按照以下顺序清理：

```c
// 正确的清理顺序：
avformat_close_input(&fmt_ctx);    // 1. 关闭格式上下文
av_free(avio_ctx->buffer);         // 2. 释放 I/O 缓冲区
avio_context_free(&avio_ctx);      // 3. 释放 AVIOContext
fclose(fp);                        // 4. 关闭底层数据源
```

### 错误 3：自定义 read 回调返回 0 导致 FFmpeg 死循环

**现象**：自定义 I/O 读到文件末尾时 FFmpeg 陷入无限循环

**原因**：read 回调在文件结束时返回了 `0`。FFmpeg 把 `0` 解释为"读取了 0 字节但没有错误"，于是不断重试

**解决**：文件结束时必须返回 `AVERROR_EOF`（不是 0）：

```c
static int custom_read(void *opaque, uint8_t *buf, int buf_size) {
    FILE *fp = (FILE *)opaque;
    int n = (int)fread(buf, 1, buf_size, fp);
    if (n == 0) return AVERROR_EOF;  // ← 必须返回 AVERROR_EOF，不是 0！
    return n;
}
```

## 本节小结

- 完整的 `media_probe` 程序展示了解封装六步流程的实际运用
- `av_seek_frame()` 实现随机访问，常用 `AVSEEK_FLAG_BACKWARD` 跳到关键帧
- Seek 只能跳到关键帧（I 帧），因为 P/B 帧依赖参考帧才能正确解码
- 精确 Seek 需要先跳到关键帧，再逐帧解码跳过中间的帧
- `AVIOContext` 允许自定义 I/O——通过 read/seek 回调从任意数据源读取
- KrKr2 通过自定义 I/O 让 FFmpeg 透明地从 XP3 归档中读取视频文件

### 题目 1：为什么 Seek 到非关键帧会导致花屏？如果一定要 Seek 到精确位置，应该怎么做？

<details>
<summary>查看答案</summary>

**为什么花屏**：

视频中的 P 帧和 B 帧不是完整图像，它们只存储与参考帧的差异。如果直接从 P/B 帧开始解码：
1. 解码器没有所需的参考帧（I 帧或之前的 P 帧）
2. 差异数据无法正确还原为完整画面
3. 输出的像素数据是随机/错误的，表现为马赛克、绿屏或色块

**精确 Seek 方法**（两步法）：

1. **第一步**：使用 `AVSEEK_FLAG_BACKWARD` Seek 到目标之前最近的关键帧
2. **第二步**：从关键帧开始，逐包读取并送入解码器。解码器输出的帧中，PTS < 目标时间的帧全部丢弃（不渲染），直到遇到 PTS ≥ 目标时间的帧，这就是我们要显示的精确帧

```c
// 伪代码
av_seek_frame(ctx, stream_idx, target_pts, AVSEEK_FLAG_BACKWARD);
avcodec_flush_buffers(codec_ctx);

while (av_read_frame(ctx, &pkt) >= 0) {
    if (pkt.stream_index == target_stream) {
        avcodec_send_packet(codec_ctx, &pkt);
        while (avcodec_receive_frame(codec_ctx, frame) >= 0) {
            if (frame->pts >= target_pts) {
                // 这是我们要的精确帧——渲染它
                render_frame(frame);
                goto done;
            }
            // 否则丢弃（不渲染但必须解码以更新参考帧）
        }
    }
    av_packet_unref(&pkt);
}
```

</details>

### 题目 2：修改 `custom_io_demo.c`，实现从内存缓冲区（而非 FILE*）读取视频数据。假设整个视频文件已经加载到内存中。

<details>
<summary>查看答案</summary>

```c
#include <libavformat/avformat.h>
#include <libavformat/avio.h>
#include <stdio.h>
#include <string.h>
#include <inttypes.h>

// 内存缓冲区的上下文
typedef struct {
    uint8_t *data;       // 数据指针
    size_t   size;       // 数据总大小
    size_t   pos;        // 当前读取位置
} MemoryBuffer;

// 从内存读取的回调
static int mem_read(void *opaque, uint8_t *buf, int buf_size) {
    MemoryBuffer *mb = (MemoryBuffer *)opaque;

    // 计算剩余可读字节数
    size_t remaining = mb->size - mb->pos;
    if (remaining == 0) return AVERROR_EOF;

    // 读取 min(buf_size, remaining) 字节
    size_t to_read = buf_size < (int)remaining ? (size_t)buf_size : remaining;
    memcpy(buf, mb->data + mb->pos, to_read);
    mb->pos += to_read;
    return (int)to_read;
}

// 内存 Seek 的回调
static int64_t mem_seek(void *opaque, int64_t offset, int whence) {
    MemoryBuffer *mb = (MemoryBuffer *)opaque;

    if (whence == AVSEEK_SIZE) {
        return (int64_t)mb->size;          // 返回总大小
    }

    int64_t new_pos;
    switch (whence) {
    case SEEK_SET: new_pos = offset; break;
    case SEEK_CUR: new_pos = (int64_t)mb->pos + offset; break;
    case SEEK_END: new_pos = (int64_t)mb->size + offset; break;
    default: return -1;
    }

    if (new_pos < 0 || new_pos > (int64_t)mb->size) return -1;
    mb->pos = (size_t)new_pos;
    return new_pos;
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <媒体文件>\n", argv[0]);
        return 1;
    }

    // 将整个文件读入内存
    FILE *fp = fopen(argv[1], "rb");
    if (!fp) { perror("fopen"); return 1; }
    fseek(fp, 0, SEEK_END);
    long file_size = ftell(fp);
    fseek(fp, 0, SEEK_SET);

    uint8_t *file_data = (uint8_t *)malloc(file_size);
    fread(file_data, 1, file_size, fp);
    fclose(fp);

    printf("已将 %ld 字节加载到内存\n", file_size);

    // 创建内存缓冲区上下文
    MemoryBuffer mb = { .data = file_data, .size = (size_t)file_size, .pos = 0 };

    // 创建自定义 I/O
    const int io_buf_size = 4096;
    uint8_t *io_buf = (uint8_t *)av_malloc(io_buf_size);
    AVIOContext *avio_ctx = avio_alloc_context(
        io_buf, io_buf_size, 0, &mb, mem_read, NULL, mem_seek);

    AVFormatContext *fmt_ctx = avformat_alloc_context();
    fmt_ctx->pb = avio_ctx;

    if (avformat_open_input(&fmt_ctx, NULL, NULL, NULL) < 0) {
        fprintf(stderr, "打开失败\n");
        av_free(avio_ctx->buffer);
        avio_context_free(&avio_ctx);
        free(file_data);
        return 1;
    }

    avformat_find_stream_info(fmt_ctx, NULL);
    printf("格式: %s, 流数: %u, 时长: %.2f秒\n",
           fmt_ctx->iformat->name, fmt_ctx->nb_streams,
           fmt_ctx->duration / (double)AV_TIME_BASE);

    // 清理
    avformat_close_input(&fmt_ctx);
    av_free(avio_ctx->buffer);
    avio_context_free(&avio_ctx);
    free(file_data);
    return 0;
}
```

</details>

### 题目 3：KrKr2 为什么需要自定义 I/O？如果不用自定义 I/O，还有什么替代方案？

<details>
<summary>查看答案</summary>

**为什么需要自定义 I/O**：

KrKr2 的游戏资源（包括视频文件）存储在 **XP3 归档**中。XP3 是 KiriKiri 引擎的专有归档格式（类似 ZIP，但结构不同）。FFmpeg 的 `avformat_open_input()` 默认通过操作系统的文件 API 读取文件，它不认识 XP3 格式，无法直接从归档内部读取视频数据。

因此 KrKr2 使用自定义 `AVIOContext`，通过 read/seek 回调函数将 KrKr2 自己的流读取接口（`CDVDInputStream`，内部调用 XP3 归档的解压缩逻辑）桥接到 FFmpeg 的 I/O 层。

**替代方案**：

1. **先解压再播放**：将视频从 XP3 归档解压到临时文件，再让 FFmpeg 打开临时文件。缺点：需要额外磁盘空间和 I/O 时间，且需要管理临时文件的创建和删除。
2. **内存映射**：将归档中的视频数据解压到内存，用内存 I/O（如本题练习 2）读取。缺点：视频文件可能很大（数百 MB），全部加载到内存不现实。
3. **注册自定义协议**：FFmpeg 支持注册自定义 URL 协议（如 `xp3://file.xp3/video.mp4`），但这需要编译自定义的 FFmpeg 版本，不如自定义 I/O 灵活。

自定义 `AVIOContext` 是最佳方案：按需读取（不用全部加载到内存）、不需要临时文件、不需要修改 FFmpeg 源码。

</details>
