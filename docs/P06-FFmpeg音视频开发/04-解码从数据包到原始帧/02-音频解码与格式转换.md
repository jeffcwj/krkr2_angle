# 4.3 音频解码完整流程
## 4.3 音频解码完整流程

### 4.3.1 音频解码与视频解码的差异

音频解码在 API 层面与视频解码几乎相同（新 API 共用 `send_packet`/`receive_frame`），但在数据组织上有重要差异：

| 特性 | 视频 | 音频 |
|------|------|------|
| 输出数据 | 像素矩阵 (width × height) | PCM 采样序列 (nb_samples × channels) |
| 数据布局 | 平面式 YUV（Y/U/V 分离）或打包式 RGB | 交错式或平面式 |
| 帧大小 | 固定（取决于分辨率） | 可变（每帧 samples 数不同） |
| `frame->data` | `[0]=Y [1]=U [2]=V` | 交错：`[0]=全部`；平面：`[i]=第i声道` |
| 典型大小 | 几百 KB ~ 几 MB | 几 KB ~ 几十 KB |

### 4.3.2 音频采样格式详解

FFmpeg 定义了多种音频采样格式（`AVSampleFormat`），分为**交错**（interleaved）和**平面**（planar）两大类：

```cpp
// 交错格式 — 所有声道数据交替存储
// [L0][R0][L1][R1][L2][R2]...
AV_SAMPLE_FMT_U8      // unsigned 8-bit
AV_SAMPLE_FMT_S16     // signed 16-bit (CD 质量)
AV_SAMPLE_FMT_S32     // signed 32-bit
AV_SAMPLE_FMT_FLT     // 32-bit float
AV_SAMPLE_FMT_DBL     // 64-bit double

// 平面格式 — 每个声道数据分别存储
// data[0]: [L0][L1][L2]...
// data[1]: [R0][R1][R2]...
AV_SAMPLE_FMT_U8P     // unsigned 8-bit planar
AV_SAMPLE_FMT_S16P    // signed 16-bit planar
AV_SAMPLE_FMT_S32P    // signed 32-bit planar
AV_SAMPLE_FMT_FLTP    // 32-bit float planar（最常见！AAC/Vorbis 等默认输出）
AV_SAMPLE_FMT_DBLP    // 64-bit double planar
```

> **重要**：大多数现代编解码器（AAC、Vorbis、Opus）默认输出**平面格式**（planar），但大多数音频 API（OpenAL、SDL、Windows WASAPI）期望**交错格式**。因此通常需要使用 libswresample 进行格式转换。

### 4.3.3 完整音频解码示例

```cpp
#include <cstdio>
extern "C" {
#include <libavcodec/avcodec.h>
#include <libavformat/avformat.h>
#include <libavutil/opt.h>
}

int main() {
    AVFormatContext *fmtCtx = nullptr;
    avformat_open_input(&fmtCtx, "input.mp4", nullptr, nullptr);
    avformat_find_stream_info(fmtCtx, nullptr);

    // 找到音频流
    int audioIdx = av_find_best_stream(fmtCtx, AVMEDIA_TYPE_AUDIO, -1, -1, nullptr, 0);
    if (audioIdx < 0) {
        fprintf(stderr, "未找到音频流\n");
        return -1;
    }

    // 查找解码器并打开
    AVCodecParameters *par = fmtCtx->streams[audioIdx]->codecpar;
    AVCodec *codec = avcodec_find_decoder(par->codec_id);
    AVCodecContext *ctx = avcodec_alloc_context3(codec);
    avcodec_parameters_to_context(ctx, par);
    avcodec_open2(ctx, codec, nullptr);

    printf("音频: %d Hz, %d 声道, 格式: %s\n",
           ctx->sample_rate, ctx->channels,
           av_get_sample_fmt_name(ctx->sample_fmt));

    // 解码循环
    AVPacket *pkt = av_packet_alloc();
    AVFrame *frame = av_frame_alloc();
    int totalSamples = 0;

    while (av_read_frame(fmtCtx, pkt) >= 0) {
        if (pkt->stream_index != audioIdx) {
            av_packet_unref(pkt);
            continue;
        }

        avcodec_send_packet(ctx, pkt);
        while (avcodec_receive_frame(ctx, frame) >= 0) {
            totalSamples += frame->nb_samples;

            // 判断是平面还是交错格式
            int isPlanar = av_sample_fmt_is_planar(ctx->sample_fmt);
            int bytesPerSample = av_get_bytes_per_sample(ctx->sample_fmt);

            printf("  音频帧: %d 采样, pts=%lld, %s, %d bytes/sample\n",
                   frame->nb_samples,
                   (long long)frame->pts,
                   isPlanar ? "平面" : "交错",
                   bytesPerSample);

            if (isPlanar) {
                // 平面格式：每个声道在 frame->data[i] 中
                for (int ch = 0; ch < ctx->channels; ch++) {
                    // frame->data[ch] 指向第 ch 个声道的数据
                    // 数据长度 = nb_samples * bytesPerSample
                    uint8_t *channelData = frame->data[ch];
                    int dataSize = frame->nb_samples * bytesPerSample;
                    // 处理 channelData...
                    (void)channelData;
                    (void)dataSize;
                }
            } else {
                // 交错格式：所有声道数据在 frame->data[0] 中
                uint8_t *interleavedData = frame->data[0];
                int dataSize = frame->nb_samples * ctx->channels * bytesPerSample;
                // 处理 interleavedData...
                (void)interleavedData;
                (void)dataSize;
            }

            av_frame_unref(frame);
        }
        av_packet_unref(pkt);
    }

    // 刷新解码器
    avcodec_send_packet(ctx, nullptr);
    while (avcodec_receive_frame(ctx, frame) >= 0) {
        totalSamples += frame->nb_samples;
        av_frame_unref(frame);
    }

    double duration = (double)totalSamples / ctx->sample_rate;
    printf("共解码 %d 采样 (%.2f 秒)\n", totalSamples, duration);

    av_frame_free(&frame);
    av_packet_free(&pkt);
    avcodec_free_context(&ctx);
    avformat_close_input(&fmtCtx);
    return 0;
}
```

## 4.4 像素格式转换（libswscale）与音频重采样（libswresample）

### 4.4.1 为什么需要格式转换

解码器输出的原始数据格式通常不是应用程序直接需要的：

- **视频**：解码器输出 YUV420P，但渲染器可能需要 RGB24 或 RGBA
- **音频**：解码器输出 FLTP（float planar），但播放器 API 需要 S16（signed 16-bit interleaved）

### 4.4.2 视频像素格式转换 — sws_scale

```cpp
#include <cstdio>
extern "C" {
#include <libavcodec/avcodec.h>
#include <libavutil/imgutils.h>
#include <libswscale/swscale.h>
}

// 假设 codecCtx 和 frame 已经通过解码获得

void convertYUVtoRGB(AVCodecContext *codecCtx, AVFrame *yuvFrame) {
    int width = codecCtx->width;
    int height = codecCtx->height;

    // 第一步：创建转换上下文
    struct SwsContext *swsCtx = sws_getContext(
        width, height, codecCtx->pix_fmt,     // 源：解码器输出格式
        width, height, AV_PIX_FMT_RGB24,      // 目标：RGB24
        SWS_BILINEAR,                          // 缩放算法（此处无缩放）
        nullptr, nullptr, nullptr
    );
    if (!swsCtx) {
        fprintf(stderr, "无法创建 SwsContext\n");
        return;
    }

    // 第二步：分配目标缓冲区
    uint8_t *dstData[4] = {nullptr};
    int dstLinesize[4] = {0};
    int bufSize = av_image_alloc(
        dstData, dstLinesize,
        width, height, AV_PIX_FMT_RGB24,
        1  // 对齐值
    );
    if (bufSize < 0) {
        fprintf(stderr, "无法分配图像缓冲区\n");
        sws_freeContext(swsCtx);
        return;
    }

    // 第三步：执行转换
    sws_scale(
        swsCtx,
        yuvFrame->data,        // 源数据
        yuvFrame->linesize,    // 源行宽
        0, height,             // 源切片：从第 0 行到第 height 行
        dstData,               // 目标数据
        dstLinesize            // 目标行宽
    );

    printf("转换完成: %dx%d RGB24, 行宽=%d, 总大小=%d bytes\n",
           width, height, dstLinesize[0], bufSize);

    // dstData[0] 现在包含 RGB24 像素数据
    // 可以写入文件、传给 OpenGL 纹理等

    // 清理
    av_freep(&dstData[0]);
    sws_freeContext(swsCtx);
}
```

> **性能提示**：`SwsContext` 应当复用而非每帧重新创建。在 KrKr2 中，渲染管线通过 `CRenderManager` 统一管理格式转换。

### 4.4.3 音频重采样 — swr_convert

```cpp
#include <cstdio>
extern "C" {
#include <libswresample/swresample.h>
#include <libavutil/opt.h>
#include <libavutil/channel_layout.h>
}

// 将解码器输出的 FLTP 44100Hz 立体声
// 转换为 S16 交错格式 48000Hz 立体声
void resampleAudio(AVFrame *inputFrame) {
    SwrContext *swrCtx = swr_alloc();
    if (!swrCtx) {
        fprintf(stderr, "无法分配 SwrContext\n");
        return;
    }

    // 设置输入参数
    av_opt_set_int(swrCtx, "in_channel_layout",  AV_CH_LAYOUT_STEREO, 0);
    av_opt_set_int(swrCtx, "in_sample_rate",      44100, 0);
    av_opt_set_sample_fmt(swrCtx, "in_sample_fmt", AV_SAMPLE_FMT_FLTP, 0);

    // 设置输出参数
    av_opt_set_int(swrCtx, "out_channel_layout", AV_CH_LAYOUT_STEREO, 0);
    av_opt_set_int(swrCtx, "out_sample_rate",     48000, 0);
    av_opt_set_sample_fmt(swrCtx, "out_sample_fmt", AV_SAMPLE_FMT_S16, 0);

    // 初始化
    if (swr_init(swrCtx) < 0) {
        fprintf(stderr, "无法初始化 SwrContext\n");
        swr_free(&swrCtx);
        return;
    }

    // 计算输出采样数（考虑采样率变换）
    int outSamples = av_rescale_rnd(
        swr_get_delay(swrCtx, 44100) + inputFrame->nb_samples,
        48000,   // 目标采样率
        44100,   // 源采样率
        AV_ROUND_UP
    );

    // 分配输出缓冲区
    uint8_t *outBuf = nullptr;
    int outBufSize = av_samples_alloc(
        &outBuf, nullptr,
        2,              // 声道数
        outSamples,     // 采样数
        AV_SAMPLE_FMT_S16,
        0               // 对齐
    );

    // 执行重采样
    int convertedSamples = swr_convert(
        swrCtx,
        &outBuf,                  // 输出缓冲区
        outSamples,               // 输出最大采样数
        (const uint8_t **)inputFrame->data,  // 输入数据
        inputFrame->nb_samples    // 输入采样数
    );

    printf("重采样: %d -> %d 采样 (44100Hz FLTP -> 48000Hz S16)\n",
           inputFrame->nb_samples, convertedSamples);

    // outBuf 现在包含 S16 交错格式的 PCM 数据
    // 可以直接传给 OpenAL/SDL 等音频 API

    av_freep(&outBuf);
    swr_free(&swrCtx);
}
```

### 4.4.4 常见错误与解决方案

**错误 1：解码后画面偏绿/偏色**

```
原因：直接将 YUV 数据当作 RGB 使用，或 linesize 处理错误
解决：必须通过 sws_scale 正确转换；注意 linesize 可能大于 width × bytesPerPixel
```

**错误 2：音频播放有杂音/噪声**

```
原因：平面格式数据未转为交错格式就直接送给播放器
解决：检查 av_sample_fmt_is_planar()，若为平面格式则用 swr_convert 转换
     或手动交错（如 KrKr2 的 FFWaveDecoder::CopySmaples 所做）
```

**错误 3：sws_getContext 返回 NULL**

```
原因：不支持的像素格式组合
解决：先用 sws_isSupportedInput/sws_isSupportedOutput 检查格式支持
     在 Windows 上确保链接了 libswscale
```
