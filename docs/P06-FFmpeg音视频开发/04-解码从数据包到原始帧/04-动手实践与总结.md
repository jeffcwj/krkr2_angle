## 实践一：视频帧提取器（save_frames）

### 目标与原理

我们的第一个实践项目是一个**视频帧提取器**：打开一个视频文件，解码前 N 帧，将每帧转换为 RGB 格式后保存为 PPM 图片。PPM（Portable Pixmap，便携式像素图）是一种极其简单的图片格式——文件头只有三行文本（魔数、宽高、最大色值），后面紧跟原始 RGB 像素数据，不需要任何压缩/解压库即可读写。

完整流程如下：

```
打开文件 → 查找视频流 → 初始化解码器 → 初始化格式转换器
    → 解码循环 { 读包 → 送入解码器 → 取出帧 → YUV→RGB → 保存PPM }
    → Flush 解码器 → 释放资源
```

### 完整代码

```cpp
// save_frames.cpp — 提取视频前 N 帧并保存为 PPM 格式
// 编译依赖：libavcodec, libavformat, libavutil, libswscale
#include <cstdio>
#include <cstdlib>
#include <cstring>

extern "C" {
#include <libavcodec/avcodec.h>
#include <libavformat/avformat.h>
#include <libavutil/imgutils.h>    // av_image_alloc — 分配图像缓冲区
#include <libswscale/swscale.h>    // sws_getContext / sws_scale — 像素格式转换
}

// 将一帧 RGB24 数据保存为 PPM 文件
// PPM 格式：P6 魔数 + 宽高 + 最大色值 255 + 逐行 RGB 数据
static void savePPM(AVFrame *frame, int width, int height, int idx) {
    char filename[64];
    snprintf(filename, sizeof(filename), "frame_%03d.ppm", idx);

    FILE *fp = fopen(filename, "wb");
    if (!fp) {
        fprintf(stderr, "错误：无法创建文件 %s\n", filename);
        return;
    }

    // PPM 文件头（P6 = 二进制 RGB 格式）
    fprintf(fp, "P6\n%d %d\n255\n", width, height);

    // 逐行写入像素数据
    // 注意：frame->linesize[0] 可能大于 width*3（存在行对齐 padding），
    // 所以必须逐行写入 width*3 字节，而不是一次写入整个 data[0]
    for (int y = 0; y < height; y++) {
        fwrite(frame->data[0] + y * frame->linesize[0],
               1, width * 3, fp);
    }

    fclose(fp);
    printf("已保存: %s (%dx%d)\n", filename, width, height);
}

// 错误处理辅助宏 — 打印 FFmpeg 错误码对应的文字描述
static void printError(const char *msg, int errnum) {
    char buf[256];
    av_strerror(errnum, buf, sizeof(buf));  // 将错误码转为可读字符串
    fprintf(stderr, "%s: %s\n", msg, buf);
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <视频文件> [帧数=10]\n", argv[0]);
        fprintf(stderr, "示例: %s test.mp4 20\n", argv[0]);
        return 1;
    }

    const char *inputFile = argv[1];
    int maxFrames = (argc >= 3) ? atoi(argv[2]) : 10;  // 默认提取 10 帧
    int ret = 0;

    // ========== 第一步：打开输入文件 ==========
    AVFormatContext *fmtCtx = nullptr;
    ret = avformat_open_input(&fmtCtx, inputFile, nullptr, nullptr);
    if (ret < 0) {
        printError("无法打开输入文件", ret);
        return 1;
    }

    // 读取流信息（某些格式的流信息不在文件头中，需要读几个包才能确定）
    ret = avformat_find_stream_info(fmtCtx, nullptr);
    if (ret < 0) {
        printError("无法获取流信息", ret);
        avformat_close_input(&fmtCtx);
        return 1;
    }

    // 打印文件基本信息（调试用，第二个参数是流索引，0 = 打印所有流）
    printf("=== 文件信息 ===\n");
    av_dump_format(fmtCtx, 0, inputFile, 0);
    printf("================\n\n");

    // ========== 第二步：查找视频流 ==========
    // av_find_best_stream：自动找到"最佳"视频流
    // 参数：上下文、媒体类型、期望流索引(-1=自动)、相关流(-1=无)、解码器指针、标志
    int videoIdx = av_find_best_stream(fmtCtx, AVMEDIA_TYPE_VIDEO,
                                       -1, -1, nullptr, 0);
    if (videoIdx < 0) {
        fprintf(stderr, "错误：未找到视频流\n");
        avformat_close_input(&fmtCtx);
        return 1;
    }
    printf("视频流索引: %d\n", videoIdx);

    // ========== 第三步：初始化解码器 ==========
    // 获取流的编解码参数（codec_id、宽高、像素格式等）
    AVCodecParameters *par = fmtCtx->streams[videoIdx]->codecpar;

    // 根据 codec_id 查找对应的解码器
    AVCodec *codec = avcodec_find_decoder(par->codec_id);
    if (!codec) {
        fprintf(stderr, "错误：找不到解码器 %s\n",
                avcodec_get_name(par->codec_id));
        avformat_close_input(&fmtCtx);
        return 1;
    }
    printf("解码器: %s (%s)\n", codec->name, codec->long_name);

    // 分配解码器上下文并从流参数填充配置
    AVCodecContext *codecCtx = avcodec_alloc_context3(codec);
    avcodec_parameters_to_context(codecCtx, par);
    codecCtx->thread_count = 4;  // 启用 4 线程并行解码（加速软解）

    // 打开解码器（此时解码器内部会分配所需资源）
    ret = avcodec_open2(codecCtx, codec, nullptr);
    if (ret < 0) {
        printError("无法打开解码器", ret);
        avcodec_free_context(&codecCtx);
        avformat_close_input(&fmtCtx);
        return 1;
    }
    printf("视频尺寸: %dx%d, 像素格式: %s\n",
           codecCtx->width, codecCtx->height,
           av_get_pix_fmt_name(codecCtx->pix_fmt));

    // ========== 第四步：初始化像素格式转换器 ==========
    // sws_getContext：创建像素格式转换上下文
    // 参数：源宽、源高、源格式 → 目标宽、目标高、目标格式、缩放算法、...
    // SWS_BILINEAR（双线性插值）是速度和质量的良好平衡
    struct SwsContext *swsCtx = sws_getContext(
        codecCtx->width, codecCtx->height, codecCtx->pix_fmt,   // 源
        codecCtx->width, codecCtx->height, AV_PIX_FMT_RGB24,    // 目标
        SWS_BILINEAR,    // 缩放算法
        nullptr, nullptr, nullptr  // 源/目标滤波器、额外参数（均不需要）
    );
    if (!swsCtx) {
        fprintf(stderr, "错误：无法创建 SwsContext\n");
        avcodec_free_context(&codecCtx);
        avformat_close_input(&fmtCtx);
        return 1;
    }

    // 分配解码用的 AVFrame 和 RGB 转换后的 AVFrame
    AVFrame *frame = av_frame_alloc();      // 接收解码器输出（YUV 格式）
    AVFrame *rgbFrame = av_frame_alloc();    // 存放 RGB24 转换结果

    // av_image_alloc：为 rgbFrame 分配图像缓冲区
    // 返回值是分配的字节数，rgbFrame->data[] 和 linesize[] 会被填充
    int bufSize = av_image_alloc(
        rgbFrame->data, rgbFrame->linesize,
        codecCtx->width, codecCtx->height,
        AV_PIX_FMT_RGB24,
        1  // 对齐值（1 = 不对齐，PPM 不需要行对齐）
    );
    if (bufSize < 0) {
        printError("无法分配图像缓冲区", bufSize);
        // 清理并退出...
        av_frame_free(&frame);
        av_frame_free(&rgbFrame);
        sws_freeContext(swsCtx);
        avcodec_free_context(&codecCtx);
        avformat_close_input(&fmtCtx);
        return 1;
    }

    AVPacket *pkt = av_packet_alloc();
    int savedFrames = 0;

    // ========== 第五步：解码循环 ==========
    printf("\n开始解码...\n");
    while (av_read_frame(fmtCtx, pkt) >= 0 && savedFrames < maxFrames) {
        // 跳过非视频流的数据包
        if (pkt->stream_index != videoIdx) {
            av_packet_unref(pkt);   // 释放包数据（不释放 AVPacket 结构本身）
            continue;
        }

        // 将压缩数据包送入解码器
        ret = avcodec_send_packet(codecCtx, pkt);
        if (ret < 0) {
            printError("send_packet 失败", ret);
            av_packet_unref(pkt);
            continue;  // 跳过错误包，继续处理下一个
        }

        // 循环取出所有已解码帧（一个 packet 可能产生多帧）
        while (avcodec_receive_frame(codecCtx, frame) >= 0
               && savedFrames < maxFrames) {
            // YUV → RGB24 格式转换
            // sws_scale 参数：上下文、源数据、源行大小、起始行、行数、
            //                 目标数据、目标行大小
            sws_scale(swsCtx,
                      frame->data, frame->linesize,     // 源（YUV）
                      0, codecCtx->height,               // 从第 0 行开始，处理全部行
                      rgbFrame->data, rgbFrame->linesize // 目标（RGB24）
            );

            savePPM(rgbFrame, codecCtx->width, codecCtx->height,
                    ++savedFrames);
            av_frame_unref(frame);  // 释放帧数据（复用 AVFrame 结构）
        }
        av_packet_unref(pkt);
    }

    // ========== 第六步：Flush 解码器 ==========
    // 发送 NULL 包通知解码器输入结束，取出缓冲区中残留的帧
    // 对于 B 帧编码的视频，最后几帧可能还在解码器内部缓冲中
    avcodec_send_packet(codecCtx, nullptr);
    while (avcodec_receive_frame(codecCtx, frame) >= 0
           && savedFrames < maxFrames) {
        sws_scale(swsCtx,
                  frame->data, frame->linesize,
                  0, codecCtx->height,
                  rgbFrame->data, rgbFrame->linesize);
        savePPM(rgbFrame, codecCtx->width, codecCtx->height,
                ++savedFrames);
        av_frame_unref(frame);
    }

    printf("\n完成！共保存 %d 帧\n", savedFrames);

    // ========== 第七步：释放资源 ==========
    // 释放顺序很重要：先释放依赖其他对象的资源
    av_freep(&rgbFrame->data[0]);   // 释放 av_image_alloc 分配的缓冲区
    av_frame_free(&rgbFrame);       // 释放 AVFrame 结构
    av_frame_free(&frame);
    av_packet_free(&pkt);
    sws_freeContext(swsCtx);        // 释放格式转换上下文
    avcodec_free_context(&codecCtx);    // 释放解码器上下文
    avformat_close_input(&fmtCtx);      // 关闭输入文件
    return 0;
}
```

### 关键函数速查

本示例用到的 FFmpeg 函数一览：

| 函数 | 所属库 | 用途 |
|------|--------|------|
| `avformat_open_input` | libavformat | 打开输入文件，分配并填充 `AVFormatContext` |
| `avformat_find_stream_info` | libavformat | 读取数据包以确定流信息（编解码器、时长等） |
| `av_find_best_stream` | libavformat | 自动选择指定类型的"最佳"流 |
| `av_dump_format` | libavformat | 打印文件的详细格式信息到 stderr |
| `avcodec_find_decoder` | libavcodec | 根据 codec_id 查找注册的解码器 |
| `avcodec_alloc_context3` | libavcodec | 分配解码器上下文 |
| `avcodec_parameters_to_context` | libavcodec | 从 `AVCodecParameters` 复制参数到 `AVCodecContext` |
| `avcodec_open2` | libavcodec | 用给定解码器初始化上下文（分配内部资源） |
| `avcodec_send_packet` | libavcodec | 向解码器发送一个压缩数据包（新 API） |
| `avcodec_receive_frame` | libavcodec | 从解码器取出一个已解码帧（新 API） |
| `av_read_frame` | libavformat | 从文件读取下一个数据包 |
| `sws_getContext` | libswscale | 创建像素格式/尺寸转换上下文 |
| `sws_scale` | libswscale | 执行像素格式转换（如 YUV→RGB） |
| `av_image_alloc` | libavutil | 为图像分配对齐的缓冲区 |
| `av_strerror` | libavutil | 将 FFmpeg 错误码转为可读字符串 |
| `av_frame_alloc` / `av_frame_free` | libavutil | 分配/释放 `AVFrame` 结构 |
| `av_frame_unref` | libavutil | 释放帧引用的缓冲区（结构本身保留以复用） |
| `av_packet_alloc` / `av_packet_free` | libavcodec | 分配/释放 `AVPacket` 结构 |
| `av_packet_unref` | libavcodec | 释放包引用的缓冲区 |

## 实践二：音频流信息统计器（audio_info）

### 目标与原理

第二个实践项目是一个**音频信息统计器**：打开媒体文件，找到音频流，解码所有音频帧，统计采样率（Sample Rate）、声道数（Channels）、采样格式（Sample Format）、总帧数、每帧采样数范围等关键参数。这个工具在调试音频问题时非常实用——当音频播放出现杂音、卡顿或静音时，首先要确认音频流的基本参数是否符合预期。

### 完整代码

```cpp
// audio_info.cpp — 统计音频流的详细信息
// 编译依赖：libavcodec, libavformat, libavutil
#include <cstdio>
#include <cstdlib>
#include <climits>

extern "C" {
#include <libavcodec/avcodec.h>
#include <libavformat/avformat.h>
#include <libavutil/samplefmt.h>   // av_get_sample_fmt_name — 采样格式名称
#include <libavutil/channel_layout.h> // av_get_channel_layout_string — 声道布局
}

// 采样格式的字节数（用于计算码率和数据量）
static int getBytesPerSample(enum AVSampleFormat fmt) {
    // av_get_bytes_per_sample：返回每个采样值占用的字节数
    // 对于平面格式（如 FLTP），返回的是单个声道每采样的字节数
    return av_get_bytes_per_sample(fmt);
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <媒体文件>\n", argv[0]);
        fprintf(stderr, "示例: %s music.mp3\n", argv[0]);
        return 1;
    }

    int ret = 0;

    // 打开文件并获取流信息
    AVFormatContext *fmtCtx = nullptr;
    ret = avformat_open_input(&fmtCtx, argv[1], nullptr, nullptr);
    if (ret < 0) {
        char buf[256];
        av_strerror(ret, buf, sizeof(buf));
        fprintf(stderr, "无法打开文件: %s\n", buf);
        return 1;
    }
    avformat_find_stream_info(fmtCtx, nullptr);

    // 查找最佳音频流
    int audioIdx = av_find_best_stream(fmtCtx, AVMEDIA_TYPE_AUDIO,
                                       -1, -1, nullptr, 0);
    if (audioIdx < 0) {
        fprintf(stderr, "错误：文件中没有音频流\n");
        avformat_close_input(&fmtCtx);
        return 1;
    }

    // 初始化音频解码器
    AVCodecParameters *par = fmtCtx->streams[audioIdx]->codecpar;
    AVCodec *codec = avcodec_find_decoder(par->codec_id);
    if (!codec) {
        fprintf(stderr, "错误：找不到音频解码器\n");
        avformat_close_input(&fmtCtx);
        return 1;
    }

    AVCodecContext *codecCtx = avcodec_alloc_context3(codec);
    avcodec_parameters_to_context(codecCtx, par);
    ret = avcodec_open2(codecCtx, codec, nullptr);
    if (ret < 0) {
        fprintf(stderr, "错误：无法打开解码器\n");
        avcodec_free_context(&codecCtx);
        avformat_close_input(&fmtCtx);
        return 1;
    }

    // 统计变量
    AVPacket *pkt = av_packet_alloc();
    AVFrame *frame = av_frame_alloc();
    int frameCount = 0;
    int64_t totalSamples = 0;
    int minSamples = INT_MAX, maxSamples = 0;
    int64_t totalBytes = 0;       // 压缩数据总大小

    // 解码所有音频帧
    while (av_read_frame(fmtCtx, pkt) >= 0) {
        if (pkt->stream_index != audioIdx) {
            av_packet_unref(pkt);
            continue;
        }
        totalBytes += pkt->size;  // 累计压缩数据量

        ret = avcodec_send_packet(codecCtx, pkt);
        if (ret < 0) {
            av_packet_unref(pkt);
            continue;
        }

        while (avcodec_receive_frame(codecCtx, frame) >= 0) {
            frameCount++;
            totalSamples += frame->nb_samples;  // nb_samples：本帧的采样数
            if (frame->nb_samples < minSamples)
                minSamples = frame->nb_samples;
            if (frame->nb_samples > maxSamples)
                maxSamples = frame->nb_samples;
            av_frame_unref(frame);
        }
        av_packet_unref(pkt);
    }

    // Flush 解码器，取出残留帧
    avcodec_send_packet(codecCtx, nullptr);
    while (avcodec_receive_frame(codecCtx, frame) >= 0) {
        frameCount++;
        totalSamples += frame->nb_samples;
        if (frame->nb_samples < minSamples)
            minSamples = frame->nb_samples;
        if (frame->nb_samples > maxSamples)
            maxSamples = frame->nb_samples;
        av_frame_unref(frame);
    }

    // 获取声道布局字符串（如 "stereo"、"5.1" 等）
    char layoutStr[128] = "未知";
    if (codecCtx->channel_layout) {
        av_get_channel_layout_string(layoutStr, sizeof(layoutStr),
                                     codecCtx->channels,
                                     codecCtx->channel_layout);
    }

    // 打印统计结果
    double durationSec = (codecCtx->sample_rate > 0)
        ? (double)totalSamples / codecCtx->sample_rate : 0;
```

## 实践三：视频缩略图生成器（thumbnail）

### 目标与原理

第三个实践项目是一个**视频缩略图生成器**：跳转到视频的指定时间点（如第 30 秒），解码该位置附近的关键帧，缩放为指定尺寸后保存为 PPM 图片。这个工具展示了**随机访问（Seek）+ 解码 + 缩放**的完整链路。

```
打开文件 → 查找视频流 → 初始化解码器 → Seek 到目标时间
    → 解码直到获得关键帧 → 缩放到目标尺寸 → 保存 PPM
```

### 完整代码

```cpp
// thumbnail.cpp — 生成视频指定时间点的缩略图
#include <cstdio>
#include <cstdlib>

extern "C" {
#include <libavcodec/avcodec.h>
#include <libavformat/avformat.h>
#include <libavutil/imgutils.h>
#include <libswscale/swscale.h>
}

static void savePPM(uint8_t *data[], int linesize[], int w, int h,
                    const char *filename) {
    FILE *fp = fopen(filename, "wb");
    if (!fp) return;
    fprintf(fp, "P6\n%d %d\n255\n", w, h);
    for (int y = 0; y < h; y++)
        fwrite(data[0] + y * linesize[0], 1, w * 3, fp);
    fclose(fp);
}

int main(int argc, char *argv[]) {
    if (argc < 3) {
        fprintf(stderr,
            "用法: %s <视频文件> <时间(秒)> [宽度=320] [高度=240]\n"
            "示例: %s movie.mp4 30 320 240\n", argv[0], argv[0]);
        return 1;
    }

    const char *inputFile = argv[1];
    double seekTimeSec = atof(argv[2]);          // 目标时间（秒）
    int thumbWidth = (argc >= 4) ? atoi(argv[3]) : 320;
    int thumbHeight = (argc >= 5) ? atoi(argv[4]) : 240;

    // 打开文件
    AVFormatContext *fmtCtx = nullptr;
    if (avformat_open_input(&fmtCtx, inputFile, nullptr, nullptr) < 0) {
        fprintf(stderr, "无法打开文件\n");
        return 1;
    }
    avformat_find_stream_info(fmtCtx, nullptr);

    // 查找视频流
    int videoIdx = av_find_best_stream(fmtCtx, AVMEDIA_TYPE_VIDEO,
                                       -1, -1, nullptr, 0);
    if (videoIdx < 0) {
        fprintf(stderr, "未找到视频流\n");
        avformat_close_input(&fmtCtx);
        return 1;
    }

    AVStream *stream = fmtCtx->streams[videoIdx];

    // 初始化解码器
    AVCodec *codec = avcodec_find_decoder(stream->codecpar->codec_id);
    AVCodecContext *codecCtx = avcodec_alloc_context3(codec);
    avcodec_parameters_to_context(codecCtx, stream->codecpar);
    avcodec_open2(codecCtx, codec, nullptr);

    // ===== 核心：Seek 到目标时间 =====
    // av_rescale_q：将秒数转换为流的时间基（time_base）单位
    // 公式：seekTimeSec * (stream->time_base.den / stream->time_base.num)
    int64_t seekTarget = av_rescale_q(
        (int64_t)(seekTimeSec * AV_TIME_BASE),  // 微秒
        AV_TIME_BASE_Q,                          // 微秒时间基 {1, 1000000}
        stream->time_base                        // 流的时间基
    );

    // AVSEEK_FLAG_BACKWARD：向前（backward）查找最近的关键帧
    // 这确保 Seek 后解码的第一个完整帧不会晚于目标时间
    int ret = av_seek_frame(fmtCtx, videoIdx, seekTarget,
                            AVSEEK_FLAG_BACKWARD);
    if (ret < 0) {
        fprintf(stderr, "Seek 失败，将从头开始\n");
        // Seek 失败不是致命错误，继续从当前位置解码
    }
    // Seek 后必须刷新解码器内部缓冲，否则可能输出 Seek 前的残留帧
    avcodec_flush_buffers(codecCtx);

    // 创建缩放+格式转换上下文（源尺寸 → 缩略图尺寸）
    struct SwsContext *swsCtx = sws_getContext(
        codecCtx->width, codecCtx->height, codecCtx->pix_fmt,
        thumbWidth, thumbHeight, AV_PIX_FMT_RGB24,
        SWS_BICUBIC,  // 双三次插值 — 缩小图片时质量更好
        nullptr, nullptr, nullptr
    );

    AVFrame *frame = av_frame_alloc();
    AVFrame *thumbFrame = av_frame_alloc();
    av_image_alloc(thumbFrame->data, thumbFrame->linesize,
                   thumbWidth, thumbHeight, AV_PIX_FMT_RGB24, 1);

    AVPacket *pkt = av_packet_alloc();
    bool saved = false;

    // 解码直到获得第一帧
    while (!saved && av_read_frame(fmtCtx, pkt) >= 0) {
        if (pkt->stream_index != videoIdx) {
            av_packet_unref(pkt);
            continue;
        }

        avcodec_send_packet(codecCtx, pkt);
        while (avcodec_receive_frame(codecCtx, frame) >= 0 && !saved) {
            // 缩放并转换格式
            sws_scale(swsCtx,
                      frame->data, frame->linesize,
                      0, codecCtx->height,
                      thumbFrame->data, thumbFrame->linesize);

            // 计算实际时间戳（用于文件名）
            double pts_sec = frame->pts * av_q2d(stream->time_base);
            char outFile[128];
            snprintf(outFile, sizeof(outFile),
                     "thumb_%.0fs_%dx%d.ppm",
                     pts_sec, thumbWidth, thumbHeight);

            savePPM(thumbFrame->data, thumbFrame->linesize,
                    thumbWidth, thumbHeight, outFile);
            printf("已保存缩略图: %s (实际时间: %.2f 秒)\n",
                   outFile, pts_sec);
            saved = true;

            av_frame_unref(frame);
        }
        av_packet_unref(pkt);
    }

    if (!saved) {
        fprintf(stderr, "错误：未能解码出任何帧\n");
    }

    // 清理
    av_freep(&thumbFrame->data[0]);
    av_frame_free(&thumbFrame);
    av_frame_free(&frame);
    av_packet_free(&pkt);
    sws_freeContext(swsCtx);
    avcodec_free_context(&codecCtx);
    avformat_close_input(&fmtCtx);
    return 0;
}
```

### 新增知识点

| 函数 | 说明 |
|------|------|
| `av_rescale_q(a, bq, cq)` | 时间基转换：`a * bq / cq`，使用 64 位中间结果避免溢出 |
| `av_seek_frame(ctx, stream_index, timestamp, flags)` | 跳转到指定时间戳，`AVSEEK_FLAG_BACKWARD` 表示向前找关键帧 |

| `av_q2d(q)` | 将 `AVRational`（分数）转为 `double`，常用于时间戳换算 |
| `SWS_BICUBIC` | 双三次插值缩放算法，比 `SWS_BILINEAR` 质量更高但更慢 |

## 实践四：解码性能基准测试

### 目标

编写一个简单的基准测试程序，测量解码视频的速度（每秒解码多少帧），对比不同线程数的性能差异。

### 完整代码

```cpp
// bench_decode.cpp — 解码性能基准测试
#include <cstdio>
#include <cstdlib>
#include <chrono>

extern "C" {
#include <libavcodec/avcodec.h>
#include <libavformat/avformat.h>
}

// 测试指定线程数的解码速度，返回每秒帧数（FPS）
static double benchDecode(const char *file, int threadCount, int maxFrames) {
    AVFormatContext *fmtCtx = nullptr;
    avformat_open_input(&fmtCtx, file, nullptr, nullptr);
    avformat_find_stream_info(fmtCtx, nullptr);

    int videoIdx = av_find_best_stream(fmtCtx, AVMEDIA_TYPE_VIDEO,
                                       -1, -1, nullptr, 0);
    if (videoIdx < 0) return 0;

    AVCodec *codec = avcodec_find_decoder(
        fmtCtx->streams[videoIdx]->codecpar->codec_id);
    AVCodecContext *ctx = avcodec_alloc_context3(codec);
    avcodec_parameters_to_context(ctx,
        fmtCtx->streams[videoIdx]->codecpar);
    ctx->thread_count = threadCount;   // 设置解码线程数
    ctx->thread_type = FF_THREAD_FRAME | FF_THREAD_SLICE;  // 同时启用帧级和片级多线程
    avcodec_open2(ctx, codec, nullptr);

    AVPacket *pkt = av_packet_alloc();
    AVFrame *frame = av_frame_alloc();
    int decoded = 0;

    // 开始计时
    auto start = std::chrono::high_resolution_clock::now();

    while (av_read_frame(fmtCtx, pkt) >= 0 && decoded < maxFrames) {
        if (pkt->stream_index != videoIdx) {
            av_packet_unref(pkt);
            continue;
        }
        avcodec_send_packet(ctx, pkt);
        while (avcodec_receive_frame(ctx, frame) >= 0 && decoded < maxFrames) {
            decoded++;
            av_frame_unref(frame);
        }
        av_packet_unref(pkt);
    }

    auto end = std::chrono::high_resolution_clock::now();
    double elapsed = std::chrono::duration<double>(end - start).count();

    double fps = (elapsed > 0) ? decoded / elapsed : 0;

    av_frame_free(&frame);
    av_packet_free(&pkt);
    avcodec_free_context(&ctx);
    avformat_close_input(&fmtCtx);

    return fps;
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <视频文件> [最大帧数=500]\n", argv[0]);
        return 1;
    }

    int maxFrames = (argc >= 3) ? atoi(argv[2]) : 500;
    printf("解码性能基准测试: %s (前 %d 帧)\n\n", argv[1], maxFrames);
    printf("线程数 | 解码帧数 |   FPS   |  相对加速比\n");
    printf("-------|----------|---------|----------\n");

    double baseFps = 0;
    // 测试 1、2、4、8 线程
    int threadCounts[] = {1, 2, 4, 8};
    for (int tc : threadCounts) {
        double fps = benchDecode(argv[1], tc, maxFrames);
        if (tc == 1) baseFps = fps;
        double speedup = (baseFps > 0) ? fps / baseFps : 0;
        printf("   %d   |   %d    | %7.1f | %.2fx\n",
               tc, maxFrames, fps, speedup);
    }

    printf("\n注意：实际加速比受 CPU 核心数、编码格式、分辨率等因素影响\n");
    printf("H.264 的帧级多线程通常在 4 线程后收益递减\n");
    return 0;
}
```

## 常见错误及解决方案

### 错误 1：解码后画面全绿或有色块

**现象**：保存的 PPM 图片全是绿色，或者有大块绿色区域。

**原因**：未正确处理像素格式转换。YUV 格式中，Y=0、U=128、V=128 对应绿色。如果 `sws_scale` 的参数错误（如源格式写错），未被正确转换的区域就显示为绿色。

```cpp
// ❌ 错误：源格式硬编码为 YUV420P，但实际可能是其他格式
struct SwsContext *swsCtx = sws_getContext(
    width, height, AV_PIX_FMT_YUV420P,  // 写死了！
    width, height, AV_PIX_FMT_RGB24,
    SWS_BILINEAR, nullptr, nullptr, nullptr
);

// ✅ 正确：使用解码器上下文报告的实际像素格式
struct SwsContext *swsCtx = sws_getContext(
    width, height, codecCtx->pix_fmt,    // 从解码器获取实际格式
    width, height, AV_PIX_FMT_RGB24,
    SWS_BILINEAR, nullptr, nullptr, nullptr
);
```

### 错误 2：视频最后几帧丢失

**现象**：提取的帧数比预期少几帧，特别是视频末尾的帧缺失。

**原因**：未 Flush 解码器。对于使用 B 帧的编码格式（H.264/H.265），解码器内部会缓存几帧用于参考帧重排序。文件读完后必须发送 NULL 包来通知解码器"不会再有输入了"，触发它输出剩余的缓存帧。

```cpp
// ❌ 错误：读完文件就直接清理
while (av_read_frame(fmtCtx, pkt) >= 0) {
    // ... 解码 ...
}
// 直接释放资源 — 丢失了缓存的帧！

// ✅ 正确：读完文件后 Flush 解码器
while (av_read_frame(fmtCtx, pkt) >= 0) {
    // ... 解码 ...
}
// Flush：发送 NULL 包，取出所有缓存帧
avcodec_send_packet(codecCtx, nullptr);
while (avcodec_receive_frame(codecCtx, frame) >= 0) {
    // 处理残留帧...
    av_frame_unref(frame);
}
```

### 错误 3：Seek 后第一帧画面花屏

**现象**：跳转到视频某个时间点后，解码出的第一帧画面有花屏或马赛克。

**原因**：Seek 后未调用 `avcodec_flush_buffers`。解码器内部仍残留 Seek 前的参考帧数据，新解码的帧试图引用这些无效数据，导致画面错误。

```cpp
// ❌ 错误：Seek 后直接继续解码
av_seek_frame(fmtCtx, videoIdx, target, AVSEEK_FLAG_BACKWARD);
// 直接读包解码 — 可能花屏！

// ✅ 正确：Seek 后清空解码器缓冲
av_seek_frame(fmtCtx, videoIdx, target, AVSEEK_FLAG_BACKWARD);
avcodec_flush_buffers(codecCtx);  // 必须！清空内部参考帧缓冲
```

### 错误 4：内存持续增长（内存泄漏）

**现象**：程序运行时内存使用量持续增长，最终耗尽内存。

**原因**：未调用 `av_frame_unref` 或 `av_packet_unref`。每次 `avcodec_receive_frame` 返回的帧和 `av_read_frame` 返回的包都引用了内部分配的缓冲区，必须显式释放。

```cpp
// ❌ 错误：忘记释放帧和包的数据引用
while (av_read_frame(fmtCtx, pkt) >= 0) {
    avcodec_send_packet(ctx, pkt);
    // 忘记 av_packet_unref(pkt) — 每次循环都泄漏一个包的缓冲区！

    while (avcodec_receive_frame(ctx, frame) >= 0) {
        // 处理帧...
        // 忘记 av_frame_unref(frame) — 每次循环都泄漏一帧的缓冲区！
    }
}

// ✅ 正确：每次使用完毕后立即释放引用
while (av_read_frame(fmtCtx, pkt) >= 0) {
    avcodec_send_packet(ctx, pkt);
    while (avcodec_receive_frame(ctx, frame) >= 0) {
        // 处理帧...
        av_frame_unref(frame);   // ← 必须！
    }
    av_packet_unref(pkt);        // ← 必须！
}
```

3. **硬件加速**（第三节）：硬件加速原理→`get_format` 协商→KrKr2 多后端支持→丢帧控制
4. **动手实践**（本节）：帧提取器→音频统计器→缩略图生成器→性能测试→常见错误排查

### 解码链路总览

```
┌──────────────────────────────────────────────────────────┐
│                     完整解码流程                          │
├──────────────────────────────────────────────────────────┤
│                                                          │
│  avformat_open_input()          打开文件                 │
│       ↓                                                  │
│  avformat_find_stream_info()    探测流信息               │
│       ↓                                                  │
│  av_find_best_stream()          选择目标流               │
│       ↓                                                  │
│  avcodec_find_decoder()         查找解码器               │
│       ↓                                                  │
│  avcodec_alloc_context3()       分配上下文               │
│  avcodec_parameters_to_context()                         │
│  avcodec_open2()                初始化解码器             │
│       ↓                                                  │
│  ┌─ av_read_frame()  ──── 解码循环 ───────────────┐     │
│  │  avcodec_send_packet()        送入压缩包       │     │
│  │  avcodec_receive_frame()      取出原始帧       │     │
│  │  [格式转换 sws_scale/swr_convert]              │     │
│  │  [处理/保存/渲染]                              │     │
│  │  av_frame_unref() / av_packet_unref()          │     │
│  └────────────────────────────────────────────────┘     │
│       ↓                                                  │
│  Flush: send_packet(NULL) + 循环 receive_frame          │
│       ↓                                                  │
│  释放: frame → packet → sws → codec → format            │
│                                                          │
└──────────────────────────────────────────────────────────┘
```

### 资源释放顺序规则

释放顺序遵循**"后分配先释放"**原则（类似栈的 LIFO）：

```cpp
// 1. 先释放 av_image_alloc 的缓冲区（如果有）
av_freep(&rgbFrame->data[0]);

// 2. 释放 AVFrame 结构
av_frame_free(&frame);
av_frame_free(&rgbFrame);

// 3. 释放 AVPacket 结构
av_packet_free(&pkt);

// 4. 释放格式转换上下文（SwsContext / SwrContext）
sws_freeContext(swsCtx);

// 5. 释放解码器上下文（会自动关闭解码器）
avcodec_free_context(&codecCtx);

// 6. 最后关闭输入文件（释放 AVFormatContext）
avformat_close_input(&fmtCtx);
```

## 练习题与答案

### 题目 1：新旧 API 的关键区别

请回答以下问题：
1. 新 API `avcodec_send_packet` 返回 `AVERROR(EAGAIN)` 时应该怎么处理？
2. 为什么新 API 中一次 `send_packet` 后需要循环调用 `receive_frame`？
3. KrKr2 的 movie 模块为什么仍使用旧 API？

<details>
<summary>查看答案</summary>

1. 返回 `AVERROR(EAGAIN)` 表示解码器内部缓冲区已满，应先循环调用 `avcodec_receive_frame` 取走所有已解码帧，然后再重新调用 `avcodec_send_packet` 发送该包。

2. 因为某些编解码器可能从一个 packet 中解码出多个 frame。例如：
   - 音频编解码器：一个 AAC packet 可能包含多个音频帧
   - 视频编解码器：B 帧（Bidirectional Frame，双向预测帧）重排可能导致之前缓存的帧在此次 send 后变为可用
   - 解码器缓冲区：某些解码器会在内部积累几帧后一次性输出

3. KrKr2 的 movie 模块移植自 Kodi/XBMC 较早版本（约 2015-2016 年），当时新 API 尚未普及。由于代码已经稳定运行，迁移到新 API 的收益有限（功能等价），但需要大量测试验证。值得注意的是，同项目的 sound 模块（`FFWaveDecoder`）因为是后来新写的，已经使用了新 API。

</details>

### 题目 2：修复音频杂音问题

以下代码从 FFmpeg 解码音频后直接送给 OpenAL（一种跨平台音频 API，将在 P07 详细学习）播放，但听到的是杂音。找出问题并修复：

```cpp
// 有 Bug 的代码
AVFrame *frame = av_frame_alloc();
avcodec_send_packet(ctx, pkt);
avcodec_receive_frame(ctx, frame);

// 直接将 frame->data[0] 送给 OpenAL
alBufferData(buffer, AL_FORMAT_STEREO16,
             frame->data[0],
             frame->nb_samples * 4,  // 2声道 × 2字节
             ctx->sample_rate);
```

<details>
<summary>查看答案</summary>

问题在于大多数现代音频编解码器（AAC、Vorbis、MP3 等）输出**平面格式**（如 `AV_SAMPLE_FMT_FLTP`——32-bit 浮点平面格式），此时：
- `frame->data[0]` 只包含**左声道**的浮点采样
- `frame->data[1]` 包含**右声道**的浮点采样
- 数据类型是 `float`（4 字节），而 OpenAL 期望的 `AL_FORMAT_STEREO16` 是 `int16`（2 字节）

直接用 `frame->data[0]` 导致：数据格式错误（float→int16 重新解释）+ 缺少右声道 = 杂音。

**修复方案：使用 `SwrContext`（libswresample 的重采样上下文）做格式转换：**

```cpp
#include <libswresample/swresample.h>

// 创建重采样/格式转换器（初始化一次，后续复用）
SwrContext *swrCtx = swr_alloc_set_opts(
    nullptr,
    AV_CH_LAYOUT_STEREO,   // 输出：立体声布局
    AV_SAMPLE_FMT_S16,     // 输出：16-bit 有符号整数交错格式
    ctx->sample_rate,       // 输出采样率（不变）
    ctx->channel_layout,    // 输入：解码器的声道布局
    ctx->sample_fmt,        // 输入：解码器的采样格式（如 FLTP）
    ctx->sample_rate,       // 输入采样率（不变）
    0, nullptr
);
swr_init(swrCtx);

// 解码
AVFrame *frame = av_frame_alloc();
avcodec_send_packet(ctx, pkt);
avcodec_receive_frame(ctx, frame);

// 格式转换：FLTP(平面float) → S16(交错int16)
uint8_t *outBuf = nullptr;
int outSamples = frame->nb_samples;
av_samples_alloc(&outBuf, nullptr, 2, outSamples, AV_SAMPLE_FMT_S16, 0);
swr_convert(swrCtx,
            &outBuf, outSamples,                        // 输出
            (const uint8_t **)frame->data, frame->nb_samples);  // 输入

// 现在 outBuf 是 S16 交错格式，可以正确送给 OpenAL
alBufferData(buffer, AL_FORMAT_STEREO16,
             outBuf,
             outSamples * 2 * 2,  // 采样数 × 声道数 × 每采样字节数
             ctx->sample_rate);

av_freep(&outBuf);
av_frame_unref(frame);
// 别忘了最后释放：swr_free(&swrCtx);
```

</details>

### 题目 3：分析 KrKr2 的丢帧策略

阅读 `CDVDVideoCodecFFmpeg::CDropControl` 的源码（`cpp/core/movie/ffmpeg/VideoCodecFFmpeg.cpp` 第 61-102 行），回答：
1. `CDropControl` 有几个状态？各状态的含义是什么？
2. 从 INIT 切换到 VALID 需要满足什么条件？
3. 从 VALID 回退到 INIT 需要满足什么条件？这个设计解决了什么问题？

<details>
<summary>查看答案</summary>

1. **两个状态**：
   - `INIT`：初始化/学习阶段——正在收集帧间 PTS（Presentation Time Stamp，显示时间戳）差值数据，此阶段不做丢帧判断
   - `VALID`：有效阶段——已经学习到稳定的帧间隔（即视频帧率），可以基于此判断是否需要丢帧

2. **INIT → VALID 的条件**：
   - 收集了超过 **10 帧**有效的 PTS 差值（`m_count > 10`）
   - 计算出的平均帧间隔 `m_diffPTS` 大于 0
   - 计算方式：累加所有相邻帧的 PTS 差值，除以帧数得到平均值

3. **VALID → INIT 的条件**：
   - 在非丢帧状态下（`!drop`），连续检测到 **超过 5 次**实际 PTS 差值与平均值偏离超过 **20%**
   - 即：`|当前PTS - 上一帧PTS - 平均间隔| > 平均间隔 × 0.2` 连续 5 次以上
   - **解决的问题**：视频流的帧率发生了实质性变化（如可变帧率视频从 30fps 区段切换到 24fps 区段），之前学习到的帧间隔已经不准确，需要重新学习。5 次连续阈值避免了因偶尔的 PTS 抖动而误触发重置。

</details>
