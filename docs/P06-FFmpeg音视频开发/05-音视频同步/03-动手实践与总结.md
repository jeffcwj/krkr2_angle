### 实践 1：实现音频主时钟同步

以下是一个简化的音视频同步框架，展示音频主时钟策略的核心逻辑：

```cpp
// av_sync_demo.cpp — 音视频同步原理演示
#include <cstdio>
#include <cmath>
#include <chrono>
#include <thread>

// 模拟时钟（简化版 CDVDClock）
class SimpleClock {
public:
    void start() {
        m_startTime = now();
        m_paused = false;
    }

    void pause(bool p) {
        if (p && !m_paused) {
            m_pauseTime = now();
            m_paused = true;
        } else if (!p && m_paused) {
            m_startTime += (now() - m_pauseTime);
            m_paused = false;
        }
    }

    // 获取当前播放时间（秒）
    double getClock() const {
        if (m_paused)
            return (m_pauseTime - m_startTime) / 1e6;
        return (now() - m_startTime) / 1e6;
    }

    // 跳变修正
    void adjust(double seconds) {
        m_startTime -= (int64_t)(seconds * 1e6);
    }

private:
    static int64_t now() {
        auto t = std::chrono::steady_clock::now().time_since_epoch();
        return std::chrono::duration_cast<std::chrono::microseconds>(t).count();
    }

    int64_t m_startTime = 0;
    int64_t m_pauseTime = 0;
    bool m_paused = false;
};

// 模拟视频帧
struct VideoFrame {
    double pts;        // 显示时间（秒）
    int frameNumber;
    bool isKeyFrame;
};

// 音频主时钟同步逻辑
class AudioMasterSync {
public:
    AudioMasterSync() : m_syncThreshold(0.04) {}  // 40ms 容差

    enum Action { DISPLAY, SKIP, WAIT };

    // 决定视频帧的处理方式
    Action decide(const VideoFrame &frame, double audioClock) {
        double diff = frame.pts - audioClock;

        if (diff < -m_syncThreshold) {
            // 视频落后超过阈值 → 丢帧
            m_droppedFrames++;
            return SKIP;
        } else if (diff > m_syncThreshold) {
            // 视频超前超过阈值 → 等待
            m_waitMs = (int)(diff * 1000);
            return WAIT;
        } else {
            // 在同步范围内 → 正常显示
            return DISPLAY;
        }
    }

    int getWaitMs() const { return m_waitMs; }
    int getDroppedFrames() const { return m_droppedFrames; }

private:
    double m_syncThreshold;
    int m_waitMs = 0;
    int m_droppedFrames = 0;
};

int main() {
    SimpleClock clock;
    AudioMasterSync sync;

    // 模拟 30fps 视频帧序列
    const int totalFrames = 150;  // 5 秒
    const double fps = 30.0;

    clock.start();

    for (int i = 0; i < totalFrames; i++) {
        VideoFrame frame;
        frame.pts = i / fps;
        frame.frameNumber = i;
        frame.isKeyFrame = (i % 30 == 0);

        // 模拟音频时钟（假设音频稳定播放，加入小随机抖动）
        double audioClock = clock.getClock();

        auto action = sync.decide(frame, audioClock);

        switch (action) {
            case AudioMasterSync::DISPLAY:
                printf("帧 #%03d [%.3fs]: 显示 (diff=%.1fms)\n",
                       i, frame.pts, (frame.pts - audioClock) * 1000);
                // 模拟显示耗时
                std::this_thread::sleep_for(std::chrono::milliseconds(33));
                break;

            case AudioMasterSync::SKIP:
                printf("帧 #%03d [%.3fs]: 丢弃 (落后 %.1fms)\n",
                       i, frame.pts, (audioClock - frame.pts) * 1000);
                break;

            case AudioMasterSync::WAIT: {
                int waitMs = sync.getWaitMs();
                printf("帧 #%03d [%.3fs]: 等待 %dms\n",
                       i, frame.pts, waitMs);
                std::this_thread::sleep_for(std::chrono::milliseconds(waitMs));
                printf("帧 #%03d [%.3fs]: 显示（等待后）\n", i, frame.pts);
                break;
            }
        }
    }

    printf("\n=== 同步统计 ===\n");
    printf("总帧数: %d\n", totalFrames);
    printf("丢弃帧: %d\n", sync.getDroppedFrames());
    printf("实际时长: %.3f 秒（期望 %.3f 秒）\n",
           clock.getClock(), totalFrames / fps);

    return 0;
}
```

### 实践 2：PTS 时间基转换工具

```cpp
// timebase_converter.cpp — 时间基转换实用工具
#include <cstdio>
#include <cstdlib>

extern "C" {
#include <libavformat/avformat.h>
#include <libavutil/mathematics.h>
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "用法: %s <媒体文件>\n", argv[0]);
        return 1;
    }

    AVFormatContext *fmtCtx = nullptr;
    avformat_open_input(&fmtCtx, argv[1], nullptr, nullptr);
    avformat_find_stream_info(fmtCtx, nullptr);

    printf("=== 流时间基信息 ===\n");
    for (unsigned i = 0; i < fmtCtx->nb_streams; i++) {
        AVStream *st = fmtCtx->streams[i];
        const char *type = av_get_media_type_string(st->codecpar->codec_type);
        printf("\n流 #%u (%s):\n", i, type ? type : "未知");
        printf("  time_base: %d/%d (1 tick = %.6f 秒)\n",
               st->time_base.num, st->time_base.den,
               av_q2d(st->time_base));
        printf("  duration:  %lld ticks = %.3f 秒\n",
               (long long)st->duration,
               st->duration * av_q2d(st->time_base));
        printf("  start_time: %lld ticks = %.3f 秒\n",
               (long long)st->start_time,
               st->start_time != AV_NOPTS_VALUE
                   ? st->start_time * av_q2d(st->time_base) : 0.0);
    }

    // 读取前 20 个包，展示 PTS/DTS 转换
    printf("\n=== 前 20 个包的时间戳 ===\n");
    printf("%-6s %-8s %-12s %-12s %-12s %-12s\n",
           "流", "类型", "PTS(tick)", "PTS(秒)", "DTS(tick)", "DTS(秒)");

    AVPacket *pkt = av_packet_alloc();
    int count = 0;
    while (av_read_frame(fmtCtx, pkt) >= 0 && count < 20) {
        AVStream *st = fmtCtx->streams[pkt->stream_index];
        const char *type = av_get_media_type_string(st->codecpar->codec_type);

        double ptsSec = (pkt->pts != AV_NOPTS_VALUE)
                            ? pkt->pts * av_q2d(st->time_base) : -1.0;
        double dtsSec = (pkt->dts != AV_NOPTS_VALUE)
                            ? pkt->dts * av_q2d(st->time_base) : -1.0;

        printf("%-6u %-8s %-12lld %-12.3f %-12lld %-12.3f\n",
               pkt->stream_index, type ? type : "?",
               (long long)pkt->pts, ptsSec,
               (long long)pkt->dts, dtsSec);

        av_packet_unref(pkt);
        count++;
    }

    av_packet_free(&pkt);
    avformat_close_input(&fmtCtx);
    return 0;
}
```

### 常见错误与解决方案

**错误 1：seek 后音画不同步**

```
原因：seek 后没有正确 flush 解码器和重置同步状态
解决：
1. avcodec_flush_buffers() 清空解码器内部缓冲
2. 清空音频输出缓冲区
3. 重置同步状态为 SYNC_STARTING
4. 用 seek 目标位置重新设置 audioClock
（KrKr2 通过 GENERAL_FLUSH 消息完成以上所有步骤）
```

**错误 2：播放一段时间后逐渐不同步**

```
原因：时钟漂移（drift）—— 声卡实际采样率与标称值有微小偏差
      例如标称 48000Hz 的声卡实际可能是 47998Hz
解决：
1. 持续监测同步误差
2. 使用微量的速度调整（speedAdjust）而非大幅跳变
3. KrKr2 通过 CDVDClock::SetSpeedAdjust() 实现微调
```

**错误 3：av_rescale_q 溢出**

```
原因：大的 PTS 值在不同时间基转换时可能溢出 int64
解决：使用 av_rescale_q_rnd() 指定舍入模式
      或用 double 中间值：pts * av_q2d(src_tb) / av_q2d(dst_tb)
```

## 5.7 本节小结

- **音视频同步**的核心问题是：音频和视频由不同硬件时钟驱动，长时间运行后会产生漂移
- **PTS/DTS/time_base** 三者共同定义了帧的时间位置，跨流/跨模块传递时必须正确转换时间基
- 三种主时钟策略：**音频主时钟**（最常用，保证音频连续性）、**视频主时钟**（重采样调整音频）、**外部主时钟**（直播场景）
- KrKr2 使用 **SYNC_DISCON**（音频主时钟 + 不连续跳变修正）为默认策略
- **CDVDClock** 维护三层时间模型（系统时间→绝对时间→播放时间），支持暂停、变速、不连续跳变
- **ErrorAdjust** 是同步修正的核心：VSync 模式下按帧步进（±1帧），非 VSync 模式下直接跳变
- **CVideoPlayerAudio** 的三状态同步状态机（STARTING→WAITSYNC→INSYNC）确保启动时的平滑同步
- 音频时钟通过 PTS 精确更新 + duration 预测递增的组合方式维护

## 5.8 练习题与答案

### 题目 1：时间基转换

一个 MPEG-TS 视频流的 `time_base = {1, 90000}`，某帧的 PTS 为 `8100000`。请计算：
1. 该帧应该在第几秒显示？
2. 如果要转换到 `time_base = {1, 1000}`（毫秒），新的 PTS 值是多少？
3. 如果要转换到 KrKr2 的 `DVD_TIME_BASE`（1000000），新的 PTS 值是多少？

<details>
<summary>查看答案</summary>

1. **显示时间**：`8100000 * (1/90000) = 90.0 秒`

2. **转为毫秒时间基**：
   ```
   new_pts = 8100000 * (1/90000) / (1/1000) = 8100000 / 90 = 90000
   ```
   或使用 `av_rescale_q(8100000, {1,90000}, {1,1000}) = 90000`
   
   PTS = 90000（即 90000 毫秒 = 90 秒）

3. **转为 DVD_TIME_BASE**：
   ```
   new_pts = 8100000 * (1/90000) / (1/1000000) = 8100000 * 1000000 / 90000 = 90000000
   ```
   或 `av_rescale_q(8100000, {1,90000}, {1,1000000}) = 90000000`
   
   PTS = 90000000（即 90000000 微秒 = 90 秒）

</details>

### 题目 2：同步策略分析

阅读以下场景，判断应该使用哪种同步策略，并说明理由：

- 场景 A：用户在电脑上观看一部 2 小时的电影
- 场景 B：直播推流，需要与真实时间对齐
- 场景 C：视觉小说游戏中播放过场动画（KrKr2 的典型场景）

<details>
<summary>查看答案</summary>

- **场景 A：音频主时钟**
  - 理由：2 小时长时间播放，音频连续性最重要（中断音频比丢视频帧更令人不适）。声卡硬件时钟稳定，适合作为长期基准。视频偶尔丢帧用户几乎感知不到。
  
- **场景 B：外部主时钟**
  - 理由：直播需要与真实时间（墙钟）对齐，不能让任何一条流"自主"决定时间。音频和视频都需要根据墙钟独立调整速率。

- **场景 C：音频主时钟（SYNC_DISCON）**
  - 理由：视觉小说的过场动画通常很短（几十秒到几分钟），用户对画面和音频的配合非常敏感（对话配音对口型）。音频主时钟保证语音连续流畅。KrKr2 默认使用 SYNC_DISCON 正是因为这个原因。短时间播放不太会遇到严重的时钟漂移问题。

</details>

### 题目 3：CDVDClock::SystemToPlaying 公式推导

`CDVDClock::SystemToPlaying` 的核心公式是：

```
playing = DVD_TIME_BASE * (current - m_startClock + m_systemAdjust) / m_systemUsed + m_iDisc
```

请回答：
1. 当播放速度从 1.0x 变为 2.0x 时，`m_systemUsed` 如何变化？为什么这能实现 2 倍速？
2. `m_iDisc` 的作用是什么？什么情况下它会被修改？
3. 为什么暂停时使用 `m_pauseClock` 而不是简单地让 `m_systemUsed = 0`？

<details>
<summary>查看答案</summary>

1. **m_systemUsed 变化**：
   - 1.0x 速度时：`m_systemUsed = m_systemFrequency * 1000 / 1000 = m_systemFrequency`
   - 2.0x 速度时：`m_systemUsed = m_systemFrequency * 1000 / 2000 = m_systemFrequency / 2`
   
   当 `m_systemUsed` 减半时，同样的系统时间差 `(current - m_startClock)` 除以 `m_systemUsed` 后得到的播放时间变为 2 倍。即系统过了 1 秒，播放时间走了 2 秒。

2. **m_iDisc（discontinuity）的作用**：
   - 它是时钟跳变的基准偏移量。当调用 `Discontinuity(clock, absolute)` 时，`m_iDisc = clock`，同时 `m_startClock` 被重置为当前绝对时间。
   - 这使得 `SystemToPlaying` 在跳变后立即返回 `m_iDisc`（因为 `current - m_startClock ≈ 0`），从而实现时钟的瞬间跳转。
   - 修改场景：seek、同步误差修正（`ErrorAdjust`）、流切换、首次播放。

3. **为什么不用 m_systemUsed = 0**：
   - 如果 `m_systemUsed = 0`，公式中会出现除以零（division by zero），导致未定义行为。
   - 使用 `m_pauseClock` 是更优雅的方案：冻结 `current` 值为暂停时刻的系统时间，公式的分子 `(m_pauseClock - m_startClock)` 变为常量，播放时间自然停止增长。
   - 恢复播放时，通过 `m_startClock += current - m_pauseClock` 补偿暂停期间的系统时间流逝，使播放时间从暂停点无缝继续。

</details>
