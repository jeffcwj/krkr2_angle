## 练习 1：添加控制台进度条显示

### 需求分析

当前播放器只有视频画面和音频输出，没有任何进度信息。我们在控制台输出一个实时更新的进度条，显示当前时间和总时长。

### 完整实现

```c
// ============================================================
// 进度条显示 —— 在 video_refresh() 末尾调用
// ============================================================

// 在 video_refresh 函数的渲染完成后（SDL_RenderPresent 之后）添加：
static void print_progress(PlayerState *ps) {
    // static 变量在函数调用间保持值，用于控制打印频率
    static double last_print_time = 0;
    double now = ps->audio_clock;

    // 每秒只打印一次，避免控制台刷屏影响性能
    if (now - last_print_time < 1.0) return;
    last_print_time = now;

    // 获取总时长（秒）
    // fmt_ctx->duration 的单位是 AV_TIME_BASE（微秒）
    // 除以 AV_TIME_BASE 转换为秒
    double duration = ps->fmt_ctx->duration / (double)AV_TIME_BASE;
    if (duration <= 0) return;  // 直播流没有总时长

    // 计算分:秒
    int cur_min = (int)now / 60;
    int cur_sec = (int)now % 60;
    int dur_min = (int)duration / 60;
    int dur_sec = (int)duration % 60;

    // 计算进度百分比
    double percent = (now / duration) * 100.0;
    if (percent > 100.0) percent = 100.0;

    // 绘制进度条（50 字符宽）
    int bar_width = 50;
    int filled = (int)(percent / 100.0 * bar_width);

    // \r 回车符将光标移到行首，实现原地更新效果
    fprintf(stderr, "\r[");
    for (int i = 0; i < bar_width; i++) {
        if (i < filled)      fprintf(stderr, "█");  // 已播放部分
        else if (i == filled) fprintf(stderr, "▶");  // 当前位置
        else                  fprintf(stderr, "░");  // 未播放部分
    }
    fprintf(stderr, "] %02d:%02d / %02d:%02d (%.1f%%)  ",
            cur_min, cur_sec, dur_min, dur_sec, percent);
    fflush(stderr);  // 立即刷新输出（stderr 默认行缓冲）
}

// 在 video_refresh 中调用：
static void video_refresh(PlayerState *ps) {
    // ... 原有的解码和渲染代码 ...

    // 渲染完成后显示进度
    print_progress(ps);
}
```

### 运行效果

```
[████████████████████▶░░░░░░░░░░░░░░░░░░░░░░░░░░░░░] 01:23 / 03:45 (36.9%)
```

### 跨平台注意事项

| 平台 | 注意 |
|------|------|
| Windows (CMD) | `█` 和 `░` 需要 UTF-8 代码页（`chcp 65001`），否则显示乱码 |
| Windows (PowerShell) | 默认支持 Unicode，无需额外设置 |
| Linux/macOS 终端 | 默认支持 UTF-8，直接可用 |
| Android | 无控制台，应使用 `__android_log_print` 输出到 Logcat |

如果不想依赖 Unicode，可以用 ASCII 替代：`#` 代替 `█`，`>` 代替 `▶`，`-` 代替 `░`。

## 练习 2：实现帧队列（FrameQueue）

### 当前设计的问题

当前 `video_refresh` 中，视频超前时直接丢弃已解码的帧。这意味着下次定时器触发时，需要重新从包队列取包、重新解码。解码是 CPU 密集操作，这种"解码了又丢"的做法浪费计算资源。

```
当前设计（浪费 CPU）：
  超前帧 → 解码 → 丢弃 → 下次再解码同一帧（实际是取下一帧）
  
改进设计（帧队列缓存）：
  超前帧 → 解码 → 存入帧队列 → 下次直接从队列取 → 无需再解码
```

### 帧队列设计

```c
// ============================================================
// 帧队列 —— 缓存解码后的视频帧
// 设计参考：ffplay 的 FrameQueue 和 KrKr2 的 CRenderManager
// ============================================================

#define FRAME_QUEUE_SIZE 4  // 缓存 4 帧（平衡内存和流畅度）

// 帧队列条目 —— 存储一帧解码后的视频数据及其显示时间
typedef struct FrameEntry {
    AVFrame *frame;   // 解码后的视频帧（YUV 数据）
    double   pts;     // 显示时间戳（秒）
    int      valid;   // 是否包含有效数据
} FrameEntry;

// 帧队列 —— 环形缓冲区（ring buffer）
// 环形缓冲区用固定大小的数组模拟无限队列，
// 通过 read_idx 和 write_idx 的循环递增实现
typedef struct FrameQueue {
    FrameEntry  entries[FRAME_QUEUE_SIZE];  // 帧数组
    int         read_idx;    // 消费者（渲染端）读取位置
    int         write_idx;   // 生产者（解码端）写入位置
    int         size;        // 当前队列中的帧数
    SDL_mutex  *mutex;       // 互斥锁（保护共享状态）
    SDL_cond   *cond;        // 条件变量（线程间通知）
} FrameQueue;

// ── 初始化 ──
static void frame_queue_init(FrameQueue *fq) {
    memset(fq, 0, sizeof(FrameQueue));
    fq->mutex = SDL_CreateMutex();
    fq->cond  = SDL_CreateCond();
    // 预分配所有帧
    for (int i = 0; i < FRAME_QUEUE_SIZE; i++) {
        fq->entries[i].frame = av_frame_alloc();
    }
}

// ── 写入帧（解码端调用）──
// 将解码后的帧存入队列，队列满时阻塞等待
static int frame_queue_push(FrameQueue *fq, AVFrame *frame, double pts) {
    SDL_LockMutex(fq->mutex);

    // 队列满时等待消费者消费
    while (fq->size >= FRAME_QUEUE_SIZE) {
        SDL_CondWait(fq->cond, fq->mutex);
    }

    // 将帧数据复制到队列的当前写入位置
    FrameEntry *e = &fq->entries[fq->write_idx];
    av_frame_unref(e->frame);        // 清除旧数据
    av_frame_ref(e->frame, frame);   // 增加引用（零拷贝）
    e->pts = pts;
    e->valid = 1;

    // 环形递增：到达数组末尾后回到 0
    fq->write_idx = (fq->write_idx + 1) % FRAME_QUEUE_SIZE;
    fq->size++;

    SDL_CondSignal(fq->cond);  // 通知等待的消费者
    SDL_UnlockMutex(fq->mutex);
    return 0;
}

// ── 查看队头帧（渲染端调用，不消费）──
// 返回队头帧的指针，用于检查 PTS 决定是否渲染
static FrameEntry *frame_queue_peek(FrameQueue *fq) {
    SDL_LockMutex(fq->mutex);
    FrameEntry *e = NULL;
    if (fq->size > 0) {
        e = &fq->entries[fq->read_idx];
    }
    SDL_UnlockMutex(fq->mutex);
    return e;
}

// ── 消费队头帧（确认渲染后调用）──
// 将读取位置前进一步，释放一个队列槽位
static void frame_queue_next(FrameQueue *fq) {
    SDL_LockMutex(fq->mutex);
    fq->read_idx = (fq->read_idx + 1) % FRAME_QUEUE_SIZE;
    fq->size--;
    SDL_CondSignal(fq->cond);  // 通知等待的生产者（有空位了）
    SDL_UnlockMutex(fq->mutex);
}

// ── 销毁 ──
static void frame_queue_destroy(FrameQueue *fq) {
    for (int i = 0; i < FRAME_QUEUE_SIZE; i++) {
        av_frame_free(&fq->entries[i].frame);
    }
    SDL_DestroyMutex(fq->mutex);
    SDL_DestroyCond(fq->cond);
}
```

### 使用帧队列的 video_refresh

```c
// 改造后的 video_refresh —— 从帧队列取帧而非直接解码
static void video_refresh_v2(PlayerState *ps) {
    FrameEntry *e = frame_queue_peek(&ps->frame_queue);
    if (!e) return;  // 队列为空

    // A/V 同步判断
    double diff = e->pts - ps->audio_clock;

    if (diff > 0.05) {
        // 视频超前 → 不渲染，不消费，等下次检查
        // 关键区别：帧留在队列中，不会被丢弃
        return;
    }

    if (diff < -0.05) {
        // 视频落后 → 丢帧追赶（消费但不渲染）
        frame_queue_next(&ps->frame_queue);
        return;
    }

    // 正常渲染
    SDL_UpdateYUVTexture(
        ps->texture, NULL,
        e->frame->data[0], e->frame->linesize[0],
        e->frame->data[1], e->frame->linesize[1],
        e->frame->data[2], e->frame->linesize[2]
    );
    SDL_RenderClear(ps->renderer);
    SDL_RenderCopy(ps->renderer, ps->texture, NULL, NULL);
    SDL_RenderPresent(ps->renderer);

    // 渲染完成，消费这一帧
    frame_queue_next(&ps->frame_queue);

    print_progress(ps);
}
```

### 帧队列与 KrKr2 的对比

| 特性 | 我们的帧队列 | KrKr2 CRenderManager |
|------|-------------|---------------------|
| 缓冲区大小 | 4 帧（固定） | 6 帧（可配置） |
| 数据结构 | 环形数组（ring buffer） | 4 个 deque（free/queued/presentsource/discard） |
| 帧引用 | `av_frame_ref`（引用计数） | 自定义 `CRenderPic` 包装 |
| 渲染策略 | 简单渲染 | 支持 Progressive/Interlaced/Blend 三种渲染模式 |
| VSync 对齐 | 无 | `PrepareNextRender()` 精确对齐 VSync 时间点 |

## 练习 3：添加 OSD 文字叠加

### 需求

在视频画面上叠加显示当前播放时间，类似视频播放器的时间戳 OSD。

SDL2 本身不支持文字渲染，需要使用 SDL_ttf 扩展库。

### 依赖安装

```bash
# Linux (Ubuntu/Debian)
sudo apt install libsdl2-ttf-dev

# macOS
brew install sdl2_ttf

# Windows (vcpkg)
vcpkg install sdl2-ttf:x64-windows

# CMakeLists.txt 添加：
find_package(SDL2_ttf REQUIRED)
target_link_libraries(player PRIVATE SDL2_ttf::SDL2_ttf)
```

### 实现代码

```c
#include <SDL_ttf.h>

// 在 PlayerState 中添加：
typedef struct PlayerState {
    // ... 其他字段 ...
    TTF_Font *font;  // OSD 字体
} PlayerState;

// ── 初始化 SDL_ttf ──
static int osd_init(PlayerState *ps) {
    // TTF_Init 初始化 SDL_ttf 库
    // 必须在 SDL_Init 之后、使用任何 TTF 函数之前调用
    if (TTF_Init() < 0) {
        fprintf(stderr, "TTF_Init 失败: %s\n", TTF_GetError());
        return -1;
    }

    // TTF_OpenFont 加载字体文件
    // 参数：
    //   path —— 字体文件路径（.ttf 或 .otf）
    //   ptsize —— 字号（磅值）
    // 返回值：TTF_Font* 或 NULL
    ps->font = TTF_OpenFont("NotoSansSC-Regular.ttf", 24);
    if (!ps->font) {
        // 尝试系统字体
        #ifdef _WIN32
        ps->font = TTF_OpenFont("C:\\Windows\\Fonts\\msyh.ttc", 24);
        #elif __APPLE__
        ps->font = TTF_OpenFont("/System/Library/Fonts/PingFang.ttc", 24);
        #else
        ps->font = TTF_OpenFont("/usr/share/fonts/truetype/noto/"
                                "NotoSansCJK-Regular.ttc", 24);
        #endif
    }

    if (!ps->font) {
        fprintf(stderr, "警告：无法加载字体，OSD 将不显示\n");
    }

    return 0;
}

// ── 渲染 OSD 文字 ──
static void osd_render_time(PlayerState *ps) {
    if (!ps->font) return;

    // 格式化时间文字
    char text[64];
    int cur_min = (int)ps->audio_clock / 60;
    int cur_sec = (int)ps->audio_clock % 60;
    double duration = ps->fmt_ctx->duration / (double)AV_TIME_BASE;
    int dur_min = (int)duration / 60;
    int dur_sec = (int)duration % 60;
    snprintf(text, sizeof(text), "%02d:%02d / %02d:%02d",
             cur_min, cur_sec, dur_min, dur_sec);

    // TTF_RenderUTF8_Blended 将文字渲染为带 Alpha 通道的 SDL_Surface
    // 参数：
    //   font  —— 字体
    //   text  —— UTF-8 文本
    //   color —— 文字颜色（SDL_Color: r, g, b, a）
    // Blended 模式生成抗锯齿文字，质量最高但最慢
    // 其他模式：Solid（快但有锯齿）、Shaded（带背景色）
    SDL_Color white = {255, 255, 255, 255};
    SDL_Surface *surface = TTF_RenderUTF8_Blended(ps->font, text, white);
    if (!surface) return;

    // 将 Surface 转换为纹理
    SDL_Texture *tex = SDL_CreateTextureFromSurface(ps->renderer, surface);
    SDL_FreeSurface(surface);  // Surface 已复制到 GPU，可以释放
    if (!tex) return;

    // 设定文字显示位置（右下角，留 10px 边距）
    SDL_Rect dst;
    SDL_QueryTexture(tex, NULL, NULL, &dst.w, &dst.h);
    dst.x = ps->video_width - dst.w - 10;   // 右对齐
    dst.y = ps->video_height - dst.h - 10;   // 底部对齐

    // 渲染文字纹理（叠加在视频画面上）
    SDL_RenderCopy(ps->renderer, tex, NULL, &dst);

    SDL_DestroyTexture(tex);  // 每帧重建纹理（简单但有开销）
}

// ── 在 video_refresh 中使用 ──
static void video_refresh(PlayerState *ps) {
    // ... 原有的解码和渲染代码 ...

    SDL_RenderClear(ps->renderer);
    SDL_RenderCopy(ps->renderer, ps->texture, NULL, NULL);

    // 在视频帧上叠加 OSD（必须在 RenderCopy 之后、RenderPresent 之前）
    osd_render_time(ps);

    SDL_RenderPresent(ps->renderer);  // 提交渲染（包含视频 + OSD）
}

// ── 清理 ──
static void osd_cleanup(PlayerState *ps) {
    if (ps->font) TTF_CloseFont(ps->font);
    TTF_Quit();
}
```

### 性能优化提示

上面的实现每帧都创建和销毁文字纹理，有优化空间：

```c
// 优化：只在时间变化时重建纹理
static SDL_Texture *cached_time_tex = NULL;
static int cached_second = -1;

static void osd_render_time_cached(PlayerState *ps) {
    int cur_sec = (int)ps->audio_clock;
    if (cur_sec != cached_second) {
        // 时间变化了，重建纹理
        if (cached_time_tex) SDL_DestroyTexture(cached_time_tex);
        // ... 创建新纹理 ...
        cached_second = cur_sec;
    }
    // 直接使用缓存的纹理渲染
    SDL_RenderCopy(ps->renderer, cached_time_tex, NULL, &dst);
}
```

### 关键设计决策回顾

| 决策 | 选择 | 原因 |
|------|------|------|
| 音频模式 | Pull（SDL 回调） | 天然节奏控制，精确时钟基准 |
| 同步基准 | 音频主时钟 | 音频硬件时钟最稳定 |
| 同步阈值 | ±50ms | 低于人耳感知阈值（80-100ms） |
| 视频渲染线程 | 主线程 | SDL 要求渲染 API 在主线程调用 |
| 像素上传 | YUV 直传 | 比 CPU 端 YUV→RGB 转换快 3-5x |
| 队列保护 | mutex + condvar | 标准生产者-消费者同步方案 |
| 队列容量 | 15MB 上限 | 平衡缓冲充足性和内存占用 |

## 本节小结

- 进度条显示：利用 `\r` 回车符实现控制台原地更新，每秒刷新一次避免性能影响
- 帧队列：使用环形缓冲区缓存解码后的帧，避免"解码了又丢"的浪费，是 ffplay 和 KrKr2 的标准做法
- OSD 文字：使用 SDL_ttf 库在视频画面上叠加文字，可通过纹理缓存优化性能
- 架构总结：3 个线程（主/解封装/SDL音频）+ 2 个队列 + 1 个时钟构成了完整的播放器
- KrKr2 映射：简单播放器的每个组件都能在 KrKr2 中找到对应的更复杂版本

## 常见错误

| 错误 | 原因 | 解决方案 |
|------|------|----------|
| 进度条不更新 | `fflush(stderr)` 缺失 | stderr 可能是行缓冲，需手动刷新 |
| OSD 文字闪烁 | 在 `SDL_RenderClear` 之前渲染 OSD | OSD 必须在 `SDL_RenderClear` 之后、`SDL_RenderPresent` 之前 |
| 帧队列死锁 | `SDL_CondWait` 未配对 `SDL_CondSignal` | 确保 push/next 中都有 Signal 调用 |
| 字体加载失败 | 字体路径错误或字体不存在 | 使用跨平台的系统字体路径，并做好 fallback |

## 练习题与答案

### 题目 1：为什么帧队列大小选择 4 帧而不是 1 帧或 100 帧？

<details>
<summary>查看答案</summary>

**1 帧（太小）的问题**：
- 没有缓冲余量，任何解码延迟都会导致渲染端饥饿（没有帧可渲染）
- 超前帧无处存放，只能丢弃，回到"即解即渲染"的低效模式
- 丢帧追赶时没有备选帧

**100 帧（太大）的问题**：
- 内存占用过高：每帧 1920×1080 YUV420P ≈ 3MB，100 帧 = 300MB
- Seek 时需要丢弃大量缓存帧，响应延迟
- 解码线程持续大量超前解码，CPU 和电池消耗增加

**4 帧（经验值）的平衡**：
- 内存：4 × 3MB = 12MB（可接受）
- 缓冲：30fps 视频下覆盖 ~133ms，足够应对短暂的解码波动
- Seek 响应：最多丢弃 4 帧，延迟极小
- 与 KrKr2 的选择（6帧）和 ffplay 的选择（16帧+但按需分配）在同一数量级

生产级播放器通常根据视频分辨率和设备内存动态调整帧队列大小。

</details>

### 题目 2：如果不调用 `osd_cleanup` 就退出程序，会发生什么？

<details>
<summary>查看答案</summary>

**直接后果**：

1. **`TTF_CloseFont` 未调用**：字体文件句柄未关闭，字体数据占用的内存未释放。但由于进程即将退出，操作系统会回收所有进程内存和文件句柄。

2. **`TTF_Quit` 未调用**：SDL_ttf 的内部状态（FreeType 库实例等）未正确清理。同样，进程退出后操作系统会处理。

**实际影响**：

- **正常退出的程序**：几乎没有可观测的影响，操作系统会清理一切。
- **长运行程序**（如反复创建/销毁播放器实例）：每次未清理都会泄漏内存和句柄，最终导致内存耗尽或文件句柄用完。
- **Valgrind 等内存检测工具**：会报告内存泄漏，增加排查真正问题的难度。

**最佳实践**：始终正确清理资源，即使进程即将退出。原因：
1. 养成良好的编程习惯
2. 代码可能被重构为库函数（多次调用）
3. 方便使用内存检测工具排查问题
4. 某些资源（如 GPU 纹理、音频设备）不一定会被操作系统正确回收

</details>

### 题目 3：将进度条从控制台输出改为 SDL 窗口内的图形进度条

<details>
<summary>查看答案</summary>

```c
// 在 video_refresh 的 SDL_RenderPresent 之前添加：
static void render_progress_bar(PlayerState *ps) {
    double duration = ps->fmt_ctx->duration / (double)AV_TIME_BASE;
    if (duration <= 0) return;

    double percent = ps->audio_clock / duration;
    if (percent > 1.0) percent = 1.0;
    if (percent < 0.0) percent = 0.0;

    // 进度条参数
    int bar_height = 6;     // 进度条高度（像素）
    int bar_y = ps->video_height - bar_height - 4;  // 底部留 4px 边距
    int bar_x = 10;         // 左边距
    int bar_width = ps->video_width - 20;  // 两侧各留 10px
    int filled_width = (int)(bar_width * percent);

    // 绘制背景条（半透明灰色）
    // SDL_SetRenderDrawBlendMode 启用 Alpha 混合
    // 使透明色可以叠加在视频画面上
    SDL_SetRenderDrawBlendMode(ps->renderer, SDL_BLENDMODE_BLEND);

    // 背景：半透明白色
    SDL_SetRenderDrawColor(ps->renderer, 255, 255, 255, 80);
    SDL_Rect bg_rect = {bar_x, bar_y, bar_width, bar_height};
    SDL_RenderFillRect(ps->renderer, &bg_rect);

    // 已播放部分：不透明亮蓝色
    SDL_SetRenderDrawColor(ps->renderer, 66, 133, 244, 220);
    SDL_Rect fill_rect = {bar_x, bar_y, filled_width, bar_height};
    SDL_RenderFillRect(ps->renderer, &fill_rect);

    // 播放头：白色小圆点（用矩形近似）
    int dot_size = bar_height + 6;
    SDL_SetRenderDrawColor(ps->renderer, 255, 255, 255, 255);
    SDL_Rect dot_rect = {
        bar_x + filled_width - dot_size / 2,
        bar_y - 3,
        dot_size, dot_size
    };
    SDL_RenderFillRect(ps->renderer, &dot_rect);

    // 恢复默认混合模式
    SDL_SetRenderDrawBlendMode(ps->renderer, SDL_BLENDMODE_NONE);
}

// 在 video_refresh 中调用：
static void video_refresh(PlayerState *ps) {
    // ... 解码和视频纹理渲染 ...

    SDL_RenderClear(ps->renderer);
    SDL_RenderCopy(ps->renderer, ps->texture, NULL, NULL);

    // 叠加进度条
    render_progress_bar(ps);

    // 叠加 OSD 文字（如果有）
    osd_render_time(ps);

    SDL_RenderPresent(ps->renderer);
}
```

**关键技术点**：
1. `SDL_SetRenderDrawBlendMode(SDL_BLENDMODE_BLEND)` 启用 Alpha 混合，让进度条半透明叠加在视频上
2. `SDL_RenderFillRect` 绘制填充矩形，用于进度条的背景和前景
3. 渲染顺序必须是：视频帧 → 进度条 → OSD 文字 → `SDL_RenderPresent`
4. 播放头用小矩形近似圆点（真正的圆需要自己画或用纹理）

</details>
