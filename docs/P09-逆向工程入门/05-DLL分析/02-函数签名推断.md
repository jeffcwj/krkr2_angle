# 函数签名推断

## 本節目標

- 掌握从反汇编代码推断函数参数个数的方法
- 学会识别参数类型（整数、指针、浮点、结构体）
- 理解返回值类型的推断技巧
- 掌握 C++ 成员函数（thiscall）的识别与参数还原
- 能够从 KiriKiri2 的反汇编代码中还原完整的函数原型

## 5.2.1 参数个数推断

### 从调用点推断

最直接的方法是观察**调用方**在调用函数前压入了多少参数：

```nasm
; 调用方代码（cdecl/stdcall 约定）
push    0                   ; 参数 4: 0
push    dword ptr [ebp-10h] ; 参数 3: 局部变量
push    ecx                 ; 参数 2: 某个值
push    eax                 ; 参数 1: 某个值
call    sub_401000          ; 调用目标函数
add     esp, 10h            ; 清理 16 字节 = 4 个参数（cdecl）

; 结论: sub_401000 有 4 个参数
; 如果没有 add esp（stdcall），则参数个数从 push 计数
; 或查看函数内的 ret N 指令
```

### 从函数体推断

观察函数内部如何访问栈参数：

```nasm
; 标准 cdecl/stdcall 函数
sub_401000:
    push    ebp
    mov     ebp, esp
    sub     esp, 20h            ; 分配局部变量空间
    
    mov     eax, [ebp+8]        ; 参数 1
    mov     ecx, [ebp+0Ch]      ; 参数 2（0x0C = 12 = 8+4）
    mov     edx, [ebp+10h]      ; 参数 3（0x10 = 16 = 8+8）
    cmp     dword ptr [ebp+14h], 0  ; 参数 4（0x14 = 20 = 8+12）
    ; 没有访问 [ebp+18h] → 最多 4 个参数

; 栈布局（调用后）:
;   [ebp+14h] = 参数 4
;   [ebp+10h] = 参数 3
;   [ebp+0Ch] = 参数 2
;   [ebp+08h] = 参数 1
;   [ebp+04h] = 返回地址
;   [ebp+00h] = 保存的 EBP（旧帧指针）
;   [ebp-04h] = 第一个局部变量
;   [ebp-08h] = 第二个局部变量
;   ...
```

### 参数个数速查表

```
ebp+偏移    含义              参数编号
─────────   ───────────────   ────────
[ebp+08h]   第 1 个参数       arg_0
[ebp+0Ch]   第 2 个参数       arg_4
[ebp+10h]   第 3 个参数       arg_8
[ebp+14h]   第 4 个参数       arg_C
[ebp+18h]   第 5 个参数       arg_10
...         每个参数 +4        ...

; 对于 thiscall，ECX = this，栈上第一个参数从 [ebp+8] 开始
; 对于 fastcall，ECX = arg1, EDX = arg2，栈上从 [ebp+8] = arg3 开始
```

### 从 ret 指令推断（stdcall）

```nasm
; stdcall 函数的 ret 指令带参数清理值
sub_401000:
    push    ebp
    mov     ebp, esp
    ...
    pop     ebp
    ret     10h             ; 清理 0x10 = 16 字节 = 4 个 DWORD 参数

; 结论: 4 个参数，stdcall 调用约定

; 对比 cdecl:
sub_402000:
    push    ebp
    mov     ebp, esp
    ...
    pop     ebp
    ret                     ; 不清理参数 → cdecl
                            ; 参数由调用方清理（add esp, N）
```

## 5.2.2 参数类型推断

### 整数/指针类型

```nasm
; 线索 1: 参数直接用于比较
mov     eax, [ebp+8]
cmp     eax, 0              ; 与 0 比较 → 可能是指针（检查 NULL）或整数
jz      error_path

; 线索 2: 参数作为指针解引用
mov     eax, [ebp+8]
mov     ecx, [eax]          ; 解引用 → 参数是指针
mov     edx, [eax+4]        ; 访问偏移 → 指向结构体的指针

; 线索 3: 参数用于算术运算
mov     eax, [ebp+8]
add     eax, [ebp+0Ch]      ; 两个参数相加 → 整数类型
imul    eax, 4               ; 乘法 → 整数

; 线索 4: 参数传递给已知 API
push    dword ptr [ebp+8]
call    ds:CreateFileW       ; 已知第一个参数是 LPCWSTR → 参数是 wchar_t*
```

### 浮点类型

```nasm
; 线索 1: 使用 FPU 指令加载参数
fld     qword ptr [ebp+8]   ; 加载 8 字节 → double 类型
fld     dword ptr [ebp+8]   ; 加载 4 字节 → float 类型

; 线索 2: 使用 SSE 指令
movss   xmm0, [ebp+8]       ; float（SSE）
movsd   xmm0, [ebp+8]       ; double（SSE2）

; 线索 3: 浮点运算
fld     qword ptr [ebp+8]   ; 加载参数
fmul    qword ptr [ebp+10h] ; 与另一参数相乘
fstp    qword ptr [ebp-8]   ; 存储结果
; 两个参数都是 double

; 注意: double 占 8 字节，在 32 位栈上占两个 DWORD 位置
; [ebp+8] 和 [ebp+C] 共同构成第一个 double 参数
; [ebp+10] 和 [ebp+14] 共同构成第二个 double 参数
```

### 字符串类型

```nasm
; 线索 1: 传递给字符串 API
push    dword ptr [ebp+8]
call    ds:lstrlenW          ; wchar_t* 参数

push    dword ptr [ebp+0Ch]
push    dword ptr [ebp+8]
call    ds:lstrcpyA          ; char* 参数

; 线索 2: 逐字节/逐字访问
mov     eax, [ebp+8]
movzx   ecx, byte ptr [eax]  ; 按字节读取 → char* (ANSI)
movzx   ecx, word ptr [eax]  ; 按字读取   → wchar_t* (Unicode)

; 线索 3: 与字符串常量比较
push    offset aV2Link       ; "V2Link"
push    dword ptr [ebp+8]
call    ds:strcmp             ; 参数是 const char*

; KiriKiri2 使用 TJS 字符串（tTJSString / ttstr）
; 内部是 wchar_t*，但外层包装了引用计数
mov     eax, [ebp+8]         ; ttstr* 或 tTJSString*
mov     ecx, [eax+4]         ; 访问内部 wchar_t* 指针
```

### 结构体/类指针

```nasm
; 线索 1: 访问多个固定偏移
mov     eax, [ebp+8]         ; 参数可能是结构体指针
mov     ecx, [eax]           ; 偏移 +0: 可能是虚函数表
mov     edx, [eax+4]         ; 偏移 +4: 成员 1
mov     esi, [eax+8]         ; 偏移 +8: 成员 2
cmp     dword ptr [eax+0Ch], 0  ; 偏移 +C: 成员 3

; 推断结构体布局:
; struct Unknown {
;     void** vtable;    // +0: 虚函数表 → 这是一个多态类
;     DWORD  member1;   // +4
;     DWORD  member2;   // +8
;     DWORD  member3;   // +C
; };

; 线索 2: 通过虚函数表调用
mov     eax, [ebp+8]         ; this 指针
mov     ecx, [eax]           ; vtable
call    dword ptr [ecx+10h]  ; vtable[4]（第 5 个虚函数）
; 参数是一个有虚函数的 C++ 对象指针
```

### 类型推断决策树

```
参数 [ebp+N]
├── 被解引用 [eax] → 指针类型
│   ├── [eax] 又被当作函数指针调用 → 虚函数表 → C++ 对象指针
│   ├── 传递给 lstrlenW/wcscpy → wchar_t*
│   ├── 传递给 strlen/strcpy → char*
│   ├── 按固定偏移访问多个成员 → 结构体指针
│   └── 只访问 [eax] → void* 或简单指针
├── FPU/SSE 加载 → 浮点类型
│   ├── fld dword/movss → float
│   └── fld qword/movsd → double
├── 算术运算（add/sub/imul） → 整数类型
│   ├── 有符号比较（jl/jg/jle/jge） → int / long
│   └── 无符号比较（jb/ja/jbe/jae） → unsigned int
├── 与 0 比较后跳转 → 可能是指针（NULL check）或布尔值
└── 直接传递给已知 API → 查 API 文档确定类型
```

## 5.2.3 返回值推断

### EAX 返回值

```nasm
; 大多数函数通过 EAX 返回
sub_401000:
    ...
    mov     eax, 1              ; 返回 1 → 布尔值 TRUE 或 HRESULT S_OK
    ret

; 调用方检查返回值:
call    sub_401000
test    eax, eax               ; 检查 EAX 是否为 0
jz      failure                ; 为 0 则失败 → 返回值是 BOOL 或指针

call    sub_401000
cmp     eax, 0                 ; 与 0 比较
jl      error                  ; 小于 0 则错误 → 返回值是 HRESULT 或有符号整数

call    sub_401000
mov     [ebp-4], eax           ; 保存返回值
mov     ecx, [eax]             ; 解引用 → 返回值是指针
```

### 浮点返回值

```nasm
; 浮点返回值通过 FPU 栈顶（ST0）或 XMM0
sub_402000:
    fld     qword ptr [ebp+8]
    fadd    qword ptr [ebp+10h]
    ret                         ; ST0 = 返回值 → double

; 调用方:
call    sub_402000
fstp    qword ptr [ebp-8]      ; 从 ST0 取出 → 返回值是 double
```

### void 返回值

```nasm
; 调用方完全不使用 EAX 返回值
call    sub_403000
; 下一条指令直接做其他事，不检查 EAX
push    esi
call    sub_403100
; → sub_403000 可能返回 void
```

### 返回值类型速查

```
调用方行为                       推断类型
──────────────────────           ────────────
test eax, eax / jz              BOOL 或指针
cmp eax, 0 / jl                 HRESULT 或 int
mov [var], eax / mov ecx,[eax]  指针类型
完全忽略 EAX                     void
fstp dword ptr [var]            float
fstp qword ptr [var]            double
movss [var], xmm0               float (SSE)
movsd [var], xmm0               double (SSE2)
```

## 5.2.4 C++ 成员函数识别

### thiscall 约定识别

```nasm
; C++ 成员函数（MSVC thiscall 约定）
; ECX = this 指针

; 调用方:
lea     ecx, [ebp-20h]         ; ECX = 对象地址（this）
push    42h                    ; 参数 2
push    eax                    ; 参数 1
call    sub_401000             ; 调用成员函数

; 被调用方:
sub_401000:
    push    ebp
    mov     ebp, esp
    push    ecx                ; 保存 this 到栈上（常见模式）
    mov     eax, [ecx]         ; 通过 this 访问 vtable
    mov     edx, [ecx+4]       ; 通过 this 访问成员
    mov     esi, [ebp+8]       ; 第一个显式参数
    mov     edi, [ebp+0Ch]     ; 第二个显式参数
```

### 虚函数调用识别

```nasm
; 虚函数调用的典型模式
mov     ecx, [ebp-10h]        ; ECX = this（对象指针）
mov     eax, [ecx]            ; EAX = *this = vtable 指针
push    42h                   ; 参数
call    dword ptr [eax+0Ch]   ; vtable[3]（第 4 个虚函数）

; 翻译为 C++:
// this->vtable[3](this, 0x42)
// 即 this->VirtualFunc3(0x42)

; vtable 偏移与虚函数索引的关系:
; [eax+00h] = vtable[0] = 第 1 个虚函数（通常是析构函数）
; [eax+04h] = vtable[1] = 第 2 个虚函数
; [eax+08h] = vtable[2] = 第 3 个虚函数
; [eax+0Ch] = vtable[3] = 第 4 个虚函数
; 公式: 索引 = 偏移 / 4（32 位）
```

### KiriKiri2 中的 iTJSDispatch2 虚函数

```nasm
; iTJSDispatch2 是 KiriKiri2 的核心接口
; 所有 TJS 对象都实现此接口

; 典型的虚函数调用:
mov     ecx, [ebp+8]          ; ECX = iTJSDispatch2* obj
mov     eax, [ecx]            ; EAX = vtable
push    0                     ; membername = NULL
push    offset aCount         ; "Count"
push    edx                   ; result
push    0                     ; flag
push    ecx                   ; objthis
call    dword ptr [eax+34h]   ; vtable[13] = PropGet

; 还原为 C++ 代码:
// obj->PropGet(0, result, L"Count", NULL, obj);
```

## 5.2.5 综合实例：还原完整函数原型

### 实例 1：简单 C 函数

```nasm
; 反汇编代码:
sub_401000:
    push    ebp
    mov     ebp, esp
    mov     eax, [ebp+8]       ; 参数 1
    mov     ecx, [ebp+0Ch]     ; 参数 2
    test    eax, eax            ; NULL check → 参数 1 是指针
    jz      short loc_401020
    movzx   edx, word ptr [eax] ; 按 word 读取 → wchar_t*
    test    dx, dx
    jz      short loc_401020
    push    ecx                 ; 参数 2 传给 wcscpy
    push    eax                 ; 参数 1 传给 wcscpy
    call    ds:wcscpy
    add     esp, 8
    mov     eax, 1              ; 返回 TRUE
    jmp     short loc_401025
loc_401020:
    xor     eax, eax            ; 返回 FALSE
loc_401025:
    pop     ebp
    ret                         ; cdecl（无清理）

; 推断过程:
; 1. 参数个数: 2 个（[ebp+8], [ebp+0Ch]）
; 2. 参数 1: 被解引用为 word → wchar_t*，有 NULL check → 可能为空
; 3. 参数 2: 直接传给 wcscpy 的第二个参数 → wchar_t*（目标缓冲区）
; 4. 返回值: 0 或 1 → BOOL
; 5. 调用约定: ret 无参数，调用方 add esp → cdecl

; 还原结果:
// BOOL __cdecl CopyString(const wchar_t* src, wchar_t* dst);
```

### 实例 2：C++ 成员函数

```nasm
; 反汇编代码:
sub_402000:
    push    ebp
    mov     ebp, esp
    sub     esp, 8
    mov     [ebp-4], ecx       ; 保存 this
    mov     eax, [ebp+8]       ; 参数 1
    mov     ecx, [ebp-4]       ; this
    mov     edx, [ecx+10h]     ; this->member_at_0x10
    add     edx, eax            ; member += arg1
    mov     [ecx+10h], edx     ; 写回
    cmp     edx, [ecx+14h]     ; 与 this->member_at_0x14 比较
    jle     short loc_402030
    mov     edx, [ecx+14h]     ; 限制最大值
    mov     [ecx+10h], edx
loc_402030:
    mov     eax, [ecx+10h]     ; 返回更新后的值
    leave
    ret     4                   ; stdcall 清理 4 字节 → 1 个参数

; 推断过程:
; 1. ECX = this（thiscall），保存到 [ebp-4]
; 2. 一个显式参数 [ebp+8]，用于加法 → int 类型
; 3. 访问 this+0x10 和 this+0x14 → 类有两个 int 成员
; 4. 返回 EAX = this->member_at_0x10 → int
; 5. ret 4 → thiscall（由被调用方清理参数）

; 推断的类布局:
// class Unknown {
//     ...                    // +0x00 到 +0x0C: 其他成员
//     int currentValue;      // +0x10
//     int maxValue;          // +0x14
// public:
//     int __thiscall AddValue(int delta);
// };
```

## 5.2.6 常见错误与解决方案

### 错误 1：将局部变量误认为参数

```
症状: 推断的参数个数过多
原因: 混淆了 [ebp+N]（参数）和 [ebp-N]（局部变量）

关键区分:
  [ebp+8], [ebp+C], ... → 参数（正偏移，在返回地址之上）
  [ebp-4], [ebp-8], ... → 局部变量（负偏移，在帧指针之下）
  [ebp+4]               → 返回地址（不是参数）
  [ebp+0]               → 保存的旧 EBP（不是参数）

常见陷阱:
  没有建立标准栈帧（FPO 函数）时，ESP 直接用于访问
  此时 [esp+N] 的含义随 push/pop 操作而变化
  需要追踪 ESP 的变化来确定实际偏移
```

### 错误 2：忽略隐含参数（this 指针）

```
症状: thiscall 函数缺少 this 参数
原因: ECX 中的 this 指针没有被计入参数

正确做法:
  如果函数开头 push ecx 或 mov [ebp-N], ecx
  且 ECX 未通过栈传入 → 这是 thiscall
  实际参数 = ECX(this) + 栈上参数

  C++ 原型: int CMyClass::MyFunc(int a, int b)
  汇编层面: ECX=this, [ebp+8]=a, [ebp+C]=b
  三个参数，但 C++ 声明只写两个（this 隐含）
```

### 错误 3：double 参数占两个栈槽

```
症状: 参数个数推断错误（少一个）
原因: double 类型占 8 字节，在 32 位栈上占两个 DWORD 位置

示例:
  void func(int a, double b, int c);
  栈布局:
    [ebp+08h] = a (int, 4 bytes)
    [ebp+0Ch] = b 低 32 位  ─┐
    [ebp+10h] = b 高 32 位  ─┘ double b (8 bytes)
    [ebp+14h] = c (int, 4 bytes)

  如果看到 fld qword ptr [ebp+0Ch]，说明 [ebp+0C] 和 [ebp+10] 
  共同组成一个 double 参数，不要误认为两个 int 参数。
```

## 本節小結

1. **参数个数**：通过调用方的 push 指令、函数体的 [ebp+N] 访问、ret N 指令三种方式推断
2. **参数类型**：根据参数的使用方式（解引用→指针、FPU加载→浮点、传给已知API→查文档）推断
3. **返回值**：观察调用方如何使用 EAX（test→BOOL、解引用→指针、忽略→void）
4. **C++ 特殊性**：thiscall 的 ECX=this 是隐含参数，虚函数通过 vtable 间接调用
5. **double 陷阱**：32 位环境下 double 占两个栈槽，容易被误认为两个参数

### 题目 1：推断函数原型

**问题**：根据以下反汇编代码，推断完整的函数原型（包括调用约定、返回类型、参数类型）。

```nasm
sub_410000:
    push    ebp
    mov     ebp, esp
    fld     qword ptr [ebp+8]     ; 加载 8 字节浮点
    fmul    qword ptr [ebp+10h]   ; 乘以另一个 8 字节浮点
    fadd    qword ptr [ebp+18h]   ; 加上第三个 8 字节浮点
    pop     ebp
    ret                            ; 无参数清理
```

<details>
<summary>查看答案</summary>

```cpp
double __cdecl sub_410000(double a, double b, double c);
// 等价于: return a * b + c;
```

推断过程：
1. **调用约定**：`ret` 无参数 → cdecl（调用方清理栈）
2. **参数类型**：三个 `qword ptr` 访问 + FPU 指令 → 三个 double
3. **参数布局**：
   - `[ebp+8]` = 参数 1（double，占 [ebp+8] 和 [ebp+C]）
   - `[ebp+10]` = 参数 2（double，占 [ebp+10] 和 [ebp+14]）
   - `[ebp+18]` = 参数 3（double，占 [ebp+18] 和 [ebp+1C]）
   - 每个 double 8 字节，偏移间隔 8
4. **返回值**：FPU 栈顶 ST(0) → double
5. **功能**：`a * b + c`（fused multiply-add）

</details>

### 题目 2：识别 thiscall

**问题**：以下代码是否是 C++ 成员函数？如果是，this 指针在哪里？有几个显式参数？

```nasm
sub_420000:
    push    ebp
    mov     ebp, esp
    push    ecx
    mov     [ebp-4], ecx
    mov     eax, [ecx]
    push    dword ptr [ebp+8]
    call    dword ptr [eax+1Ch]
    test    eax, eax
    jnz     short loc_420020
    mov     ecx, [ebp-4]
    mov     dword ptr [ecx+0Ch], 1
loc_420020:
    leave
    ret     4
```

<details>
<summary>查看答案</summary>

**是 C++ 成员函数（thiscall）**。

分析：
1. **this 指针**：`ECX`，在 `mov [ebp-4], ecx` 处保存到局部变量
2. **显式参数**：1 个 → `[ebp+8]`，由 `ret 4` 确认（清理 4 字节）
3. **虚函数调用**：`mov eax, [ecx]` + `call [eax+1Ch]` → 调用 vtable[7]
4. **成员访问**：`mov dword ptr [ecx+0Ch], 1` → 设置 this+0xC 成员为 1

还原的 C++ 代码：
```cpp
// this->vtable[7](this, arg1)
// if (result == 0) this->member_at_0C = 1;

// 类布局:
// class Unknown {
//     void** vtable;   // +0
//     ...              // +4, +8
//     int  flag;       // +C
// public:
//     void __thiscall SomeMethod(int arg1);
// };
```

完整原型：`void __thiscall Unknown::SomeMethod(int arg1);`

</details>
