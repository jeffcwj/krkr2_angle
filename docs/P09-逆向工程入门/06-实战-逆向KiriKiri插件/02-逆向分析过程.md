## 6.2.1 逆向分析的系统化流程

逆向一个 KiriKiri2 插件 DLL，经历四个阶段：

```
阶段 1: 信息收集          阶段 2: 静态分析          阶段 3: 动态验证
┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
│ 文件基本属性      │    │ IDA/Ghidra 加载   │    │ x64dbg 加载运行  │
│ PE 头 / 导入导出  │ →  │ V2Link 反编译     │ →  │ 断点验证推测     │
│ 字符串提取       │    │ 注册逻辑追踪      │    │ 参数值 / 内存确认 │
│ 依赖库识别       │    │ 关键函数分析      │    │ 返回值检查       │
└─────────────────┘    └─────────────────┘    └─────────────────┘
                              │
                              ▼
                       阶段 4: 知识整合 → 准备还原源码
```

本节以假设的 `imageEx.dll`（图像处理扩展插件，无源码）为例，演示完整逆向过程。所有技术完全适用于项目 `doc/krkr2_plugins.md` 中列出的无源码插件。

## 6.2.2 阶段一：信息收集

### 步骤 1：文件属性与 PE 头

## 6.2.3 阶段二：IDA/Ghidra 静态分析

### 步骤 1：加载与配置

**IDA**：File → Open → 选择 DLL，确认 MetaPC (x86)，等待自动分析完成。

**Ghidra**：File → Import File，Language 自动识别 x86:LE:32。分析时确保勾选 **Windows x86 PE RTTI Analyzer**（识别 C++ 类）和 **Decompiler Parameter ID**。

### 步骤 2：从 V2Link 入手

在 Exports 窗口找到 V2Link，按 F5（IDA）或双击（Ghidra）反编译：

```c
// IDA 初始反编译（未标注类型）
int __stdcall V2Link(int a1) {
    int result = TVPInitImportStub(a1);
    if (result >= 0) {
        sub_10001200();   // 未知函数 1
        sub_10001350();   // 未知函数 2
        result = 0;
    }
    return result;
}
```

### 步骤 3：导入 KiriKiri2 类型定义

为提升反编译可读性，导入类型。在 IDA 中 File → Load file → C header file，或在 Ghidra 中 File → Parse C Source：

```c
// krkr2_types.h — 简化类型定义
typedef unsigned int tjs_uint;
typedef int tjs_int;
typedef int tjs_error;
typedef wchar_t tjs_char;

struct iTVPFunctionExporter_vtbl {
    bool (__stdcall *QueryFunctions)(void*, const tjs_char**, void**, tjs_uint);
    bool (__stdcall *QueryFunctionsByNarrow)(void*, const char**, void**, tjs_uint);
};
struct iTVPFunctionExporter { struct iTVPFunctionExporter_vtbl* vtbl; };

struct iTJSDispatch2_vtbl {
    tjs_uint (__stdcall *AddRef)(void*);     // +0x00
    tjs_uint (__stdcall *Release)(void*);    // +0x04
    tjs_error (__stdcall *FuncCall)(void*,   // +0x08
        tjs_uint, const tjs_char*, tjs_uint*, void*, tjs_int, void**, void*);
    tjs_error (__stdcall *FuncCallByNum)(void*, tjs_uint, tjs_int, // +0x0C
        void*, tjs_int, void**, void*);
    tjs_error (__stdcall *PropGet)(void*,    // +0x10
        tjs_uint, const tjs_char*, tjs_uint*, void*, void*);
    tjs_error (__stdcall *PropGetByNum)(void*, tjs_uint, tjs_int,  // +0x14
        void*, void*);
    tjs_error (__stdcall *PropSet)(void*,    // +0x18
        tjs_uint, const tjs_char*, tjs_uint*, const void*, void*);
};
struct iTJSDispatch2 { struct iTJSDispatch2_vtbl* vtbl; };
```

### 步骤 4：追踪注册逻辑

标注类型后，sub_10001200 变为可读的 ncbind 注册函数：

```c
// 经过类型标注和变量重命名后
void ncbind_autoregist_callback() {
    iTJSDispatch2* global = TVPGetScriptDispatch();
    if (!global) return;
    
    iTJSDispatch2* cls = TJSCreateNativeClass(L"ImageFilter");
    
    // 注册构造函数 — 第5参数 0x01 = TJS_NCM_CONSTRUCTOR
    TJSNativeClassRegisterNCM(cls, L"ImageFilter",
        sub_10001500, "NativeImageFilter", 0x01);
    
    // 注册方法 — 第5参数 0x00 = TJS_NCM_METHOD
    TJSNativeClassRegisterNCM(cls, L"adjustBrightness",
        sub_10001600, "adjustBrightness", 0x00);
    TJSNativeClassRegisterNCM(cls, L"adjustContrast",
        sub_10001700, "adjustContrast", 0x00);
    TJSNativeClassRegisterNCM(cls, L"gaussianBlur",
        sub_10001800, "gaussianBlur", 0x00);
    
    // 注册属性 — 0x20 = getter, 0x40 = setter
    TJSNativeClassRegisterNCM(cls, L"radius",
        sub_10001900, "radius", 0x20);
    TJSNativeClassRegisterNCM(cls, L"radius",
        sub_10001950, "radius", 0x40);
    
    // 注册到全局 TJS 对象
    tTJSVariant var;
    var.type = 0x06; // TJS_VT_OBJECT
    var.obj  = cls;
    global->vtbl->PropSet(global, 1, L"ImageFilter", NULL, &var, global);
    global->vtbl->Release(global);
    cls->vtbl->Release(cls);
}
```

> **IDA 操作技巧**：右键参数 → Set type 设置类型；右键变量 → Rename 改名；Ctrl+X 查看交叉引用追踪回调函数地址。

## 6.2.4 关键函数深入分析

### adjustBrightness 方法还原

```c
// 地址: 0x10001600 — 标注后的伪代码
tjs_error __fastcall NativeImageFilter_adjustBrightness(
    NativeImageFilter* this_ptr,
    tjs_int numparams, tTJSVariant** params, tTJSVariant* result)
{
    if (numparams < 1) return TJS_E_BADPARAMCOUNT;
    
    double brightness = TJSVariantToReal(params[0]);
    if (brightness < -1.0 || brightness > 1.0) {
        TVPThrowExceptionMessage(L"brightness must be between -1.0 and 1.0");
        return TJS_E_FAIL;
    }
    
    tjs_uint32* pixels = this_ptr->pixel_buffer;  // +0x04
    int total = this_ptr->width * this_ptr->height; // +0x08, +0x0C
    int adjustment = (int)(brightness * 255.0);
    
    for (int i = 0; i < total; i++) {
        tjs_uint32 px = pixels[i];
        int r = clamp(((px >> 16) & 0xFF) + adjustment, 0, 255);
        int g = clamp(((px >> 8)  & 0xFF) + adjustment, 0, 255);
        int b = clamp(( px        & 0xFF) + adjustment, 0, 255);
        pixels[i] = (px & 0xFF000000) | (r << 16) | (g << 8) | b;
    }
    return TJS_S_OK;
}
```

### NativeImageFilter 结构体还原

通过分析多个方法中对 this 指针的偏移访问，还原结构体：

```c
struct NativeImageFilter {
    void*         vtable;       // +0x00  C++ 虚函数表
    tjs_uint32*   pixel_buffer; // +0x04  ARGB 像素数据
    int           width;        // +0x08  宽度
    int           height;       // +0x0C  高度
    int           stride;       // +0x10  行字节步长
    int           radius;       // +0x14  模糊半径
    int           bpp;          // +0x18  每像素位数 (32)
};
```

**IDA**：Shift+F9 打开 Structures，Insert 新建，按 'd' 添加成员。**Ghidra**：Data Type Manager → New → Structure。创建后将 this 参数 Retype 为 `NativeImageFilter*`，反编译器自动用字段名替换偏移。

### gaussianBlur 算法识别

```c
// 地址: 0x10001800 — 核心逻辑
tjs_error NativeImageFilter_gaussianBlur(NativeImageFilter* this_ptr, ...) {
    int radius = this_ptr->radius;
    if (radius <= 0) return TJS_S_OK;
    
    int w = this_ptr->width, h = this_ptr->height;
    tjs_uint32* src = this_ptr->pixel_buffer;
    tjs_uint32* temp = (tjs_uint32*)malloc(w * h * 4);
    
    // sub_10001A00: 生成高斯核 — 识别特征: exp(-x*x/(2*sigma*sigma))
    double* kernel = calculate_gaussian_kernel(radius);
    
    // 两趟分离式高斯模糊：先水平后垂直
    // 水平方向: 对每个像素，用 kernel 加权 [-radius, +radius] 范围内的邻居
    for (int y = 0; y < h; y++)
        for (int x = 0; x < w; x++) {
            double r=0, g=0, b=0, a=0, wsum=0;
            for (int k = -radius; k <= radius; k++) {
                int sx = clamp(x+k, 0, w-1);
                tjs_uint32 px = src[y*w+sx];
                double wt = kernel[k+radius];
                r += ((px>>16)&0xFF)*wt; g += ((px>>8)&0xFF)*wt;
                b += (px&0xFF)*wt;       a += ((px>>24)&0xFF)*wt;
                wsum += wt;
            }
            temp[y*w+x] = ((int)(a/wsum)<<24)|((int)(r/wsum)<<16)
                         |((int)(g/wsum)<<8)|(int)(b/wsum);
        }
    // 垂直方向: 对 temp 同样处理，写回 src
    // ... 类似逻辑，改为遍历 y 轴邻居
    memcpy(src, temp, w*h*4);
    free(temp); free(kernel);
    return TJS_S_OK;
}
```

> **复杂算法识别技巧**：
> 1. 双重循环 + 卷积核权重 → 图像滤波
> 2. `exp()` 调用 + `sigma` 变量 → 高斯分布
> 3. 两趟独立处理（水平+垂直）→ 分离式卷积优化
> 4. 浮点用 FPU (fld/fmul/fadd) 或 SSE (movss/mulss/addss)

## 6.2.5 阶段三：x64dbg 动态验证

### 环境准备

```javascript
// startup.tjs — TJS 测试脚本
var filter = new ImageFilter();
filter.radius = 5;
filter.adjustBrightness(0.3);
filter.gaussianBlur();
```

### 拦截插件加载

```
1. x32dbg 打开 krkr.exe（KiriKiri2 是 32 位）
2. bp LoadLibraryW → F9 运行
3. 在断下时检查 [ESP+4] 是否为 imageEx.dll 路径
4. 确认后 F8 步过 → EAX = DLL 基址
5. bp GetProcAddress → 检查 [ESP+8] 为 "V2Link"
6. F8 → EAX = V2Link 地址 → bp EAX
7. F9 → 断在 V2Link 入口
```

### 验证结构体布局

```
1. 在构造函数 (sub_10001500) 设断点
2. 断下后 EAX = 新分配的对象地址
3. F8 逐步执行，观察内存写入:
   mov [eax+04h], 0    ; pixel_buffer = NULL
   mov [eax+08h], 0    ; width = 0
   mov [eax+0Ch], 0    ; height = 0
   mov [eax+14h], 3    ; radius = 3 (默认值!)
   mov [eax+18h], 20h  ; bpp = 32
4. 对比静态分析的结构体 → 确认/修正偏移
```

### 验证方法参数

```
1. bp 0x10001600 (adjustBrightness)
2. 断下后查看:
   this_ptr → 对象地址 → d [this+0x04] 查看像素数据
   params[0] 的 Real 值是否为 0.3
3. F8 步过后再次查看 pixel_buffer → 像素值已变化
4. 选取几个像素手动计算验证:
   原值 R=100 + adjustment(76) = 176 ✓
```

## 6.2.6 常见错误与解决方案

### 错误 1：反编译结果全是无意义名称

```
症状: sub_XXXXX、v1、a1 等符号，逻辑难以理解
原因: 未导入类型信息

解决: 从 V2Link 开始渐进式标注——先标注参数类型，
      再标注 TVPInitImportStub 等已知函数，
      然后根据字符串引用标注 ncbind 函数，
      最后创建结构体。每标注一处，反编译器自动更新。
```

### 错误 2：虚函数调用无法识别

```
症状: (**(code**)(*param_1 + 0x18))(...) 
原因: 反编译器不知道对象类型

解决: 将变量类型改为 iTJSDispatch2*，反编译器自动显示虚函数名。

iTJSDispatch2 vtable 偏移速查（32位）:
  +0x00 AddRef      +0x04 Release     +0x08 FuncCall
  +0x0C FuncCallByNum  +0x10 PropGet  +0x14 PropGetByNum
  +0x18 PropSet     +0x1C PropSetByNum +0x20 CreateNew
```

### 错误 3：x32dbg 中找不到 V2Link

```
原因: DLL 未加载（路径错误）或使用了 x64dbg（应用 x32dbg）
解决: 确认 32 位程序用 x32dbg；确认 DLL 在 plugin/ 目录下；
      用 bp LoadLibraryW 拦截加载时机。
```

## 本節小結

1. **系统化四阶段流程**是逆向成功的关键——信息收集 → 静态分析 → 动态验证 → 知识整合
2. **V2Link 是起点**——所有 KiriKiri2 插件分析从导出函数 V2Link 开始
3. **字符串是最佳线索**——宽字符串中的类名、方法名直接揭示插件功能
4. **类型标注是核心**——导入 KiriKiri2 类型定义后，反编译质量大幅提升
5. **ncbind 注册模式固定**——掌握展开模式后，可快速识别注册的类、方法和属性
6. **动态调试验证推测**——静态分析得出假设，x64dbg 断点验证实际值和执行流程

## 練習題与答案

### 题目 1：导出表分析

**问题**：一个 KiriKiri2 插件除了 V2Link 和 V2Unlink 外，还导出了 `GetPluginInfo`。这意味着什么？如何分析？

<details>
<summary>查看答案</summary>

```
标准 KiriKiri2 插件只导出 V2Link 和 V2Unlink。额外的 GetPluginInfo 可能是:
  a) 返回插件描述/版本信息的查询接口
  b) 某些 KiriKiri 扩展框架要求的附加接口

逆向方法:
1. 在 IDA/Ghidra 中反编译 GetPluginInfo
2. 观察返回值类型:
   - 返回 const wchar_t* → 插件描述字符串
   - 返回结构体指针 → 包含版本号、作者等
3. 在 x32dbg 中手动调用验证:
   在命令行输入 call imageEx.GetPluginInfo
   观察 EAX 是否指向可读字符串
4. 示例:
   const wchar_t* GetPluginInfo() {
       return L"ImageFilter Plugin v1.2 by Author";
   }
```

</details>

### 题目 2：虚函数表偏移计算

**问题**：判断以下代码调用了 iTJSDispatch2 的哪个虚函数，解释参数传递。

```nasm
mov     ecx, [ebp-14h]        ; (1) iTJSDispatch2* obj
mov     edx, [ecx]            ; (2) vtable 指针
push    ecx                   ; (3) objthis
push    0                     ; (4) hint = NULL
lea     eax, [ebp-30h]
push    eax                   ; (5) result
push    0                     ; (6) hint = NULL
push    offset aScore         ; (7) membername = L"score"
push    0                     ; (8) flag = 0
call    dword ptr [edx+10h]   ; (9) vtable[?]
```

<details>
<summary>查看答案</summary>

```
vtable 偏移 0x10 = 16 字节，索引 = 16/4 = 4

iTJSDispatch2 vtable 布局:
  [0] AddRef(+00) [1] Release(+04) [2] FuncCall(+08)
  [3] FuncCallByNum(+0C) [4] PropGet(+10) ★

vtable[4] = PropGet

参数（从右往左压栈，stdcall）:
  flag=0, membername=L"score", hint=NULL, result=[ebp-30h], hint=NULL, objthis=ecx

等价 C++: obj->PropGet(0, L"score", NULL, &result, obj);
等价 TJS: var value = obj.score;  // 读取 score 属性
```

</details>

### 题目 3：ncbind 注册模式识别

**问题**：根据以下反编译代码推断 TJS 接口。第5参数标志位：`0x01`=构造函数，`0x00`=方法，`0x20`=属性getter，`0x40`=属性setter。

```c
void registration() {
    void* cls = TJSCreateNativeClass(L"AudioMixer");
    TJSNativeClassRegisterNCM(cls, L"AudioMixer", fA, "s", 0x01);
    TJSNativeClassRegisterNCM(cls, L"play",       fB, "s", 0x00);
    TJSNativeClassRegisterNCM(cls, L"stop",       fC, "s", 0x00);
    TJSNativeClassRegisterNCM(cls, L"volume",     fD, "s", 0x20);
    TJSNativeClassRegisterNCM(cls, L"volume",     fE, "s", 0x40);
    TJSNativeClassRegisterNCM(cls, L"isPlaying",  fF, "s", 0x20);
}
```

<details>
<summary>查看答案</summary>

```
逐行分析:
  L"AudioMixer", 0x01 → 构造函数
  L"play",       0x00 → 方法
  L"stop",       0x00 → 方法
  L"volume",     0x20 → 属性 getter
  L"volume",     0x40 → 属性 setter (与上一行组成读写属性)
  L"isPlaying",  0x20 → 只读属性

还原 ncbind 代码:
NCB_REGISTER_CLASS(NativeAudioMixer) {
    NCB_CONSTRUCTOR(());
    NCB_METHOD(play);
    NCB_METHOD(stop);
    NCB_PROPERTY(volume, GetVolume, SetVolume);
    NCB_PROPERTY_RO(isPlaying, IsPlaying);
}
```
